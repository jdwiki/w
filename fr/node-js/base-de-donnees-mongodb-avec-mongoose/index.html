Connexion Mangouste #

Assurez-vous d’avoir mongodb en cours d’exécution en premier ! mongod --dbpath data/

package.json

"dépendances": {
    "mongoose": "^4.5.5",
}

server.js (ECMA 6)

importer la mangouste de 'mongoose' ;

mongoose.connect('mongodb://localhost:27017/stackoverflow-example');
const db = mangouste.connexion;
db.on('error', console.error.bind(console, 'DB connection error!'));

server.js (ECMA 5.1)

var mangouste = require('mangouste');

mongoose.connect('mongodb://localhost:27017/stackoverflow-example');
var db = mangouste.connexion ;
db.on('error', console.error.bind(console, 'DB connection error!'));

Model #

Define your model(s):

app/models/user.js (ECMA 6)

importer la mangouste de 'mongoose' ;

const userSchema = new mongoose.Schema({
    name: String,
    password: String
});

const Utilisateur = mongoose.model('Utilisateur', userSchema);

exporter l'utilisateur par défaut ;

app/model/user.js (ECMA 5.1)

var mangouste = require('mangouste');

var userSchema = new mongoose.Schema({
    name: String,
    password: String
});

var Utilisateur = mongoose.model('Utilisateur', userSchema);

module.exports=Utilisateur

Insert data #

ECMA 6:

utilisateur const = nouvel utilisateur ({
nom : 'Pile',
mot de passe : 'Débordement',
   }) ;

user.save((err) => {
    if (err) throw err;

    console.log('User saved!');
});

ECMA5.1:

var utilisateur = nouvel utilisateur ({
nom : 'Pile',
mot de passe : 'Débordement',
   }) ;

user.save(fonction (erreur) {
    if (err) throw err;

    console.log('User saved!');
});

Read data #

ECMA6:

Utilisateur.findOne({
    name: 'stack'
}, (err, utilisateur) => {
    if (err) throw err;

    if (!user) {
        console.log('No user was found');
    } else {
        console.log('User was found');
    }
});

ECMA5.1:

Utilisateur.findOne({
    name: 'stack'
}, fonction (erreur, utilisateur) {
    if (err) throw err;

    if (!user) {
        console.log('No user was found');
    } else {
        console.log('User was found');
    }
});
