Mesurez et vérifiez vos hypothèses de performance #

Cet exemple est écrit avec F # à l’esprit mais les idées sont applicables dans tous environnements

La première règle lors de l’optimisation des performances est de ne pas se fier à l’hypothèse ; Mesurez et vérifiez toujours vos hypothèses.

Comme nous n’écrivons pas de code machine directement, il est difficile de prédire comment le compilateur et JIT:er transforme votre programme en code machine. C’est pourquoi nous devons mesurer le temps d’exécution pour voir que nous obtenons l’amélioration des performances que nous attendons et Vérifiez que le programme réel ne contient aucun surcoût caché.

La vérification est le processus assez simple qui implique l’ingénierie inverse du exécutable en utilisant par exemple des outils comme ILSpy. Le JIT:er complique la vérification en ce sens que voir le code machine réel est délicat mais faisable. Cependant, l’examen du “code IL” donne généralement de gros gains.

Le problème le plus difficile est la mesure ; plus difficile parce que c’est difficile à configurer réaliste situations qui permettent de mesurer les améliorations du code. Toujours mesurer est inestimable.

Analyse de fonctions F# simples

Examinons quelques fonctions F# simples qui accumulent tous les entiers dans 1..n écrit de différentes manières. Comme la gamme est une simple série arithmétique le résultat peut être calculé directement, mais pour les besoins de cet exemple, nous itérer sur la plage.

Nous définissons d’abord quelques fonctions utiles pour mesurer le temps que prend une fonction :

// now () returns current time in milliseconds since start
let now : unit -> int64 =
  let sw = System.Diagnostics.Stopwatch ()
  sw.Start ()
  fun () -> sw.ElapsedMilliseconds

// time estimates the time 'action' repeated a number of times
let time repeat action : int64*'T =
  let v = action ()  // Warm-up and compute value

  let b = now ()
  for i = 1 to repeat do
    action () |> ignore
  let e = now ()

  e - b, v

time exécute une action à plusieurs reprises, nous devons exécuter les tests pendant quelques centaines millisecondes pour réduire la variance.

Ensuite, nous définissons quelques fonctions qui accumulent tous les entiers dans 1..n dans différentes façons.

// Accumulates all integers 1..n using 'List'
let accumulateUsingList n =
  List.init (n + 1) id
  |> List.sum

// Accumulates all integers 1..n using 'Seq'
let accumulateUsingSeq n =
  Seq.init (n + 1) id
  |> Seq.sum

// Accumulates all integers 1..n using 'for-expression'
let accumulateUsingFor n =
  let mutable sum = 0
  for i = 1 to n do
    sum <- sum + i
  sum

// Accumulates all integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach n =
  let mutable sum = 0
  for i in 1..n do
    sum <- sum + i
  sum

// Accumulates all integers 1..n using 'foreach-expression' over list range
let accumulateUsingForEachOverList n =
  let mutable sum = 0
  for i in [1..n] do
    sum <- sum + i
  sum

// Accumulates every second integer 1..n using 'foreach-expression' over range
let accumulateUsingForEachStep2 n =
  let mutable sum = 0
  for i in 1..2..n do
    sum <- sum + i
  sum

// Accumulates all 64 bit integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach64 n =
  let mutable sum = 0L
  for i in 1L..int64 n do
    sum <- sum + i
  sum |> int

// Accumulates all integers n..1 using 'for-expression' in reverse order
let accumulateUsingReverseFor n =
  let mutable sum = 0
  for i = n downto 1 do
    sum <- sum + i
  sum

// Accumulates all 64 integers n..1 using 'tail-recursion' in reverse order
let accumulateUsingReverseTailRecursion n =
  let rec loop sum i =
    if i > 0 then
      loop (sum + i) (i - 1)
    else
      sum
  loop 0 n

Nous supposons que le résultat est le même (sauf pour l’une des fonctions qui utilise incrément de ‘2’) mais y a-t-il une différence de performances. Pour mesurer cela, le fonction suivante est définie :

let testRun (path : string) =
  use testResult = new System.IO.StreamWriter (path)
  let write   (l : string)  = testResult.WriteLine l
  let writef  fmt           = FSharp.Core.Printf.kprintf write fmt

  write "Name\tTotal\tOuter\tInner\tCC0\tCC1\tCC2\tTime\tResult"

  // total is the total number of iterations being executed
  let total   = 10000000
  // outers let us variate the relation between the inner and outer loop
  //  this is often useful when the algorithm allocates different amount of memory
  //  depending on the input size. This can affect cache locality
  let outers  = [| 1000; 10000; 100000 |]
  for outer in outers do
    let inner = total / outer

    // multiplier is used to increase resolution of certain tests that are significantly
    //  faster than the slower ones

    let testCases =
      [|
    //   Name of test                         multiplier    action
        "List"                              , 1           , accumulateUsingList
        "Seq"                               , 1           , accumulateUsingSeq
        "for-expression"                    , 100         , accumulateUsingFor
        "foreach-expression"                , 100         , accumulateUsingForEach
        "foreach-expression over List"      , 1           , accumulateUsingForEachOverList
        "foreach-expression increment of 2" , 1           , accumulateUsingForEachStep2
        "foreach-expression over 64 bit"    , 1           , accumulateUsingForEach64
        "reverse for-expression"            , 100         , accumulateUsingReverseFor
        "reverse tail-recursion"            , 100         , accumulateUsingReverseTailRecursion
      |]
    for name, multiplier, a in testCases do
      System.GC.Collect (2, System.GCCollectionMode.Forced, true)
      let cc g = System.GC.CollectionCount g

      printfn "Accumulate using %s with outer=%d and inner=%d ..." name outer inner

      // Collect collection counters before test run
      let pcc0, pcc1, pcc2 = cc 0, cc 1, cc 2

      let ms, result       = time (outer*multiplier) (fun () -> a inner)
      let ms               = (float ms / float multiplier)

      // Collect collection counters after test run
      let acc0, acc1, acc2 = cc 0, cc 1, cc 2
      let cc0, cc1, cc2    = acc0 - pcc0, acc1 - pcc1, acc1 - pcc1
      printfn "  ... took: %f ms, GC collection count %d,%d,%d and produced %A" ms cc0 cc1 cc2 result

      writef "%s\t%d\t%d\t%d\t%d\t%d\t%d\t%f\t%d" name total outer inner cc0 cc1 cc2 ms result

[![Résultats des tests sur .NET 4.5.2 x64][1]][1]

Nous constatons une différence spectaculaire et certains des résultats sont étonnamment mauvais.

Regardons les mauvais cas :

Liste

// Accumulates all integers 1..n using 'List'
let accumulateUsingList n =
  List.init (n + 1) id
  |> List.sum

Ce qui se passe ici, c’est qu’une liste complète contenant tous les entiers 1..n est créée et réduit à l’aide d’une somme. Cela devrait être plus coûteux que de simplement itérer et s’accumulant sur la plage, il semble environ ~ 42 fois plus lent que la boucle for.

De plus, nous pouvons voir que le GC a fonctionné environ 100 fois pendant le test car le code a alloué beaucoup d’objets. Cela coûte également du CPU.

Suite

// Accumulates all integers 1..n using 'Seq'
let accumulateUsingSeq n =
  Seq.init (n + 1) id
  |> Seq.sum

La version Seq n’alloue pas une List complète, il est donc un peu surprenant que ceci ~ 270x plus lent que la boucle for. De plus, nous voyons que le GC a exécuté 661x.

Seq est inefficace lorsque la quantité de travail par élément est très petite (dans ce cas agrégeant deux nombres entiers).

Le but est de ne jamais utiliser Seq. Le but est de Mesurer.

(manofstick edit : Seq.init est le coupable de ce grave problème de performances. Il est beaucoup plus efficace d’utiliser l’expression { 0 .. n } au lieu de Seq.init (n+1) id". Cela deviendra beaucoup plus efficace encore lorsque [ce PR](https://github.com/Microsoft/visualfsharp/pull/2587) sera fusionné et publié. Même après la publication, le Seq.init . .. |> Seq.sumsera toujours lent, mais quelque peu contre-intuitif,Seq.init … |> Seq.map id |> Seq.sumsera assez rapide. Ceci afin de maintenir la rétrocompatibilité avec L'implémentation deSeq.init, qui ne calcule pas Currentinitialement, mais les enveloppe plutôt dans un objetLazy` - bien que cela aussi devrait fonctionner un peu mieux en raison de [ce PR](https://github. com/dotnet/coreclr/pull/8963). Note à l’éditeur : désolé, ce sont des notes décousues, mais je ne veux pas que les gens soient rebutés par Seq alors que l’amélioration est imminente… Quand ce moment arrivera il serait bon de mettre à jour les graphiques qui se trouvent sur cette page. )

foreach-expression sur la liste

// Accumulates all integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach n =
  let mutable sum = 0
  for i in 1..n do
    sum <- sum + i
  sum

// Accumulates all integers 1..n using 'foreach-expression' over list range
let accumulateUsingForEachOverList n =
  let mutable sum = 0
  for i in [1..n] do
    sum <- sum + i
  sum

La différence entre ces deux fonctions est très subtile mais la performance différence n’est pas, environ ~ 76x. Pourquoi? Inversons l’ingénierie du mauvais code :

public static int accumulateUsingForEach(int n)
{
  int sum = 0;
  int i = 1;
  if (n >= i)
  {
    do
    {
      sum += i;
      i++;
    }
    while (i != n + 1);
  }
  return sum;
}

public static int accumulateUsingForEachOverList(int n)
{
  int sum = 0;
  FSharpList<int> fSharpList = SeqModule.ToList<int>(Operators.CreateSequence<int>(Operators.OperatorIntrinsics.RangeInt32(1, 1, n)));
  for (FSharpList<int> tailOrNull = fSharpList.TailOrNull; tailOrNull != null; tailOrNull = fSharpList.TailOrNull)
  {
    int i = fSharpList.HeadOrDefault;
    sum += i;
    fSharpList = tailOrNull;
  }
  return sum;
}

accumulateUsingForEach est implémenté comme une boucle while efficace mais for i in [1..n] est converti en :

FSharpList<int> fSharpList =
  SeqModule.ToList<int>(
    Operators.CreateSequence<int>(
      Operators.OperatorIntrinsics.RangeInt32(1, 1, n)));

Cela signifie que nous créons d’abord un Seq sur 1..n et appelons finalement ToList.

Chere.

incrément d’expression foreach de 2

// Accumulates all integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach n =
  let mutable sum = 0
  for i in 1..n do
    sum <- sum + i
  sum

// Accumulates every second integer 1..n using 'foreach-expression' over range
let accumulateUsingForEachStep2 n =
  let mutable sum = 0
  for i in 1..2..n do
    sum <- sum + i
  sum

Encore une fois la différence entre ces deux fonctions est subtile mais les performances la différence est brutale : ~25x

Exécutons à nouveau ILSpy :

public static int accumulateUsingForEachStep2(int n)
{
  int sum = 0;
  IEnumerable<int> enumerable = Operators.OperatorIntrinsics.RangeInt32(1, 2, n);
  foreach (int i in enumerable)
  {
    sum += i;
  }
  return sum;
}

Un Seq est créé sur 1..2..n puis nous itérons sur Seq en utilisant le recenseur.

Nous nous attendions à ce que F# crée quelque chose comme ceci :

public static int accumulateUsingForEachStep2(int n)
{
  int sum = 0;
  for (int i = 1; i < n; i += 2)
  {
    sum += i;
  }
  return sum;
}

Cependant, le compilateur F # ne prend en charge que les boucles for efficaces sur les plages int32 qui incrémenter de un. Pour tous les autres cas, il retombe sur Operators.OperatorIntrinsics.RangeInt32. Ce qui expliquera la prochaine surprise résultat

foreach-expression sur 64 bits

// Accumulates all 64 bit integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach64 n =
  let mutable sum = 0L
  for i in 1L..int64 n do
    sum <- sum + i
  sum |> int

Cela s’exécute ~ 47 fois plus lentement que la boucle for, la seule différence est que nous itérons sur des entiers de 64 bits. ILSpy nous montre pourquoi :

public static int accumulateUsingForEach64(int n)
{
  long sum = 0L;
  IEnumerable<long> enumerable = Operators.OperatorIntrinsics.RangeInt64(1L, 1L, (long)n);
  foreach (long i in enumerable)
  {
    sum += i;
  }
  return (int)sum;
}

F # ne prend en charge que les boucles efficaces pour les nombres int32, il doit utiliser le repli Operators.OperatorIntrinsics.RangeInt64.

Les autres cas fonctionnent à peu près de la même manière :

[![Résultats des tests sur .NET 4.5.2 x64][2]][2]

La raison pour laquelle les performances se dégradent pour les exécutions de test plus importantes est que la surcharge de l’invocation de “l’action” se développe alors que nous faisons de moins en moins de travail dans “l’action”.

Boucler vers ‘0’ peut parfois améliorer les performances car cela peut économiser un Registre CPU mais dans ce cas, le CPU a des registres à revendre donc il ne semble pas pour faire la difference.

Conclusion

Mesurer est important car sinon nous pourrions penser que toutes ces alternatives sont équivalents, mais certaines alternatives sont ~ 270 fois plus lentes que d’autres.

L’étape de vérification implique la rétro-ingénierie de l’exécutable qui nous aide expliquez pourquoi nous avons obtenu ou non les performances attendues. En outre, La vérification peut nous aider à prédire les performances dans les cas où il est trop difficile de faire une mesure appropriée.

Il est difficile de prédire les performances là-bas. Mesurez toujours, vérifiez toujours votre hypothèses de rendement.

Utilisation de la récursivité terminale pour une itération efficace #

Venant de langages impératifs, de nombreux développeurs se demandent comment écrire une boucle for qui se termine plus tôt que F # ne prend pas en charge break, continue ou return. La réponse en F # est d’utiliser [tail-recursion] (http://stackoverflow.com/questions/33923/what-is-tail-recursion) qui est un flexible et une manière idiomatique d’itérer tout en offrant d’excellentes performances.

Disons que nous voulons implémenter tryFind pour List. Si F# prenait en charge return, nous écririons tryFind un peu comme ceci :

let tryFind predicate vs =
  for v in vs do
    if predicate v then
      return Some v
  None

Cela ne fonctionne pas en F#. Au lieu de cela, nous écrivons la fonction en utilisant tail-recursion :

let tryFind predicate vs =
  let rec loop = function
    | v::vs -> if predicate v then 
                   Some v 
               else 
                   loop vs
    | _ -> None
  loop vs

La récursivité terminale est performante dans F# car lorsque le compilateur F# détecte qu’une fonction est récursive terminale, elle réécrit la récursivité dans une boucle while efficace. En utilisant ILSpy, nous pouvons voir que cela est vrai pour notre fonction loop :

internal static FSharpOption<a> loop@3-10<a>(FSharpFunc<a, bool> predicate, FSharpList<a> _arg1)
{
  while (_arg1.TailOrNull != null)
  {
    FSharpList<a> fSharpList = _arg1;
    FSharpList<a> vs = fSharpList.TailOrNull;
    a v = fSharpList.HeadOrDefault;
    if (predicate.Invoke(v))
    {
      return FSharpOption<a>.Some(v);
    }
    FSharpFunc<a, bool> arg_2D_0 = predicate;
    _arg1 = vs;
    predicate = arg_2D_0;
  }
  return null;
}

Mis à part quelques affectations inutiles (qui, espérons-le, le JIT-er élimine), il s’agit essentiellement d’une boucle efficace.

De plus, la récursivité de queue est idiomatique pour F # car elle nous permet d’éviter l’état mutable. Considérez une fonction sum qui additionne tous les éléments d’une List. Un premier essai évident serait celui-ci :

let sum vs =
  let mutable s = LanguagePrimitives.GenericZero
  for v in vs do
    s <- s + v
  s

Si nous réécrivons la boucle en récursion terminale, nous pouvons éviter l’état mutable :

let sum vs =
  let rec loop s = function
    | v::vs -> loop (s + v) vs
    | _ -> s
  loop LanguagePrimitives.GenericZero vs

Pour plus d’efficacité, le compilateur F# transforme cela en une while-loop qui utilise un état mutable.

Comparaison de différents pipelines de données F# #

Dans F#, il existe de nombreuses options pour créer des pipelines de données, par exemple : Liste, Seq et Tableau.

Quel pipeline de données est préférable du point de vue de l’utilisation de la mémoire et des performances ?

Afin de répondre à cette question, nous comparerons les performances et l’utilisation de la mémoire à l’aide de différentes canalisations.

** Pipeline de données **

Afin de mesurer les frais généraux, nous utiliserons un pipeline de données à faible coût en processeur par envois traités :

let seqTest n =
  Seq.init (n + 1) id
  |> Seq.map    int64
  |> Seq.filter (fun v -> v % 2L = 0L)
  |> Seq.map    ((+) 1L)
  |> Seq.sum

Nous allons créer des pipelines équivalents pour toutes les alternatives et les comparer.

Nous allons faire varier la taille de n mais laissez le nombre total de travail être le même.

Alternatives au pipeline de données

Nous allons comparer les alternatives suivantes :

  1. Code impératif
  2. Tableau (non paresseux)
  3. Liste (non paresseux)
  4. LINQ (flux d’extraction paresseux)
  5. Seq (flux de tirage paresseux)
  6. Nessos (flux pull/push paresseux)
  7. PullStream (flux d’extraction simpliste)
  8. PushStream (flux push simpliste)

Bien qu’il ne s’agisse pas d’un pipeline de données, nous comparerons avec le code “impératif” car cela correspondent le plus à la façon dont le processeur exécute le code. Cela devrait être le plus rapide possible façon de calculer le résultat nous permettant de mesurer la surcharge de performance des pipelines de données.

Array et List calculent un Array/List complet à chaque étape, nous nous attendons donc à surcharge de mémoire.

LINQ et Seq sont tous deux basés sur IEnumerable<'T> qui est un flux d’extraction paresseux (pull signifie que le flux consommateur extrait les données du flux producteur). Nous attendez-vous donc à ce que les performances et l’utilisation de la mémoire soient identiques.

Nessos est une bibliothèque de flux hautes performances qui prend en charge à la fois le push et le pull (comme Java Stream).

PullStream et PushStream sont des implémentations simplistes des flux “Pull” et “Push”.

Résultats de performances sur : F# 4.0 - .NET 4.6.1 - x64

[![Résultats de performances après exécution sur : F# 4.0 - .NET 4.6.1 - x64][1]][1]

Les barres indiquent le temps écoulé, plus bas c’est mieux. Le montant total de le travail utile est le même pour tous les tests, de sorte que les résultats sont comparables. Cela signifie également que peu d’exécutions impliquent des ensembles de données plus volumineux.

Comme d’habitude lors de la mesure on voit des résultats intéressants.

  1. Les performances médiocres de la liste sont comparées à d’autres alternatives pour les grands ensembles de données. Cela peut être dû à GC ou à une mauvaise localisation du cache. 2. Performances de la “matrice” meilleures que prévu.
  2. LINQ fonctionne mieux que Seq, ce qui est inattendu car les deux sont basés sur IEnumerable<'T>. Cependant, “Seq” est basé en interne sur une implémentation générique pour tous les algorithmes, tandis que “LINQ” utilise des algorithmes spécialisés.
  3. Push fonctionne mieux que Pull. Ceci est normal puisque le pipeline de données push a moins de vérifications
  4. Les pipelines de données simplistes “Push” fonctionnent de manière comparable à “Nessos”. Cependant, Nessos prend en charge la traction et le parallélisme.
  5. Pour les petits pipelines de données, les performances de “Nessos” se dégradent en raison de la surcharge de configuration des pipelines.
  6. Comme prévu, le code “Impératif” a donné les meilleurs résultats

** Nombre de collections GC depuis l’exécution : F# 4.0 - .NET 4.6.1 - x64 **

[![GC Collection count from running on : F# 4.0 - .NET 4.6.1 - x64][2]][2]

Les barres indiquent le nombre total de comptages de collecte « GC » pendant le test, plus le nombre est bas, mieux c’est. Il s’agit d’une mesure du nombre d’objets créés par le pipeline de données.

Comme d’habitude lors de la mesure on voit des résultats intéressants.

  1. List est censé créer plus d’objets que Array car une List est essentiellement une seule liste liée de nœuds. Un tableau est une zone de mémoire continue.
  2. En regardant les nombres sous-jacents, List & Array force les collections de 2 générations. Ce genre de collecte coûte cher.
  3. Seq déclenche une quantité surprenante de collectes. C’est étonnamment encore pire que “Liste” à cet égard.
  4. LINQ, Nessos, Push et Pull ne déclenchent aucune collecte pendant quelques exécutions. Cependant, les objets sont alloués de sorte que le GC devra éventuellement s’exécuter.
  5. Comme prévu puisque le code Imperative n’alloue aucun objet, aucune collection GC n’a été déclenchée.

Conclusion

Tous les pipelines de données effectuent la même quantité de travail utile dans tous les cas de test, mais nous voyons des différences significatives de performances et d’utilisation de la mémoire entre les différents pipelines.

De plus, nous remarquons que l’overhead des pipelines de données diffère selon la taille des données traitées. Par exemple, pour les petites tailles, Array fonctionne assez bien.

Il faut garder à l’esprit la quantité de travail effectuée à chaque étape du pipeline est très petit pour mesurer le surcoût. Dans des situations “réelles”, le surcoût de Seq peut ne pas avoir d’importance car le travail réel prend plus de temps.

Les différences d’utilisation de la mémoire sont plus préoccupantes. GC n’est pas gratuit et il est bénéfique pour les applications de longue durée pour maintenir la pression GC basse.

Pour les développeurs F# soucieux des performances et de l’utilisation de la mémoire, il est recommandé de vérifier sur Flux Nessos.

Si vous avez besoin de performances de premier ordre, le code “impératif” stratégiquement placé mérite d’être considéré.

Enfin, lorsqu’il s’agit de performances, ne faites pas d’hypothèses. Mesurer et vérifier.

module PushStream =
  type Receiver<'T> = 'T -> bool
  type Stream<'T>   = Receiver<'T> -> unit

  let inline filter (f : 'T -> bool) (s : Stream<'T>) : Stream<'T> =
    fun r -> s (fun v -> if f v then r v else true)

  let inline map (m : 'T -> 'U) (s : Stream<'T>) : Stream<'U> =
    fun r -> s (fun v -> r (m v))

  let inline range b e : Stream<int> =
    fun r ->
      let rec loop i = if i <= e && r i then loop (i + 1)
      loop b

  let inline sum (s : Stream<'T>) : 'T =
    let mutable state = LanguagePrimitives.GenericZero<'T>
    s (fun v -> state <- state + v; true)
    state

module PullStream =

  [<Struct>]
  [<NoComparison>]
  [<NoEqualityAttribute>]
  type Maybe<'T>(v : 'T, hasValue : bool) =
    member    x.Value        = v
    member    x.HasValue     = hasValue
    override  x.ToString ()  =
      if hasValue then
        sprintf "Just %A" v
      else
        "Nothing"

  let Nothing<'T>     = Maybe<'T> (Unchecked.defaultof<'T>, false)
  let inline Just v   = Maybe<'T> (v, true)

  type Iterator<'T> = unit -> Maybe<'T>
  type Stream<'T>   = unit -> Iterator<'T>

  let filter (f : 'T -> bool) (s : Stream<'T>) : Stream<'T> =
    fun () ->
      let i = s ()
      let rec pop () =
        let mv = i ()
        if mv.HasValue then
          let v = mv.Value
          if f v then Just v else pop ()
        else
          Nothing
      pop

  let map (m : 'T -> 'U) (s : Stream<'T>) : Stream<'U> =
    fun () ->
      let i = s ()
      let pop () =
        let mv = i ()
        if mv.HasValue then
          Just (m mv.Value)
        else
          Nothing
      pop

  let range b e : Stream<int> =
    fun () ->
      let mutable i = b
      fun () ->
        if i <= e then
          let p = i
          i <- i + 1
          Just p
        else
          Nothing

  let inline sum (s : Stream<'T>) : 'T =
    let i = s ()
    let rec loop state =
      let mv = i ()
      if mv.HasValue then
        loop (state + mv.Value)
      else
        state
    loop LanguagePrimitives.GenericZero<'T>

module PerfTest =

  open System.Linq
#if USE_NESSOS
  open Nessos.Streams
#endif

  let now =
    let sw = System.Diagnostics.Stopwatch ()
    sw.Start ()
    fun () -> sw.ElapsedMilliseconds

  let time n a =
    let inline cc i       = System.GC.CollectionCount i

    let v                 = a ()

    System.GC.Collect (2, System.GCCollectionMode.Forced, true)

    let bcc0, bcc1, bcc2  = cc 0, cc 1, cc 2
    let b                 = now ()

    for i in 1..n do
      a () |> ignore

    let e = now ()
    let ecc0, ecc1, ecc2  = cc 0, cc 1, cc 2

    v, (e - b), ecc0 - bcc0, ecc1 - bcc1, ecc2 - bcc2

  let arrayTest n =
    Array.init (n + 1) id
    |> Array.map    int64
    |> Array.filter (fun v -> v % 2L = 0L)
    |> Array.map    ((+) 1L)
    |> Array.sum

  let imperativeTest n =
    let rec loop s i =
      if i >= 0L then
        if i % 2L = 0L then
          loop (s + i + 1L) (i - 1L)
        else
          loop s (i - 1L)
      else
        s
    loop 0L (int64 n)

  let linqTest n =
    (((Enumerable.Range(0, n + 1)).Select int64).Where(fun v -> v % 2L = 0L)).Select((+) 1L).Sum()

  let listTest n =
    List.init (n + 1) id
    |> List.map     int64
    |> List.filter  (fun v -> v % 2L = 0L)
    |> List.map     ((+) 1L)
    |> List.sum

#if USE_NESSOS
  let nessosTest n =
    Stream.initInfinite id
    |> Stream.take    (n + 1)
    |> Stream.map     int64
    |> Stream.filter  (fun v -> v % 2L = 0L)
    |> Stream.map     ((+) 1L)
    |> Stream.sum
#endif

  let pullTest n =
    PullStream.range 0 n
    |> PullStream.map     int64
    |> PullStream.filter  (fun v -> v % 2L = 0L)
    |> PullStream.map     ((+) 1L)
    |> PullStream.sum

  let pushTest n =
    PushStream.range 0 n
    |> PushStream.map     int64
    |> PushStream.filter  (fun v -> v % 2L = 0L)
    |> PushStream.map     ((+) 1L)
    |> PushStream.sum

  let seqTest n =
    Seq.init (n + 1) id
    |> Seq.map    int64
    |> Seq.filter (fun v -> v % 2L = 0L)
    |> Seq.map    ((+) 1L)
    |> Seq.sum

  let perfTest (path : string) =
    let testCases =
      [|
        "array"       , arrayTest       
        "imperative"  , imperativeTest  
        "linq"        , linqTest        
        "list"        , listTest        
        "seq"         , seqTest         
#if USE_NESSOS
        "nessos"      , nessosTest      
#endif
        "pull"        , pullTest        
        "push"        , pushTest        
      |]
    use out                   = new System.IO.StreamWriter (path)
    let write (msg : string)  = out.WriteLine msg
    let writef fmt            = FSharp.Core.Printf.kprintf write fmt

    write "Name\tTotal\tOuter\tInner\tElapsed\tCC0\tCC1\tCC2\tResult"

    let total   = 10000000
    let outers = [| 10; 1000; 1000000 |]
    for outer in outers do
      let inner = total / outer
      for name, a in testCases do
        printfn "Running %s with total=%d, outer=%d, inner=%d ..." name total outer inner
        let v, ms, cc0, cc1, cc2 = time outer (fun () -> a inner)
        printfn "  ... %d ms, cc0=%d, cc1=%d, cc2=%d, result=%A" ms cc0 cc1 cc2 v
        writef "%s\t%d\t%d\t%d\t%d\t%d\t%d\t%d\t%d" name total outer inner ms cc0 cc1 cc2 v

[<EntryPoint>]
let main argv =
  System.Environment.CurrentDirectory <- System.AppDomain.CurrentDomain.BaseDirectory
  PerfTest.perfTest "perf.tsv"
  0
