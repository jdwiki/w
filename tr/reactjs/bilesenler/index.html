import React, { Component } from 'react';
import { render } from 'react-dom';

class FirstComponent extends Component {
    render() {
        return (
            <div>
                Hello, {this.props.name}! I am a FirstComponent.
            </div>
        );
    }
}

    <FirstComponent name={ 'User' } />,
    document.getElementById('content')
);

Böyle bir durumda, bazı kişiler ES6 ok işlevlerine dayalı Durumsuz İşlevsel Bileşenleri kullanmayı tercih eder. Ok_işlevleri).

Durumsuz İşlevsel Bileşenler #

Birçok uygulamada, durumu tutan, ancak yalnızca sahne alan ve HTML’yi JSX olarak döndüren aptal bileşenler oluşturan akıllı bileşenler vardır. Durum bilgisi olmayan işlevsel bileşenler çok daha fazla yeniden kullanılabilir ve uygulamanız üzerinde olumlu bir performans etkisine sahiptir.

2 ana özelliği vardır:

  1. İşlendiğinde, aktarılan tüm aksesuarlarla birlikte bir nesne alırlar.
  2. Oluşturulacak JSX’i iade etmeleri gerekir
// When using JSX inside a module you must import React
import React from 'react';
import PropTypes from 'prop-types';

const FirstComponent = props => (
    <div>
        Hello, {props.name}! I am a FirstComponent.
    </div>
);

//arrow components also may have props validation
FirstComponent.propTypes = {
    name: PropTypes.string.isRequired,
}

// To use FirstComponent in another file it must be exposed through an export call:
export default FirstComponent;

Durum Bilgisi Bileşenler #

Yukarıda gösterilen ‘durumsuz’ bileşenlerin aksine, ‘durumlu’ bileşenler, ‘setState’ yöntemiyle güncellenebilen bir durum nesnesine sahiptir. Durumun ayarlanabilmesi için “yapıcı” içinde başlatılması gerekir:

import React, { Component } from 'react';

class SecondComponent extends Component {
    constructor(props) {
        super(props);

        this.state = {
            toggle: true
        };

        // This is to bind context when passing onClick as a callback
        this.onClick = this.onClick.bind(this);
    }

    onClick() {
        this.setState((prevState, props) => ({
            toggle: !prevState.toggle
        }));
    }
    
    render() {
        return (
            <div onClick={this.onClick}>
                Hello, {this.props.name}! I am a SecondComponent.
                <br />
                Toggle is: {this.state.toggle}
            </div>
        );
    }
}

Bir bileşeni ‘Component’ yerine ‘PureComponent’ ile genişletmek, sığ prop ve durum karşılaştırmasıyla ‘shouldComponentUpdate()’ yaşam döngüsü yöntemini otomatik olarak uygular. Bu, oluşan gereksiz işleme miktarını azaltarak uygulamanızın daha performanslı olmasını sağlar. Bu, bileşenlerinizin ‘Saf’ olduğunu varsayar ve her zaman aynı çıktıyı aynı durum ve sahne girdisiyle oluşturur.

Daha Yüksek Dereceli Bileşenler #

Daha yüksek dereceli bileşenler (HOC), bileşen işlevselliğini paylaşmaya izin verir.

import React, { Component } from 'react';

const PrintHello = ComposedComponent => class extends Component {
    onClick() {
        console.log('hello');
    }
    
    /* The higher order component takes another component as a parameter 
    and then renders it with additional props */
    render() {
        return <ComposedComponent {...this.props } onClick={this.onClick} />
    }
}

const FirstComponent = props => (
    <div onClick={ props.onClick }>
        Hello, {props.name}! I am a FirstComponent.
    </div>
);

const ExtendedComponent = PrintHello(FirstComponent);

Daha yüksek dereceli bileşenler, ne kadar farklı işlediklerine bakılmaksızın mantığı birkaç bileşen arasında paylaşmak istediğinizde kullanılır.

Yuvalama Bileşenleri #

ReactJS’nin gücünün çoğu, bileşenlerin yuvalanmasına izin verme yeteneğidir. Aşağıdaki iki bileşeni alın:

var React = require('react');
var createReactClass = require('create-react-class');

var CommentList = reactCreateClass({
  render: function() {
    return (
      <div className="commentList">
        Hello, world! I am a CommentList.
      </div>
    );
  }
});

var CommentForm = reactCreateClass({
  render: function() {
    return (
      <div className="commentForm">
        Hello, world! I am a CommentForm.
      </div>
    );
  }
});

Farklı bir bileşenin tanımında bu bileşenleri yuvalayabilir ve bunlara başvurabilirsiniz:

var React = require('react');
var createReactClass = require('create-react-class');

var CommentBox = reactCreateClass({
  render: function() {
    return (
      <div className="commentBox">
        <h1>Comments</h1>
        <CommentList /> // Which was defined above and can be reused
        <CommentForm /> // Same here
      </div>
    );
  }
});

Daha fazla yuvalama, hepsinin kullanılacak kendi yerleri olan üç şekilde yapılabilir.

1. Çocuk kullanmadan yuvalama #

(yukarıdan devam)

var CommentList = reactCreateClass({
  render: function() {
    return (
      <div className="commentList">
        <ListTitle/>
        Hello, world! I am a CommentList.
      </div>
    );
  }
});

Bu, A’nın B’yi ve B’nin C’yi oluşturduğu stildir.

Artıları #

  • UI öğelerini ayırmak kolay ve hızlı
  • Ana bileşenin durumuna göre çocuklara sahne malzemeleri enjekte etmek kolaydır

Eksileri #

  • Kompozisyon mimarisine daha az görünürlük
  • Daha az yeniden kullanılabilirlik

İyiyse #

  • B ve C sadece sunum bileşenleridir
  • B, C’nin yaşam döngüsünden sorumlu olmalıdır

2. Çocukları kullanarak yuvalama #

(yukarıdan devam)

var CommentBox = reactCreateClass({
  render: function() {
    return (
      <div className="commentBox">
        <h1>Comments</h1>
        <CommentList>
            <ListTitle/> // child
        </CommentList>
        <CommentForm />
      </div>
    );
  }
});

Bu, A’nın B’yi oluşturduğu ve A’nın B’ye C’yi oluşturmasını söylediği stildir. Ana bileşenlere daha fazla güç.

Artıları #

  • Daha iyi bileşen yaşam döngüsü yönetimi
  • Kompozisyon mimarisine daha iyi görünürlük
  • Daha iyi yeniden kullanılabilirlik

Eksileri #

  • Enjeksiyon malzemeleri biraz pahalı olabilir
  • Alt bileşenlerde daha az esneklik ve güç

İyiyse #

  • B, gelecekte veya başka bir yerde C’den farklı bir şey bestelemeyi kabul etmelidir.
  • A, C’nin yaşam döngüsünü kontrol etmelidir

B, “this.props.children” kullanarak C’yi oluşturur ve B’nin bu çocukların ne için olduğunu bilmesinin yapılandırılmış bir yolu yoktur. Bu nedenle, B, alt bileşenleri aşağı ek destek vererek zenginleştirebilir, ancak B’nin tam olarak ne olduklarını bilmesi gerekiyorsa, #3 daha iyi bir seçenek olabilir.

Artıları #

  • Özellik olarak kompozisyon
  • Kolay doğrulama
  • Daha iyi birleştirilebilirlik

Eksileri #

  • Enjeksiyon malzemeleri biraz pahalı olabilir
  • Alt bileşenlerde daha az esneklik ve güç

İyiyse #

  • B, bir şey oluşturmak için tanımlanmış belirli özelliklere sahiptir
  • B sadece nasıl oluşturulacağını bilmeli, ne oluşturulacağını değil

#3 genellikle bileşenlerden oluşan bir genel kitaplık oluşturmak için bir zorunluluktur, ancak aynı zamanda genel olarak birleştirilebilir bileşenler yapmak ve kompozisyon özelliklerini açıkça tanımlamak için iyi bir uygulamadır. #1, işe yarayan bir şey yapmanın en kolay ve en hızlısıdır, ancak #2 ve #3, çeşitli kullanım durumlarında belirli faydalar sağlamalıdır.

Sahne #

Proplar, bilgileri bir React bileşenine aktarmanın bir yoludur, işlevler de dahil olmak üzere herhangi bir türe sahip olabilirler - bazen geri aramalar olarak da adlandırılırlar.

JSX’te sahne özellikleri sözdizimi özniteliği ile iletilir

<MyComponent userID={123} />

MyComponent userID tanımının içinde artık props nesnesinden erişilebilir olacak

// The render function inside MyComponent
render() {
    return (
        <span>The user's ID is {this.props.userID}</span>
    )
}

Tüm “sahne” öğelerini, türlerini ve uygun olduğunda varsayılan değerlerini tanımlamak önemlidir:

// defined at the bottom of MyComponent
MyComponent.propTypes = {
    someObject: React.PropTypes.object,
    userID: React.PropTypes.number.isRequired,
    title: React.PropTypes.string
};

MyComponent.defaultProps = {
    someObject: {},
    title: 'My Default Title'
}

Bu örnekte, “someObject” desteği isteğe bağlıdır, ancak “userID” desteği gereklidir. “MyComponent"e “userID” sağlamazsanız, çalışma zamanında React motoru gerekli desteğin sağlanmadığı konusunda sizi uyaran bir konsol gösterecektir. Ancak dikkatli olun, bu uyarı yalnızca React kitaplığının geliştirme sürümünde gösterilir, üretim sürümü herhangi bir uyarı kaydetmez.

‘defaultProps’ kullanımı basitleştirmenizi sağlar

const { title = 'My Default Title' } = this.props;
console.log(title);

ile

console.log(this.props.title);

Aynı zamanda ’nesne’ ‘dizi’ ve ‘fonksiyonlar’ kullanımı için bir korumadır. Bir nesne için varsayılan bir pervane sağlamazsanız, pervane geçilmezse aşağıdakiler bir hata verir:

if (this.props.someObject.someKey)

Yukarıdaki örnekte, “this.props.someObject” “tanımsız"dır ve bu nedenle “someKey” denetimi bir hata verir ve kod bozulur. defaultProps kullanımı ile yukarıdaki kontrolü güvenle kullanabilirsiniz.

Bileşen durumları - Dinamik kullanıcı arayüzü #

Aşağıdaki davranışa sahip olmak istediğimizi varsayalım - Bir başlığımız var (h3 öğesi diyelim) ve tıkladığımızda, başlık adını değiştirebilmemiz için bir giriş kutusu olmasını istiyoruz. React, bileşen durumlarını ve if else ifadelerini kullanarak bunu oldukça basit ve sezgisel hale getirir. (Kod açıklaması aşağıdadır)

// I have used ReactBootstrap elements. But the code works with regular html elements also
var Button = ReactBootstrap.Button;
var Form = ReactBootstrap.Form;
var FormGroup = ReactBootstrap.FormGroup;
var FormControl = ReactBootstrap.FormControl;

var Comment = reactCreateClass({
  getInitialState: function() {
    return {show: false, newTitle: ''};
  },

  handleTitleSubmit: function() {
     //code to handle input box submit - for example, issue an ajax request to change name in database
  },

  handleTitleChange: function(e) {
    //code to change the name in form input box. newTitle is initialized as empty string. We need to update it with the string currently entered by user in the form
    this.setState({newTitle: e.target.value});
  },

  changeComponent: function() {
    // this toggles the show variable which is used  for dynamic UI
    this.setState({show: !this.state.show)};
  },

  render: function() {

    var clickableTitle;

    if(this.state.show) {
        clickableTitle = <Form inline onSubmit={this.handleTitleSubmit}>
                             <FormGroup controlId="formInlineTitle">
                                 <FormControl type="text" onChange={this.handleTitleChange}>
                             </FormGroup>
                         </Form>;
    } else {
        clickabletitle = <div>
                            <Button bsStyle="link" onClick={this.changeComponent}>
                                <h3> Default Text </h3>
                            </Button>
                         </div>;
    }

    return (
        <div className="comment">
            {clickableTitle}
        </div>
    );
  }
});

ReactDOM.render(
    <Comment />, document.getElementById('content')
);

Kodun ana kısmı clickableTitle değişkenidir. show durum değişkenine bağlı olarak, bir Form öğesi veya bir Düğme öğesi olabilir. React, bileşenlerin iç içe geçmesine izin verir.

Böylece, oluşturma işlevine bir {clickableTitle} öğesi ekleyebiliriz. clickableTitle değişkenini arar. ’this.state.show’ değerine bağlı olarak, ilgili öğeyi görüntüler.

Durumsuz İşlevsel Bileşenlerin Varyasyonları #

const languages = [
  'JavaScript',
  'Python',
  'Java',
  'Elm',
  'TypeScript',
  'C#',
  'F#'
]

// one liner
const Language = ({language}) => <li>{language}</li>

Language.propTypes = {
  message: React.PropTypes.string.isRequired
}

/**
* If there are more than one line.
* Please notice that round brackets are optional here,
* However it's better to use them for readability
*/
const LanguagesList = ({languages}) => {
  <ul>
    {languages.map(language => <Language language={language} />)}
  </ul>
}

LanguagesList.PropTypes = {
  languages: React.PropTypes.array.isRequired
}

/**
 * This syntax is used if there are more work beside just JSX presentation
 * For instance some data manipulations needs to be done.
 * Please notice that round brackets after return are required,
 * Otherwise return will return nothing (undefined)
 */
const LanguageSection = ({header, languages}) => {
  // do some work
  const formattedLanguages = languages.map(language => language.toUpperCase())
  return (
    <fieldset>
      <legend>{header}</legend>
      <LanguagesList languages={formattedLanguages} />
    </fieldset>
  )
}

LanguageSection.PropTypes = {
  header: React.PropTypes.string.isRequired,
  languages: React.PropTypes.array.isRequired
}

ReactDOM.render(
  <LanguageSection 
    header="Languages" 
    languages={languages} />,
  document.getElementById('app')
)

Burada bunun çalışan örneğini bulabilirsiniz.

setState tuzakları #

“setState"i eşzamansız bir bağlamda kullanırken dikkatli olmalısınız. Örneğin, bir alma isteğinin geri aramasında “setState” öğesini çağırmayı deneyebilirsiniz:

class MyClass extends React.Component {
    constructor() {
        super();

        this.state = {
            user: {}
        };
    }

    componentDidMount() {
        this.fetchUser();
    }

    fetchUser() {
        $.get('/api/users/self')
            .then((user) => {
                this.setState({user: user});
            });
    }

    render() {
        return <h1>{this.state.user}</h1>;
    }
}

Bu, sorunlara yol açabilir - “Bileşen” kaldırıldıktan sonra geri arama çağrılırsa, “this.setState” bir işlev olmaz. Bu durumda, setState kullanımınızın iptal edilebilir olduğundan emin olmalısınız.

Bu örnekte, bileşen ayrıldığında XHR talebini iptal etmek isteyebilirsiniz:

class MyClass extends React.Component {
    constructor() {
        super();

        this.state = {
            user: {},
            xhr: null
        };
    }

    componentWillUnmount() {
        let xhr = this.state.xhr;

        // Cancel the xhr request, so the callback is never called
        if (xhr && xhr.readyState != 4) {
            xhr.abort();
        }
    }

    componentDidMount() {
        this.fetchUser();
    }

    fetchUser() {
        let xhr = $.get('/api/users/self')
            .then((user) => {
                this.setState({user: user});
            });

        this.setState({xhr: xhr});
    }
}

Zaman uyumsuz yöntem bir durum olarak kaydedilir. ‘ComponentWillUnmount’ta, XHR isteğini iptal etmek de dahil olmak üzere tüm temizlemenizi gerçekleştirirsiniz.

Daha karmaşık bir şey de yapabilirsiniz. Bu örnekte, bu nesneyi argüman olarak kabul eden ve “cancel” işlevi çağrıldığında “this.setState” öğesini engelleyen bir “stateSetter” işlevi oluşturuyorum:

function stateSetter(context) {
    var cancelled = false;
    return {
        cancel: function () {
            cancelled = true;
        },
        setState(newState) {
            if (!cancelled) {
                context.setState(newState);
            }
        }
    }
}

class Component extends React.Component {
    constructor(props) {
        super(props);
        this.setter = stateSetter(this);
        this.state = {
            user: 'loading'
        };
    }
    componentWillUnmount() {
        this.setter.cancel();
    }
    componentDidMount() {
        this.fetchUser();
    }
    fetchUser() {
        $.get('/api/users/self')
            .then((user) => {
                this.setter.setState({user: user});
            });
    }
    render() {
        return <h1>{this.state.user}</h1>
    }
}

Bu işe yarar, çünkü oluşturduğumuz “setState” kapanışında “iptal” değişkeni görünür.
