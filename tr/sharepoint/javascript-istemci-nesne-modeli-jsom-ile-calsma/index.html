JavaScript İstemci Nesne Modeli (JSOM) ile Çalışma

Arka fon

JavaScript Nesne Modeli, SharePoint 2010’da tanıtıldı. Daha önce yalnızca sunucu tarafı kodu veya özel web hizmetleri aracılığıyla erişilebilen birçok nesneyi istemci tarafında gösterir.

JavaScript’i SharePoint Sayfalarına Gömme

SharePoint 2013’te JavaScript’inizi bir Komut Dosyası Düzenleyicisi web bölümüne koyabilirsiniz.

SharePoint 2010’da, katıştırılmış komut dosyanızı içeren bir HTML dosyasına bağlanmak için İçerik Düzenleyicisi web bölümünün “içerik bağlantısı” özelliğini kullanabilirsiniz.

Nesne Referansı

‘SP’ ad alanında bulunan tüm nesnelerin yapıcıları, yöntemleri ve özellikleri, SharePoint 2013 istemci nesne modeli başvurusunda burada belgelenmiştir.

SharePoint 2010 JavaScript istemci nesne modeli başvurusu burada mevcuttur.

JSOM’un Eşzamansız Programlama Modeli

JavaScript istemci nesne modelini kullanırken, kod genellikle aşağıdaki kalıbı alır:

  1. Bir “ClientContext” nesnesi edinin.
  2. Listeler, klasörler, görünümler gibi SharePoint nesne modelindeki varlıkları temsil eden nesneleri almak için ‘ClientContext’ nesnesini kullanın.
  3. Nesnelere karşı gerçekleştirilecek talimatları sıraya alın. Bu talimatlar henüz sunucuya iletilmemiştir.
  4. ‘ClientContext’e sunucudan hangi bilgileri geri almak istediğinizi söylemek için ’load’ işlevini kullanın.
  5. Kuyruğa alınmış talimatları sunucuya göndermek için ClientContext nesnesinin executeQueryAsync işlevini çağırın, başarı veya başarısızlık durumunda çalışacak iki geri arama işlevi iletin.
  6. Geri arama işlevinde, sunucudan döndürülen sonuçlarla çalışın.

Alternatifler

JSOM’nin istemci tarafı alternatifleri arasında SharePoint’in web hizmetleri, REST uç noktaları ve [.NET istemci nesne modeli](http:// bulunur. stackoverflow.com/documentation/sharepoint/2679/working-with-managed-client-side-object-model-csom).

Kitaplık adını kullanarak kitaplık içerik türlerini alma #

function getContentTypes(site_url,name_of_the_library){
    var ctx = new SP.ClientContext(site_url);
    var web = ctx.get_web();
    list = web.get_lists().getByTitle(name_of_the_library);

    // You can include any property of the SP.ContentType object (sp.js), for this example we are just getting the name
    ctx.load(list,'ContentTypes.Include(Name)');
    ctx.executeQueryAsync(onQuerySucceeded, onQueryFailed);
}

function onQuerySucceeded(sender, args) {
    // var list is the one that we used in function "getContentTypes"
    var contentTypesEnumerator = (list.get_contentTypes()).getEnumerator();

    while (contentTypesEnumerator.moveNext()) {
        var contentType = contentTypesEnumerator.get_current();
        alert(contentType.get_name());
    }
}

function onQueryFailed(sender, args) {
    alert('Request failed. ' + args.get_message() + '\n' + args.get_stackTrace());
}

CAML Sorgusu ile Liste Öğelerini Al #

Temel Örnek

Öğeleri almak üzere bir CAML sorgusu belirtmek için SP.CamlQuery nesnesinin “set_viewXml” yöntemini kullanın.

SP.SOD.executeOrDelayUntilScriptLoaded(showListItems,"core.js");

function showListItems(){
    var clientContext = new SP.ClientContext();
    var list = clientContext.get_web().get_lists().getByTitle("List Title");
    var camlQuery = new SP.CamlQuery();
    camlQuery.set_viewXml(
        "<View><Query>" +
            "<Where>" +
                "<Eq><FieldRef Name=\"Title\"/><Value Type=\"Text\">Value</Value></Eq>" + 
            "</Where>" +
            "<OrderBy><FieldRef Name=\"Modified\" Ascending=\"FALSE\"/></OrderBy>" +
        "</Query>"+
        //"<RowLimit>5000</RowLimit>" +
        "</View>");
    var items = list.getItems(camlQuery);
    clientContext.load(items);
    clientContext.executeQueryAsync(function(){
        var itemArray = [];
        var itemEnumerator = items.getEnumerator();
        while(itemEnumerator.moveNext()){
            var item = itemEnumerator.get_current();
            var id = item.get_item("ID");
            var title = item.get_item("Title");
            itemArray.push(id + ": " + title);
        }
        alert("ID: Title\n"+itemArray.join("\n"));
    },function(sender,args){alert(args.get_message());});
}

Bir CAML sorgusunun sonuçlarını sayfalama

Her sorguda sonuçların yalnızca bir alt kümesini almak için bir CAML sorgusundaki ‘RowLimit’ öğesinden yararlanabilirsiniz.

Geçerli konumu almak için bir liste öğesi koleksiyonunun “get_listItemCollectionPosition” yöntemini kullanın, ardından sonraki sonuç grubunu almak için SP.CamlQuery nesnesinin “set_listItemCollectionPosition” yönteminde bu değeri parametre olarak kullanın.

SP.SOD.executeOrDelayUntilScriptLoaded(showListItems,"sp.js");

function showListItems(){
    var itemArray = [];
    var clientContext = new SP.ClientContext();
    var list = clientContext.get_web().get_lists().getByTitle("List Title");
    var viewXml = 
        "<View><Query>" +
            "<OrderBy><FieldRef Name=\"Modified\" Ascending=\"FALSE\"/></OrderBy>" +
        "</Query>"+
          "<RowLimit>1</RowLimit>" +
        "</View>";
    var camlQuery = new SP.CamlQuery();
    camlQuery.set_viewXml(viewXml);            
    var items = list.getItems(camlQuery);
    clientContext.load(items);
    clientContext.executeQueryAsync(loadResults,showError);

    function loadResults(){
        var resultsFound = false;            
        var itemEnumerator = items.getEnumerator();
        while(itemEnumerator.moveNext()){
            var item = itemEnumerator.get_current();
            var id = item.get_item("ID");
            var title = item.get_item("Title");
            itemArray.push(id + ": " + title);
        }
        var pos = items.get_listItemCollectionPosition();// <- get position
        if(pos !== null){ // <-- position is null when no more results are returned
            if(confirm("Results so far: \nID: Title\n"+itemArray.join("\n"))){
                camlQuery = new SP.CamlQuery();
                camlQuery.set_listItemCollectionPosition(pos);// <- set position for next batch
                camlQuery.set_viewXml(viewXml);            
                items = list.getItems(camlQuery);
                clientContext.load(items);
                clientContext.executeQueryAsync(loadResults,showError);
            }
        }else{
            alert("Total Results: \nID: Title\n"+itemArray.join("\n")); // <- display when no more results
        }
    }
    function showError(sender,args){
        alert(args.get_message());
    }
}

Listedeki bir öğeyi silme #

SP.SOD.executeOrDelayUntilScriptLoaded( function(){ deleteItem(1); }, "sp.js");

function deleteItem(id){
    var clientContext = new SP.ClientContext();
    var list = clientContext.get_web().get_lists().getByTitle("List Title");
    var item = list.getItemById(id);
    item.deleteObject();
    clientContext.executeQueryAsync(function(){
        alert("Item #"+id+" deleted successfully!");
    },function(sender,args){alert(args.get_message());});
}

Öğeler veya Klasörler Oluşturma #

Liste Öğeleri Oluşturma

SP.SOD.executeOrDelayUntilScriptLoaded(createItem,"sp.js");

function createItem(){
    var clientContext = new SP.ClientContext();
    var list = clientContext.get_web().get_lists().getByTitle("List Title");
    var newItem = list.addItem();
    newItem.set_item("Title","Example Title");
    newItem.update();
    clientContext.load(newItem); // only needed to retrieve info from newly created item
    clientContext.executeQueryAsync(function(){
        var itemId = newItem.get_item("ID");
        alert("Item #"+itemId+" Created Successfully!");
    },function(sender,args){
        alert(args.get_message());
    });
}

Yukarıdaki örnek, aşağıdakileri gerçekleştirerek bir liste öğesinin oluşturulduğunu gösterir:

  1. Bir item nesnesi almak için bir liste nesnesinin ‘addItem’ yöntemini çağırın
  2. Her alan değerini istediğiniz gibi ayarlamak için, sonuçtaki liste öğesi nesnesinde ‘set_item’ yöntemini çağırın
  3. Değişikliklerin gerçekleştirileceğini belirtmek için liste öğesi nesnesindeki ‘güncelleme’ yöntemini çağırın.
  4. Kuyruğa alınan talimatları yürütmek için istemci bağlam nesnesinin ’executeQueryAsync’ yöntemini çağırın

Öğeyi oluşturmak için yeni öğe nesnesini istemci bağlamının “yükleme” yöntemine geçirmeniz gerekmediğine dikkat edin. Bu adım, yalnızca öğenin alan değerlerinden herhangi birini sunucudan almak istiyorsanız gereklidir.

Klasör Oluşturma

Klasör oluşturmak, listeye öğe eklemeye benzer. Aradaki fark, kişinin önce bir ‘ListItemCreationInformation’ nesnesi oluşturması ve ‘underlyingObjectType’ özelliğini ‘SP.FileSystemObjectType.folder’ ve ’leafName’ özelliğini yeni klasörün istenen adına ayarlaması gerektiğidir.

Nesne daha sonra klasörü oluşturmak için kitaplıktaki ‘addItem’ yönteminde bir parametre olarak iletilir.

// ...
var itemCreateInfo = new SP.ListItemCreationInformation();
itemCreateInfo.set_underlyingObjectType(SP.FileSystemObjectType.folder);
itemCreateInfo.set_leafName(folderName);
var newItem = list.addItem(itemCreateInfo);
// ...

Değişikliği uygulamak için, kitaplığa erişilen ‘ClientContext’ nesnesinin ’executeQueryAsync’ yöntemini çağırın.

Aşağıdaki tam örnek, geçerli zaman damgasını temel alan bir ada sahip bir klasör oluşturur ve ardından bu klasörü kalıcı bir iletişim kutusunda açar.

SP.SOD.executeOrDelayUntilScriptLoaded(createFolder,"sp.js");

function createFolder(){
    var now = new Date();
    var timeStamp = now.getYear() + "-" + (now.getMonth()+1) + "-" + now.getDate() 
        + "T" + now.getHours()+"_"+now.getMinutes()+" "+now.getSeconds()+"_"+now.getMilliseconds();
    var clientContext = new SP.ClientContext();
    var list = clientContext.get_web().get_lists().getByTitle("Library Title");
    var itemCreateInfo = new SP.ListItemCreationInformation();
    itemCreateInfo.set_underlyingObjectType(SP.FileSystemObjectType.folder);
    itemCreateInfo.set_leafName(timeStamp);
    var newItem = list.addItem(itemCreateInfo);
    newItem.update();
    clientContext.load(newItem);
    var rootFolder = list.get_rootFolder(); // Note: use a list's root folder to determine its server relative URL
    clientContext.load(rootFolder);
    clientContext.executeQueryAsync(function(){
        var itemId = newItem.get_item("ID");
        var name = newItem.get_item("FileLeafRef");
        SP.UI.ModalDialog.showModalDialog(
            { 
                title: "Folder \""+name+"\" (#"+itemId+") Created Successfully!", 
                url: rootFolder.get_serverRelativeUrl() + "/" + name
            }
        ); 
    },function(sender,args){alert(args.get_message());});
}

Mevcut Kullanıcı Bilgilerini Alın #

SP.SOD.executeOrDelayUntilScriptLoaded(showUserInfo,"sp.js");

function showUserInfo(){
    var clientContext = new SP.ClientContext();
    var user = clientContext.get_web().get_currentUser();
    clientContext.load(user);
    clientContext.executeQueryAsync(function(){
            var details = "ID: "+user.get_id()+"\n"+
                "Title: "+user.get_title()+"\n"+
                "Login: "+user.get_loginName()+"\n"+
                "Email: "+user.get_email();
            alert(details);
        },function(sender,args){alert(args.get_message());})
}

Kimliğe göre Liste Öğesi Alın #

SP.SOD.executeOrDelayUntilScriptLoaded(myFunction,"sp.js");

function myFunction(){
    var clientContext = new SP.ClientContext();
    var list = clientContext.get_web().get_lists().getByTitle("List Title");
    var item = list.getItemById(1); // get item with ID == 1
    clientContext.load(item);
    clientContext.executeQueryAsync(
        function(){ // onSuccess
            var title = item.get_item("Title");
            alert(title);
        },
        function(sender,args){ // onError
            alert(args.get_message());
        }
    );
}
