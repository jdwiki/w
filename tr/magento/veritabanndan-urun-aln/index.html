Ürünü sku ile alın #

$sku = 'sku-goes-here';
$product = Mage::getModel('catalog/product')->loadByAttribute('sku', $sku);

Kimliğe göre ürün alın #

$id = 1;
$product = Mage::getModel('catalog/product')->load($id);
if($product->getId()){
    //product was found
}

Özelliğe göre ürün koleksiyonu alın #

$collection = Mage::getModel('catalog/product')->getCollection();
// Using operator
$collection->addAttributeToFilter('status', array('eq' => 1)); 
// Without operator (automatically uses 'equal' operator
$collection->addAttributeToFilter('status', 1);

Ürün koleksiyonu - GİBİ sorgusu #

$collection = Mage::getModel('catalog/product')->getCollection();
$collection->addAttributeToFilter('sku', array('like' => 'UX%'));

Ürün koleksiyonu - niteliklerle #

//all attributes
$collection = Mage::getModel('catalog/product')
    ->getCollection()
    ->addAttributeToSelect('*');
//specific attributes
$collection = Mage::getModel('catalog/product')
    ->getCollection()
    ->addAttributeToSelect('name');
//certain attributes are special, such as price and images
//for images, then you can use 'getMediaGalleryImages'
$product->load('media_galley');

SKU listesinden ürün koleksiyonu alın #

$ skuList = dizi ('SKU-1', 'SKU-2', ..., 'SKU-n);

$_productCollection = Mage::getModel('catalog/product')
->getCollection()
->addAttributeToFilter('sku', array('in' => $skuList));

VEYA

$_productCollection = Mage::getResourceModel('catalog/product_collection')
->addAttributeToFilter('sku', array('in' => $skuList));

Ürün nesnesinden veri alın #

// First load a product object

$product->getSku();
$product->getName();

// Alternative method
$product->getData('sku');
$product->getData('name');

Ürün koleksiyonundan veri alın #

// First load a collection object

foreach($collection as $product) {
    
    $product->getSku();
    $product->getName();

    // Alternative method
    $product->getData('sku');
    $product->getData('name'); 
}

Ürünün doğru yüklenip yüklenmediğini kontrol edin #

$productFound = ($product->getId() !== null)

SKU’ya göre ürün kimliğini alın #

$sku = 'some-sku';
$productId = Mage::getModel('catalog/product')->getIdBySku($sku);
if($productId){
   //sku exists
}

Ürün koleksiyonunda Limit Belirle #

$collection = Mage::getModel('catalog/product')
            ->getCollection()
            ->setPageSize(20)
            ->setCurPage(1);
