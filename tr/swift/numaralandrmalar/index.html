Yapılar gibi ve sınıflardan farklı olarak, numaralandırmalar değer türleridir ve iletildiğinde başvurulmak yerine kopyalanır.

Numaralandırmalar hakkında daha fazla bilgi için bkz. Swift Programlama Dili.

Temel numaralandırmalar #

Bir enum bir dizi ilgili değer sağlar:

enum Direction {
    case up
    case down
    case left
    case right
}

enum Direction { case up, down, left, right }

Numaralandırma değerleri, tam nitelikli adlarıyla kullanılabilir, ancak tür adını çıkarılabileceği durumlarda atlayabilirsiniz:

let dir = Direction.up
let dir: Direction = Direction.up
let dir: Direction = .up

// func move(dir: Direction)...
move(Direction.up)
move(.up)

obj.dir = Direction.up
obj.dir = .up

Numaralandırma değerlerini karşılaştırmanın/çıkarmanın en temel yolu bir switch ifadesidir:

switch dir {
case .up:
    // handle the up case
case .down:
    // handle the down case
case .left:
    // handle the left case
case .right:
    // handle the right case
}

Basit numaralandırmalar otomatik olarak Hashable, Equatable olur ve dize dönüşümlerine sahiptir:

if dir == .down { ... }

let dirs: Set<Direction> = [.right, .left]

print(Direction.up)  // prints "up"
debugPrint(Direction.up)  // prints "Direction.up"

İlişkili değerlere sahip numaralandırmalar #

Numaralandırma durumları bir veya daha fazla yük (ilişkili değerler) içerebilir:

enum Action {
    case jump
    case kick
    case move(distance: Float)  // The "move" case has an associated distance
}

Enum değeri somutlaştırılırken yük sağlanmalıdır:

performAction(.jump)
performAction(.kick)
performAction(.move(distance: 3.3))
performAction(.move(distance: 0.5))

“switch” ifadesi, ilişkili değeri çıkarabilir:

switch action {
case .jump:
    ...
case .kick:
    ...
case .move(let distance):  // or case let .move(distance):
    print("Moving: \(distance)") 
}

“if case” kullanılarak tek bir vaka çıkarma işlemi yapılabilir:

if case .move(let distance) = action {
    print("Moving: \(distance)") 
}

‘Guard case’ sözdizimi, daha sonra kullanım ayıklamak için kullanılabilir:

guard case .move(let distance) = action else {
    print("Action is not move")
    return
}

İlişkili değerlere sahip numaralandırmalar varsayılan olarak “Equalable” değildir. == operatörünün uygulanması manuel olarak yapılmalıdır:

extension Action: Equatable { }

func ==(lhs: Action, rhs: Action) -> Bool {
    switch lhs {
    case .jump: if case .jump = rhs { return true }
    case .kick: if case .kick = rhs { return true }
    case .move(let lhsDistance): if case .move (let rhsDistance) = rhs { return lhsDistance == rhsDistance }
    }
    return false
}

Dolaylı yükler #

Normalde, numaralandırmalar özyinelemeli olamaz (çünkü sonsuz depolama gerektirirler):

enum Tree<T> {
    case leaf(T)
    case branch(Tree<T>, Tree<T>)  // error: recursive enum 'Tree<T>' is not marked 'indirect'
}

indirect anahtar sözcüğü, enum’un yükünü satır içi depolamak yerine bir dolaylı katmanla depolamasını sağlar. Bu anahtar kelimeyi tek bir durumda kullanabilirsiniz:

enum Tree<T> {
    case leaf(T)
    indirect case branch(Tree<T>, Tree<T>)
}

let tree = Tree.branch(.leaf(1), .branch(.leaf(2), .leaf(3)))

‘dolaylı’ ayrıca tüm numaralandırma üzerinde çalışır ve gerektiğinde herhangi bir durumu dolaylı hale getirir:

indirect enum Tree<T> {
    case leaf(T)
    case branch(Tree<T>, Tree<T>)
}

Ham ve Hash değerleri #

Yükü olmayan numaralandırmalar, herhangi bir hazır bilgi türünde ham değerlere sahip olabilir:

enum Rotation: Int {
    case up = 0
    case left = 90
    case upsideDown = 180
    case right = 270
}

Belirli bir türü olmayan numaralandırmalar rawValue özelliğini göstermez

enum Rotation {
    case up
    case right
    case down
    case left
}

let foo = Rotation.up
foo.rawValue //error

Tamsayı ham değerlerinin 0’dan başladığı ve monoton olarak arttığı varsayılır:

enum MetasyntacticVariable: Int {
    case foo  // rawValue is automatically 0
    case bar  // rawValue is automatically 1
    case baz = 7
    case quux  // rawValue is automatically 8
}

Dize ham değerleri otomatik olarak sentezlenebilir:

enum MarsMoon: String {
    case phobos  // rawValue is automatically "phobos"
    case deimos  // rawValue is automatically "deimos"
}

Ham değerli bir numaralandırma otomatik olarak RawRepresentable ile uyumludur. Bir enum değerinin karşılık gelen ham değerini .rawValue ile alabilirsiniz:

func rotate(rotation: Rotation) {
    let degrees = rotation.rawValue
    ...
}

Ayrıca init?(rawValue:) kullanarak bir ham değerden* bir numaralandırma oluşturabilirsiniz:

let rotation = Rotation(rawValue: 0)  // returns Rotation.Up
let otherRotation = Rotation(rawValue: 45)  // returns nil (there is no Rotation with rawValue 45)

if let moon = MarsMoon(rawValue: str) {
    print("Mars has a moon named \(str)")
} else {
    print("Mars doesn't have a moon named \(str)")
}

Belirli bir numaralandırmanın karma değerini almak istiyorsanız, hashValue değerine erişebilirsiniz, Hash değeri, sıfırdan başlayarak enum dizinini döndürür.

let quux = MetasyntacticVariable(rawValue: 8)// rawValue is 8
quux?.hashValue //hashValue is 3

Başlatıcılar #

Numaralandırmalar, varsayılan “init?(rawValue:)“dan daha yararlı olabilecek özel başlatma yöntemlerine sahip olabilir. Numaralandırmalar ayrıca değerleri de saklayabilir. Bu, başlangıçtaki değerleri depolamak ve bu değeri daha sonra almak için yararlı olabilir.

enum CompassDirection {
    case north(Int)
    case south(Int)
    case east(Int)
    case west(Int)

    init?(degrees: Int) {
        switch degrees {
        case 0...45:
            self = .north(degrees)
        case 46...135:
            self = .east(degrees)
        case 136...225:
            self = .south(degrees)
        case 226...315:
            self = .west(degrees)
        case 316...360:
            self = .north(degrees)
        default:
            return nil
        }
    }
    
    var value: Int = {
        switch self {
            case north(let degrees):
                return degrees
            case south(let degrees):
                return degrees
            case east(let degrees):
                return degrees
            case west(let degrees):
                return degrees
        }    
    }
}

Bu başlatıcıyı kullanarak şunu yapabiliriz:

var direction = CompassDirection(degrees: 0) // Returns CompassDirection.north
direction = CompassDirection(degrees: 90) // Returns CompassDirection.east
print(direction.value) //prints 90
direction = CompassDirection(degrees: 500) // Returns nil

Numaralandırmalar birçok özelliği sınıflar ve yapılarla paylaşır #

Swift’deki numaralandırmalar, C gibi diğer dillerdeki bazı karşılıklarından çok daha güçlüdür. Sınıflar ve yapılar ile başlatıcılar, hesaplanmış özellikler, [örnek yöntemleri][6], [protokol uygunlukları][ 7] ve [uzantılar][8].

protocol ChangesDirection {
    mutating func changeDirection()
}

enum Direction {
    
    // enumeration cases
    case up, down, left, right
    
    // initialise the enum instance with a case
    // that's in the opposite direction to another
    init(oppositeTo otherDirection: Direction) {
        self = otherDirection.opposite
    }
    
    // computed property that returns the opposite direction
    var opposite: Direction {
        switch self {
        case .up:
            return .down
        case .down:
            return .up
        case .left:
            return .right
        case .right:
            return .left
        }
    }
}

// extension to Direction that adds conformance to the ChangesDirection protocol
extension Direction: ChangesDirection {
    mutating func changeDirection() {
        self = .left
    }
}


var dir = Direction(oppositeTo: .down) // Direction.up

dir.changeDirection() // Direction.left

let opposite = dir.opposite // Direction.right

İç İçe Numaralandırmalar #

Numaralandırmaları iç içe yerleştirebilirsiniz; bu, hiyerarşik numaralandırmaları daha düzenli ve net olacak şekilde yapılandırmanıza olanak tanır.

enum Orchestra {
    enum Strings {
        case violin
        case viola
        case cello
        case doubleBasse
    }
    
    enum Keyboards {
        case piano
        case celesta
        case harp
    }
    
    enum Woodwinds {
        case flute
        case oboe
        case clarinet
        case bassoon
        case contrabassoon
    }
}

Ve bunu şu şekilde kullanabilirsiniz:

let instrment1 = Orchestra.Strings.viola
let instrment2 = Orchestra.Keyboards.piano
