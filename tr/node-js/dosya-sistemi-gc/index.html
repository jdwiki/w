Node.js’de, G/Ç gibi yoğun kaynak işlemleri zaman uyumsuz olarak gerçekleştirilir, ancak eşzamanlı bir karşılığı vardır (örneğin, bir “fs.readFile” vardır ve karşılığı “fs.readFileSync"dir). Düğüm tek iş parçacıklı olduğundan, senkronize işlemleri kullanırken dikkatli olmalısınız, çünkü bunlar tüm işlemi engeller.

Bir işlem senkronize bir işlem tarafından engellenirse, tüm yürütme döngüsü (olay döngüsü dahil) durdurulur. Bu, olaylar ve olay işleyicileri dahil olmak üzere diğer eşzamansız kodun çalışmayacağı ve programınızın tek engelleme işlemi tamamlanana kadar beklemeye devam edeceği anlamına gelir.

Hem senkron hem de asenkron işlemler için uygun kullanımlar vardır, ancak bunların doğru bir şekilde kullanılmasına dikkat edilmelidir.

Dosyalardan Eşzamansız Olarak Okuma #

Tüm dosya işlemleri için dosya sistemi modülünü kullanın:

const fs = require('fs');

Kodlama ile #

Bu örnekte, “/tmp” dizininden “merhaba.txt” dosyasını okuyun. Bu işlem arka planda tamamlanacak ve tamamlama veya başarısızlık durumunda geri arama gerçekleşir:

fs.readFile('/tmp/hello.txt', { encoding: 'utf8' }, (err, content) => {
  // If an error occurred, output it and return
  if(err) return console.error(err);

  // No error occurred, content is a string
  console.log(content);
});

Kodlama Olmadan #

Geçerli dizinden binary.txt ikili dosyasını arka planda eşzamansız olarak okuyun. ‘Kodlama’ seçeneğini ayarlamadığımızı unutmayın - bu, Node.js’nin içeriğin bir dizeye kodunu çözmesini engeller:

fs.readFile('binary', (err, binaryContent) => {
  // If an error occurred, output it and return
  if(err) return console.error(err);

  // No error occurred, content is a Buffer, output it in
  // hexadecimal representation.
  console.log(content.toString('hex'));
});

Göreli yollar #

Genel durumda, komut dosyanızın rastgele bir geçerli çalışma dizini ile çalıştırılabileceğini unutmayın. Geçerli komut dosyasına göre bir dosyayı adreslemek için “__dirname” veya “__filename” kullanın:

fs.readFile(path.resolve(__dirname, 'someFile'), (err, binaryContent) => {
  //Rest of Function
}

Dizin İçeriğini readdir veya readdirSync ile Listeleme #

const fs = require('fs');

// Read the contents of the directory /usr/local/bin asynchronously.
// The callback will be invoked once the operation has either completed
// or failed.
fs.readdir('/usr/local/bin', (err, files) => {
  // On error, show it and return
  if(err) return console.error(err);

  // files is an array containing the names of all entries
  // in the directory, excluding '.' (the directory itself)
  // and '..' (the parent directory).

  // Display directory entries
  console.log(files.join(' '));
});

Ana iş parçacığını bloke eden ve bu nedenle eşzamansız kodun aynı anda yürütülmesini önleyen ‘readdirSync’ olarak eşzamanlı bir varyant mevcuttur. Çoğu geliştirici, performansı artırmak için senkronize IO işlevlerinden kaçınır.

let files;

try {
  files = fs.readdirSync('/var/tmp');
} catch(err) {
  // An error occurred
  console.error(err);
}

Jeneratör kullanma #

const fs = require('fs');

// Iterate through all items obtained via
// 'yield' statements
// A callback is passed to the generator function because it is required by
// the 'readdir' method
function run(gen) {
  var iter = gen((err, data) => {
    if (err) { iter.throw(err); }

    return iter.next(data);
  });

  iter.next();
}

const dirPath = '/usr/local/bin';

// Execute the generator function
run(function* (resume) {
  // Emit the list of files in the directory from the generator
  var contents = yield fs.readdir(dirPath, resume);
  console.log(contents);
});

Bir dosyadan eşzamanlı okuma #

Herhangi bir dosya işlemi için dosya sistemi modülüne ihtiyacınız olacak:

const fs = require('fs');

Bir Dize Okuma #

fs.readFileSync, fs.readFile işlevine benzer şekilde davranır, ancak eşzamanlı olarak tamamlandığı için bir geri arama almaz ve bu nedenle ana iş parçacığını engeller. Çoğu node.js geliştiricisi, programın yürütülmesinde neredeyse hiç gecikmeye neden olmayacak olan eşzamansız değişkenleri tercih eder.

Bir “kodlama” seçeneği belirtilirse, bir dize döndürülür, aksi takdirde bir “Tampon” döndürülür.

// Read a string from another file synchronously
let content;
try {
  content = fs.readFileSync('sync.txt', { encoding: 'utf8' });
} catch(err) {
  // An error occurred
  console.error(err);
}

Akışları borulayarak dosyaları kopyalama #

Bu program, akış sınıfı tarafından sağlanan pipe() işleviyle okunabilir ve yazılabilir bir akış kullanarak bir dosyayı kopyalar.

// require the file system module
var fs = require('fs');

/*
    Create readable stream to file in current directory named 'node.txt'
    Use utf8 encoding 
    Read the data in 16-kilobyte chunks
*/
var readable = fs.createReadStream(__dirname + '/node.txt', { encoding: 'utf8', highWaterMark: 16 * 1024 });

// create writable stream
var writable = fs.createWriteStream(__dirname + '/nodePipe.txt');

// use pipe to copy readable to writable
readable.pipe(writable);

Bir Dosyanın veya Dizinin İzinlerini Kontrol Edin #

fs.access(), bir yolun var olup olmadığını ve kullanıcının bu yoldaki dosya veya dizine hangi izinlere sahip olduğunu belirler. fs.access sonuç döndürmez, hata döndürmezse yol mevcuttur ve kullanıcı istenen izinlere sahiptir.

İzin modları, “fs” nesnesi, “fs.constants” üzerinde bir özellik olarak mevcuttur.

  • fs.constants.F_OK - Okuma/yazma/yürütme izinlerine sahiptir (Hiçbir mod sağlanmadıysa, bu varsayılandır)
  • fs.constants.R_OK - Okuma izinleri var
  • fs.constants.W_OK - Yazma izinleri var
  • fs.constants.X_OK - Yürütme izinlerine sahiptir (Windows’ta fs.constants.F_OK ile aynı şekilde çalışır)

Eşzamansız olarak #

var fs = require('fs');
var path = '/path/to/check';

// checks execute permission
fs.access(path, fs.constants.X_OK, (err) => {
    if (err) {
        console.log("%s doesn't exist", path);
    } else {
        console.log('can execute %s', path);
    }
});
// Check if we have read/write permissions
// When specifying multiple permission modes
// each mode is separated by a pipe : `|`
fs.access(path, fs.constants.R_OK | fs.constants.W_OK, (err) => {
    if (err) {
        console.log("%s doesn't exist", path);
    } else {
        console.log('can read/write %s', path);
    }
});

eşzamanlı #

“fs.access” aynı zamanda senkronize bir “fs.accessSync” sürümüne sahiptir. fs.accessSync kullanırken onu bir dene/yakala bloğu içine almalısınız.

// Check write permission
try {
    fs.accessSync(path, fs.constants.W_OK);
    console.log('can write %s', path);
}
catch (err) {
    console.log("%s doesn't exist", path);
}

var fs = require('fs');

// Save the string "Hello world!" in a file called "hello.txt" in
// the directory "/tmp" using the default encoding (utf8).
// This operation will be completed in background and the callback
// will be called when it is either done or failed.
fs.writeFile('/tmp/hello.txt', 'Hello world!', function(err) {
  // If an error occurred, show it and return
  if(err) return console.error(err);
  // Successfully wrote to the file!
});

// Save binary data to a file called "binary.txt" in the current
// directory. Again, the operation will be completed in background.
var buffer = new Buffer([ 0x48, 0x65, 0x6c, 0x6c, 0x6f ]);
fs.writeFile('binary.txt', buffer, function(err) {
  // If an error occurred, show it and return
  if(err) return console.error(err);

});

Akışları kullanarak bir dosyayı Arabelleğe okuma #

Bir dosyadan içerik okumak, fs.readFile() yöntemini kullanarak zaten eşzamansız olsa da, bazen verileri basit bir geri çağırma yerine bir Akışta almak isteriz. Bu, bu verileri diğer konumlara aktarmamıza veya geldikleri anda işlememize ve sonunda hepsini aynı anda işlememize olanak tanır.

const fs = require('fs');

// Store file data chunks in this array
let chunks = [];
// We can use this variable to store the final data
let fileBuffer;

// Read file into stream.Readable
let fileStream = fs.createReadStream('text.txt');

// An error occurred with the stream
fileStream.once('error', (err) => {
    // Be sure to handle this properly!
    console.error(err); 
});

// File is done being read
fileStream.once('end', () => {
    // create the final data Buffer from data chunks;
    fileBuffer = Buffer.concat(chunks);
    
    // Of course, you can do anything else you need to here, like emit an event!
});

// Data is flushed from fileStream in chunks,
// this callback will be executed for each chunk
fileStream.on('data', (chunk) => {
    chunks.push(chunk); // push data chunk to array

    // We can perform actions on the partial data we have so far!
});

Mevcut bir dizini oluştururken veya kullanırken yarış koşullarından kaçınma #

Node’un eşzamansız doğası gereği, önce bir dizin oluşturmak veya kullanmak:

  1. fs.stat() ile varlığını kontrol edin, ardından
  2. Varlık denetiminin sonuçlarına bağlı olarak oluşturma veya kullanma,

klasör, kontrol zamanı ile oluşturma zamanı arasında oluşturulursa bir yarış koşuluna yol açabilir. Aşağıdaki yöntem, “fs.mkdir()” ve “fs.mkdirSync()” öğelerini, kodu “EEXIST” ise (zaten mevcutsa) istisnanın geçmesine izin veren hata yakalayıcı sarmalayıcılarda sarar. Hata, “EPERM” (izin reddedildi) gibi başka bir şeyse, yerel işlevlerin yaptığı gibi bir hatayı atın veya iletin.

fs.mkdir() ile eşzamansız sürüm

var fs = require('fs');

function mkdir (dirPath, callback) {
  fs.mkdir(dirPath, (err) => {
    callback(err && err.code !== 'EEXIST' ? err : null);
  });
}

mkdir('./existingDir', (err) => {

  if (err)
    return console.error(err.code);

  // Do something with `./existingDir` here

});

fs.mkdirSync() ile eşzamanlı sürüm

function mkdirSync (dirPath) {
  try {
    fs.mkdirSync(dirPath);
  } catch(e) {
    if ( e.code !== 'EEXIST' ) throw e;
  }
}

mkdirSync('./existing-dir');
// Do something with `./existing-dir` now

Akışları kullanarak bir dosyayı klonlama #

Bu program, dosya sistemi modülü tarafından sağlanan ‘createReadStream()’ ve ‘createWriteStream()’ işlevlerini kullanarak okunabilir ve yazılabilir akışları kullanarak bir dosyanın nasıl kopyalanabileceğini gösterir.

//Require the file System module
var fs = require('fs');

/*
  Create readable stream to file in current directory (__dirname) named 'node.txt'
  Use utf8 encoding 
  Read the data in 16-kilobyte chunks
*/
var readable = fs.createReadStream(__dirname + '/node.txt', { encoding: 'utf8', highWaterMark: 16 * 1024 });

// create writable stream
var writable = fs.createWriteStream(__dirname + '/nodeCopy.txt');

// Write each chunk of data to the writable stream
readable.on('data', function(chunk) {
    writable.write(chunk);
});

Bir metin dosyasının içeriğini değiştirme #

Örnek. ‘index.txt’ metin dosyasındaki ’e-posta’ kelimesini bir ‘ad’ ile değiştirecek ve basit RegExp ‘replace(/email/gim, ’name’)’ ile değiştirilecektir.

var fs = require('fs');
 
fs.readFile('index.txt', 'utf-8', function(err, data) {
    if (err) throw err;
 
    var newValue = data.replace(/email/gim, 'name');
 
    fs.writeFile('index.txt', newValue, 'utf-8', function(err, data) {
        if (err) throw err;
        console.log('Done!');
    })
})
