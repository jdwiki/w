Bir Veritabanına Bağlanma #

Düğüm uygulamasından bir mongo veritabanına bağlanmak için mongoose’a ihtiyacımız var.

Mongoose’u Yüklemek Uygulamanızın en başına gidin ve mongoose’u şu şekilde kurun:

npm install mongoose

Ardından veritabanına bağlanıyoruz.

var mongoose = require('mongoose');

//connect to the test database running on default mongod port of localhost  
mongoose.connect('mongodb://localhost/test');



//Connecting with custom credentials
mongoose.connect('mongodb://USER:PASSWORD@HOST:PORT/DATABASE');


//Using Pool Size to define the number of connections opening
//Also you can use a call back function for error handling
mongoose.connect('mongodb://localhost:27017/consumers', 
                 {server: { poolSize: 50 }}, 
                 function(err) {
                    if(err) {
                        console.log('error in this')
                        console.log(err);
                        // Do whatever to handle the error 
                    } else {
                        console.log('Connected to the database');
                    }
                });

Yeni Koleksiyon Oluşturma #

Mongoose ile her şey bir Şemadan türetilir. Bir şema oluşturalım.

var mongoose = require('mongoose');
        
var Schema = mongoose.Schema;

var AutoSchema = new Schema({
    name : String,
    countOf: Number,
});
// defining the document structure 

// by default the collection created in the db would be the first parameter we use (or the plural of it)    
module.exports = mongoose.model('Auto', AutoSchema); 

// we can over write it and define the collection name by specifying that in the third parameters. 
module.exports = mongoose.model('Auto', AutoSchema, 'collectionName'); 


// We can also define methods in the models. 
AutoSchema.methods.speak = function () {
  var greeting = this.name
    ? "Hello this is  " + this.name+ " and I have counts of "+ this.countOf
    : "I don't have a name";
  console.log(greeting);
}
mongoose.model('Auto', AutoSchema, 'collectionName');

Unutmayın, yukarıdaki gibi mongoose.model() ile derlenmeden önce şemaya yöntemler eklenmelidir.

Belgeleri Ekleme #

Koleksiyona yeni bir belge eklemek için şemanın bir nesnesini oluşturuyoruz.

var Auto = require('models/auto')
var autoObj = new Auto({
    name: "NewName", 
    countOf: 10
});

Aşağıdaki gibi kaydediyoruz

autoObj.save(function(err, insertedAuto) {
    if (err) return console.error(err);
    insertedAuto.speak();
    // output: Hello this is NewName and I have counts of 10
});

Bu, koleksiyona yeni bir belge ekleyecektir

Okuma #

Koleksiyondan Veri Okumak çok kolaydır. Koleksiyonun tüm verilerini alma.

var Auto = require('models/auto')
Auto.find({}, function (err, autos) {
      if (err) return console.error(err);
       // will return a json array of all the documents in the collection
      console.log(autos); 
})

Bir koşulla veri okuma

Auto.find({countOf: {$gte: 5}}, function (err, autos) {
      if (err) return console.error(err);
       // will return a json array of all the documents in the collection whose count is greater than 5
      console.log(autos); 
})

İkinci parametreyi, ihtiyacınız olan tüm alanların nesnesi olarak da belirleyebilirsiniz.

Auto.find({},{name:1}, function (err, autos) {
      if (err) return console.error(err);
       // will return a json array of name field of all the documents in the collection
      console.log(autos); 
})

Bir koleksiyonda bir belge bulma.

Auto.findOne({name:"newName"}, function (err, auto) {
      if (err) return console.error(err);
     //will return the first object of the document whose name is "newName"
      console.log(auto); 
})

id tarafından bir koleksiyonda bir belge bulma.

Auto.findById(123, function (err, auto) {
      if (err) return console.error(err);
     //will return the first json object of the document whose id is 123
      console.log(auto); 
})

Güncelleme #

Koleksiyonları ve belgeleri güncellemek için şu yöntemlerden herhangi birini kullanabiliriz:

Yöntemler #

  • Güncelleme()
  • güncellemeBir()
  • güncellemeMany()
  • birini değiştir()

Güncelleme() #

  • update() yöntemi bir veya daha fazla belgeyi değiştirir (güncelleme parametreleri)*
db.lights.update(
   { room: "Bedroom" },
   { status: "On" }
)

Bu işlem, “oda"nın Yatak odası (1. parametre) olduğu bir belge için “ışıklar” koleksiyonunu arar. Ardından, eşleşen belgelerin status özelliğini Açık (2. parametre) olarak günceller ve şuna benzeyen bir WriteResult nesnesi döndürür:

{ "nMatched" : 1, "nUpserted" : 0, "nModified" : 1 }

GüncellemeBir #

UpdateOne() yöntemi, BİR belgeyi değiştirir (güncelleme parametreleri)

db.countries.update(
   { country: "Sweden" },
   { capital: "Stockholm" }
)

Bu işlem, ‘ülke’nin İsveç (1. parametre) olduğu bir belge için ‘ülkeler’ koleksiyonunu arar. Ardından, eşleşen belge özelliği ‘capital’i Stockholm (2. parametre) olarak günceller ve şuna benzeyen bir WriteResult nesnesi döndürür:

{ "acknowledged" : true, "matchedCount" : 1, "modifiedCount" : 1 }

GüncellemeÇoğu #

UpdateMany() yöntemi, birden çok belgeyi değiştirir (güncelleme parametreleri)

db.food.updateMany(
   { sold: { $lt: 10 } },
   { $set: { sold: 55 } }
)

Bu işlem, “satılan"ı 55 olarak ayarlayarak, “satılan"ın 10’dan küçük olduğu *(1. parametre) tüm belgeleri (bir “gıda” koleksiyonundaki) günceller. Daha sonra şuna benzeyen bir WriteResult nesnesi döndürür:

{ "acknowledged" : true, "matchedCount" : a, "modifiedCount" : b }

a = Eşleşen belge sayısı b = Değiştirilen belge sayısı #

Birini Değiştir #

İlk eşleşen belgeyi değiştirir (yedek belge)

ülkeler adlı bu örnek koleksiyon 3 belge içerir:

{ "_id" : 1, "country" : "Sweden" }
{ "_id" : 2, "country" : "Norway" }
{ "_id" : 3, "country" : "Spain" }

Aşağıdaki işlem, { country: "İspanya" } belgesini { country: "Finlandiya" } belgesiyle değiştirir

db.countries.replaceOne(
   { country: "Spain" },
   { country: "Finland" }
)

Ve döner:

{ "acknowledged" : true, "matchedCount" : 1, "modifiedCount" : 1 }

Örnek koleksiyon ülkeler artık şunları içermektedir:

{ "_id" : 1, "country" : "Sweden" }
{ "_id" : 2, "country" : "Norway" }

Siliniyor #

Mongoose’da bir koleksiyondan belgelerin silinmesi aşağıdaki şekilde yapılır.

Auto.remove({_id:123}, function(err, result){
    if (err) return console.error(err);
    console.log(result); // this will specify the mongo default delete result.
});
