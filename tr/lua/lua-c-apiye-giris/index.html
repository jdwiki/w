Sözdizimi #

  • lua_State *L = lua_open(); // Yeni bir VM durumu oluşturun; Lua 5.0
  • iki_Durum *L = ikiL_newstate(); // Yeni bir VM durumu oluşturun; İki 5.1+
  • int luaL_dofile(lua_State *L, const char *dosyaadı); // Belirtilen lua_State’i kullanarak verilen dosyaadı ile bir lua betiği çalıştırın
  • geçersiz luaL_openlibs(lua_State *L); // Tüm standart kitaplıkları belirtilen lua_State’e yükleyin
  • void lua_close(lua_State *L); // VM durumunu kapatın ve içindeki kaynakları serbest bırakın
  • void lua_call(lua_State *L, int nargs, int sonuçlar); // -(nargs + 1) dizinindeki luadeğerini çağırın

Lua ayrıca Sanal Makinesine uygun bir C API sağlar. VM’nin aksine, C API arayüzü yığın tabanlıdır. Bu nedenle, verilerle kullanılması amaçlanan işlevlerin çoğu, ya sanal yığının üstüne bazı şeyler eklemek ya da ondan kaldırmaktır. Ayrıca, tüm API çağrıları yığın ve sınırlamaları içinde dikkatli bir şekilde kullanılmalıdır.

Genel olarak, Lua dilinde mevcut olan her şey onun C API’si kullanılarak yapılabilir. Ayrıca, dahili kayıt defterine doğrudan erişim, standart bellek ayırıcının veya çöp toplayıcının davranışını değiştirme gibi bazı ek işlevler de vardır.

Sağlanan Lua C API örneklerini, terminalinizde aşağıdakileri yürüterek derleyebilirsiniz:

$ gcc -Wall ./example.c -llua -ldl -lm

Lua işlevlerini çağırma #

#include <stdlib.h>

#include <lauxlib.h>
#include <lua.h>
#include <lualib.h>

int main(void)
{
    lua_State *lvm_hnd = lua_open();
    luaL_openlibs(lvm_hnd);

    /* Load a standard Lua function from global table: */
    lua_getglobal(lvm_hnd, "print");

    /* Push an argument onto Lua C API stack: */
    lua_pushstring(lvm_hnd, "Hello C API!");

    /* Call Lua function with 1 argument and 0 results: */
    lua_call(lvm_hnd, 1, 0);

    lua_close(lvm_hnd);

    return EXIT_SUCCESS;
 }

Yukarıdaki örnekte şunları yapıyoruz:

  • ilk örnekte gösterildiği gibi Lua VM’nin oluşturulması ve ayarlanması
  • global Lua tablosundan bir Lua fonksiyonunu sanal yığına almak ve itmek
  • “Merhaba C API” dizesini sanal yığına giriş argümanı olarak itmek
  • VM’ye zaten yığında bulunan bir bağımsız değişkenle bir işlevi çağırma talimatı verme
  • kapatma ve temizleme

NOT:

Unutmayın, ’lua_call()’ işlevi açar ve yığından gelen argümanlar yalnızca sonucu bırakır.

Ayrıca, bunun yerine Lua korumalı çağrı - lua_pcall() kullanmak daha güvenli olacaktır.

#include <lua.h>
#include <lauxlib.h>
#include <lualib.h>

int main(void)
{

  /* Start by creating a new VM state */
  lua_State *L = luaL_newstate();

  /* Load standard Lua libraries: */
  luaL_openlibs(L);
son sürüm 
  /* For older version of Lua use lua_open instead */
  lua_State *L = lua_open();

  /* Load standard libraries*/
  luaopen_base(L);
  luaopen_io(L);
  luaopen_math(L);
  luaopen_string(L);
  luaopen_table(L);
son sürüm

    luaL_dofile(L, "some_input_file.lua");

    /* done? Close it then and exit. */
    lua_close(L);

}

Özel API ve Lua Özelleştirme ile Gömülü Lua Yorumlayıcı #

Bir lua yorumlayıcısının C koduna nasıl gömüleceğini, C tanımlı bir işlevi Lua betiğine nasıl göstereceğinizi, bir Lua betiğini nasıl değerlendireceğinizi, Lua’dan C tanımlı bir işlevi nasıl çağıracağınızı ve C’den (ana bilgisayar) Lua tanımlı bir işlevi nasıl çağıracağınızı gösterin.

Bu örnekte, ruh halinin bir Lua betiği tarafından ayarlanmasını istiyoruz. İşte mood.lua:

-- Get version information from host
major, minor, build = hostgetversion()
print( "The host version is ", major, minor, build)
print("The Lua interpreter version is ", _VERSION)

-- Define a function for host to call
function mood( b )

    -- return a mood conditional on parameter
    if (b and major > 0) then
        return 'mood-happy'
    elseif (major == 0) then
        return 'mood-confused'
    else
        return 'mood-sad'
    end
end

Dikkat, mood() komut dosyasında çağrılmaz. Sadece ana bilgisayar uygulamasının araması için tanımlanmıştır. Ayrıca betiğin, betikte tanımlanmayan “hostgetversion()” adlı bir işlevi çağırdığına dikkat edin.

Ardından, ‘mood.lua’ kullanan bir ana bilgisayar uygulaması tanımlarız. İşte ‘hostlua.c’:

#include <stdio.h>
#include <lua.h>
#include <lualib.h>
#include <lauxlib.h>

/* 
 * define a function that returns version information to lua scripts
 */
static int hostgetversion(lua_State *l)
{
    /* Push the return values */
    lua_pushnumber(l, 0);
    lua_pushnumber(l, 99);
    lua_pushnumber(l, 32);

    /* Return the count of return values */
    return 3;
}

int main (void)
{
    lua_State *l = luaL_newstate();
    luaL_openlibs(l);

    /* register host API for script */
    lua_register(l, "hostgetversion", hostgetversion);

    /* load script */
    luaL_dofile(l, "mood.lua");

    /* call mood() provided by script */
    lua_getglobal(l, "mood");
    lua_pushboolean(l, 1);
    lua_call(l, 1, 1);

    /* print the mood */
    printf("The mood is %s\n", lua_tostring(l, -1));
    lua_pop(l, 1);

    lua_close(l);
    return 0;
}

Ve işte çıktı:

The host version is     0    99    32
Lua interpreter version is     Lua 5.2
The mood is mood-confused

‘hostlua.c’yi derledikten sonra bile, programımızın çıktısını değiştirmek için ‘mood.lua’yı değiştirmekte hala özgürüz!

Tablo manipülasyonu #

Bir tablodaki bir dizine erişmek veya bir dizini değiştirmek için tabloyu bir şekilde yığına yerleştirmeniz gerekir.
Bu örnekler için tablonuzun tbl adında global bir değişken olduğunu varsayalım.

İçeriği belirli bir dizinde alma: #

int getkey_index(lua_State *L)
{
  lua_getglobal(L, "tbl");    // this put the table in the stack
  lua_pushstring(L, "index"); // push the key to access
  lua_gettable(L, -2);        // retrieve the corresponding value; eg. tbl["index"]

  return 1;                   // return value to caller
}

Gördüğümüz gibi, tek yapmanız gereken tabloyu yığına itmek, dizini itmek ve lua_gettable’ı çağırmak. -2 argümanı, tablonun yığının tepesinden ikinci öğe olduğu anlamına gelir.
lua_gettable, metametotları tetikler. Metametotları tetiklemek istemiyorsanız, bunun yerine lua_rawget kullanın. Aynı argümanları kullanır.

İçeriği belirli bir dizinde ayarlama: #

int setkey_index(lua_State *L)
{
  // setup the stack
  lua_getglobal(L, "tbl");
  lua_pushstring(L, "index");
  lua_pushstring(L, "value");
  // finally assign the value to table; eg. tbl.index = "value"
  lua_settable(L, -3);

  return 0;
}

İçeriği almakla aynı tatbikat. Yığını itmeniz, dizini itmeniz ve ardından değeri yığının içine itmeniz gerekir. bundan sonra lua_settable’ı çağırırsınız. -3 argümanı, tablonun yığındaki konumudur. Metametotların tetiklenmesini önlemek için lua_settable yerine lua_rawset kullanın. Aynı argümanları kullanır.

İçeriği bir tablodan diğerine aktarma: #

int copy_tableindex(lua_State *L)
{
    lua_getglobal(L, "tbl1"); // (tbl1)
    lua_getglobal(L, "tbl2");// (tbl1)(tbl2)
    lua_pushstring(L, "index1");// (tbl1)(tbl2)("index1")
    lua_gettable(L, -3);// (tbl1)(tbl2)(tbl1.index1)
    lua_pushstring(L, "index2");// (tbl1)(tbl2)(tbl1.index1)("index2")
    lua_pushvalue(L, -2); // (tbl1)(tbl2)(tbl1.index1)("index2")(tbl1.index1)
    lua_settable(L, -4);// (tbl1)(tbl2)(tbl1.index1)
    lua_pop(L, 1);

    return 0;
}

Şimdi burada öğrendiklerimizi bir araya getiriyoruz. Yığın içeriğini yorumlara koyuyorum ki kaybolmayasınız.

Her iki tabloyu da yığına koyuyoruz, tablo 1’in dizinini yığına itiyoruz ve tbl1.index1' değerini alıyoruz. Gettable'daki '-3' argümanına dikkat edin. İkinci masaya değil (üstten üçüncü) ilk masaya bakıyorum. Sonra ikinci tablonun indeksini itiyoruz, tbl1.index1i yığının en üstüne kopyalıyoruz ve ardından üstten 4. öğede lua_settable` diyoruz.

Ev temizliği uğruna, üst elemanı temizledim, böylece yığında sadece iki masa kalıyor.
