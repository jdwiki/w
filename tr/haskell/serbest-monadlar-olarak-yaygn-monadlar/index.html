Boş Boş ~~ Kimlik #

verilen

data Empty a

sahibiz

data Free Empty a
     = Pure a
-- the Free constructor is impossible!

izomorf olan

data Identity a
     = Identity a

Serbest Kimlik ~~ (Nat,) ~~ Yazar Nat #

verilen

data Identity a = Identity a

sahibiz

data Free Identity a
     = Pure a
     | Free (Identity (Free Identity a))

izomorf olan

data Deferred a
     = Now a
     | Later (Deferred a)

veya eşdeğer olarak (ilk önce ilk öğeyi değerlendirmeye söz verirseniz) (Nat, a), diğer adıyla Writer Nat a, ile

data Nat = Z | S Nat
data Writer Nat a = Writer Nat a

Ücretsiz Belki ~~ MaybeT (Yazar Nat) #

verilen

data Maybe a = Just a
             | Nothing

sahibiz

data Free Maybe a
     = Pure a
     | Free (Just (Free Maybe a))
     | Free Nothing

hangi eşdeğerdir

data Hopes a
     = Confirmed a
     | Possible (Hopes a)
     | Failed

veya eşdeğeri (ilk önce ilk öğeyi değerlendirmeye söz verirseniz) (Nat, Maybe a), namı diğer MaybeT (Writer Nat) a ile

data Nat = Z | S Nat
data Writer Nat a = Writer Nat a
data MaybeT (Writer Nat) a = MaybeT (Nat, Maybe a)

Ücretsiz (Yazar w) ~~ Yazar [w] #

verilen

data Writer w a = Writer w a

sahibiz

data Free (Writer w) a
     = Pure a
     | Free (Writer w (Free (Writer w) a))

izomorf olan

data ProgLog w a
     = Done a
     | After w (ProgLog w a)

veya eşdeğer olarak, (önce günlüğü değerlendirmeye söz verirseniz), Yazar [w] a.

Serbest (Const c) ~~ Her iki c #

verilen

data Const c a = Const c

sahibiz

data Free (Const c) a
     = Pure a
     | Free (Const c)

izomorf olan

data Either c a
     = Right a
     | Left c

Ücretsiz (Okuyucu x) ~~ Okuyucu (Akış x) #

verilen

data Reader x a = Reader (x -> a)

sahibiz

data Free (Reader x) a
     = Pure a
     | Free (x -> Free (Reader x) a)

izomorf olan

data Demand x a
     = Satisfied a
     | Hungry (x -> Demand x a)

veya eşdeğer olarak Akım x -> a ile

data Stream x = Stream x (Stream x)
