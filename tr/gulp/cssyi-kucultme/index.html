gulp-clean-css ve gulp-rename kullanma #

İlk olarak, gulp, gulp-clean-css ve gulp-rename kurun. npmjs.com/package/gulp-rename) dizini yerel olarak projelendirmek için

npm install --save-dev gulp gulp-clean-css gulp-rename

Ardından gulpfile.js dosyanıza aşağıdaki minify-css görevini ekleyin.

var gulp = require('gulp');
var cleanCSS = require('gulp-clean-css');
var rename = require('gulp-rename');

gulp.task('minify-css', function() {
    return gulp.src('css/dist/dist.css')
        .pipe(cleanCSS())
        .pipe(rename('dist.min.css'))
        .pipe(gulp.dest('css/dist'));
});

gulp.task('watch', function(){
  gulp.watch('css/dist/**/*.css', ['minify-css']); 
  // Other watchers
});

gulp.task('default', ['minify-css', 'watch']);

Burada .pipe(cleanCSS()), css/dist/dist.css dosyanızın küçültülmesini yürütür ve .pipe(rename('concat.min.css')) onu dist.min.css olarak yeniden adlandırır.

Sass ve Css - Gulp ile Ön İşleme #

Yutmaya başlamadan önce node.js ve npm yüklememiz gerekiyor. Ardından gulp-sacc’ı kurun

$ npm i gulp-sass --save-dev // i = install

yudum kafa

var gulp = require('gulp');
// Requires the gulp-sass plugin
var sass = require('gulp-sass');

Yulaf Gövdesi

gulp.task('sass', function(){
  return gulp.src('app/scss/*.scss') // searching for sass files
    .pipe(sass()) // Converts Sass to CSS with gulp-sass
    .pipe(gulp.dest('app/css')) // destination of the css file
});

Gulp izle

gulp.task('watch', function(){
  gulp.watch('app/scss/**/*.scss', ['sass']); 
  // Other watchers
})
