Veritabanınızda bakım komut dosyalarını çalıştırmak genellikle gereklidir. Alanlar yeniden adlandırılır; veri yapıları değişir; desteklemek için kullandığınız özellikler kaldırılır; hizmetler taşınır. Şemanızı değiştirmek isteyebileceğiniz nedenlerin listesi oldukça sınırsızdır. Yani, ’neden’ oldukça açıklayıcıdır.

‘Nasıl’ biraz daha tanıdık değil. SQL işlevlerine alışmış kişiler için yukarıdaki veritabanı komut dosyaları garip görünecektir. Ancak hepsinin nasıl javascript’te olduğuna ve hem sunucuda hem de istemcide Meteor’da kullandığımız API’nin aynısını nasıl kullandıklarına dikkat edin. Veritabanımız, sunucumuz ve istemcimiz aracılığıyla tutarlı bir API’ye sahibiz.

Meteor mongo kabuğundan şema geçiş komutlarını çalıştırın:

# run meteor
meteor

# access the database shell in a second terminal window
meteor mongo

Koleksiyondaki Tüm Kayıtlara Sürüm Alanı Ekleme #

db.posts.find().forEach(function(doc){
    db.posts.update({_id: doc._id}, {$set:{'version':'v1.0'}}, false, true);
});

Bir Koleksiyondaki Tüm Kayıtlardan Diziyi Kaldırma #

db.posts.find().forEach(function(doc){
    if(doc.arrayOfObjects){
        // the false, true at the end refers to $upsert, and $multi, respectively   
        db.accounts.update({_id: doc._id}, {$unset: {'arrayOfObjects': "" }}, false, true);
    }
});

Koleksiyonu Yeniden Adlandır #

db.originalName.renameCollection("newName" );

Belirli Dize İçeren Alanı Bul #

Normal ifadenin gücü büyük sorumluluk getirir….

db.posts.find({'text': /.*foo.*|.*bar.*/i})

Eski Alandan Yeni Alan Oluştur #

db.posts.find().forEach(function(doc){
    if(doc.oldField){
        db.posts.update({_id: doc._id}, {$set:{'newField':doc.oldField}}, false, true);
    }
});

Nesneleri Diziden Çıkarın ve Yeni Bir Alana Yerleştirin #

db.posts.find().forEach(function(doc){
    if(doc.commenters){
        var firstCommenter = db.users.findOne({'_id': doc.commenters[0]._id });
        db.clients.update({_id: doc._id}, {$set:{'firstPost': firstCommenter }}, false, true);

        var firstCommenter = db.users.findOne({'_id': doc.commenters[doc.commenters.length - 1]._id });
        db.clients.update({_id: doc._id}, {$set:{'lastPost': object._id }}, false, true);
    }
});

Bir Koleksiyondan Başka Bir Koleksiyona Blob Kaydı (ör. Birleştirmeyi Kaldır ve Düzleştir) #

db.posts.find().forEach(function(doc){
    if(doc.commentsBlobId){
        var commentsBlob = db.comments.findOne({'_id': commentsBlobId });
        db.posts.update({_id: doc._id}, {$set:{'comments': commentsBlob }}, false, true);
    }
});

Alanın Var Olduğundan Emin Olun #

db.posts.find().forEach(function(doc){
    if(!doc.foo){
        db.posts.update({_id: doc._id}, {$set:{'foo':''}}, false, true);
    }
});

Alanın Özel Değeri Olduğundan Emin Olun #

db.posts.find().forEach(function(doc){
    if(!doc.foo){
        db.posts.update({_id: doc._id}, {$set:{'foo':'bar'}}, false, true);
    }
});

Belirli Alan Belirli Değer ise Kaydı Kaldır #

db.posts.find().forEach(function(doc){
    if(doc.foo === 'bar'){
        db.posts.remove({_id: doc._id});
    }
});

Alanın Belirli Değerini Yeni Değerle Değiştir #

db.posts.find().forEach(function(doc){
    if(doc.foo === 'bar'){
        db.posts.update({_id: doc._id}, {$set:{'foo':'squee'}}, false, true);
    }
});

Belirli Alanı Boş Olarak Ayarla #

db.posts.find().forEach(function(doc){
    if(doc.oldfield){
        // the false, true at the end refers to $upsert, and $multi, respectively
        db.accounts.update({_id: doc._id}, {$unset: {'oldfield': "" }}, false, true);
    }
});

ObjectId’yi String’e Dönüştür #

db.posts.find().forEach(function(doc){
     db.accounts.update({_id: doc._id}, {$set: {'_id': doc._id.str }}, false, true);
});

Alan Değerlerini Sayılardan Dizelere Dönüştür #

var newvalue = "";
db.posts.find().forEach(function(doc){
     if(doc.foo){
         newvalue = '"' + doc.foo + '"';
         db.accounts.update({_id: doc._id}, {$set: {'doc.foo': newvalue}});
     }
});

Alan Değerlerini Dizelerden Sayılara Dönüştür #

var newvalue = null;
db.posts.find().forEach(function(doc){
     if(doc.foo){
         newvalue = '"' + doc.foo + '"';
         db.accounts.update({_id: doc._id}, {$set: {'doc.foo': newvalue}});
     }
});

_id Alanındaki bir ObjectID’den Zaman Damgası oluşturun #

db.posts.find().forEach(function(doc){
    if(doc._id){
        db.posts.update({_id: doc._id}, {$set:{ timestamp: new Date(parseInt(doc._id.str.slice(0,8), 16) *1000) }}, false, true);
    }
});

Bir Tarih Nesnesinden bir ObjectID oluşturun #

var timestamp = Math.floor(new Date(1974, 6, 25).getTime() / 1000);
var hex       = ('00000000' + timestamp.toString(16)).substr(-8); // zero padding
var objectId  = new ObjectId(hex + new ObjectId().str.substring(8));

Bir Dizide Öğeleri Olan Tüm Kayıtları Bul #

Burada yaptığımız nokta notasyonu kullanarak dizi indeksine referans vermektir.

db.posts.find({"tags.0": {$exists: true }})
