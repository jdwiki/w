Basit bir kontrol: derecelendirme #

Şu şekilde kullanılması amaçlanan basit bir kontrol, bir derecelendirme aracı oluşturalım:

<rating min="0" max="5" nullifier="true" ng-model="data.rating"></rating>

Şimdilik süslü CSS yok; bu şu şekilde işleyecektir:

0 1 2 3 4 5 x

Bir sayıya tıklamak o derecelendirmeyi seçer; ve “x"e tıklamak, derecelendirmeyi boş olarak ayarlar.

app.directive('rating', function() {

    function RatingController() {
        this._ngModel = null;
        this.rating = null;
        this.options = null;
        this.min = typeof this.min === 'number' ? this.min : 1;
        this.max = typeof this.max === 'number' ? this.max : 5;
    }
    
    RatingController.prototype.setNgModel = function(ngModel) {
        this._ngModel = ngModel;
        
        if( ngModel ) {
            // KEY POINT 1
            ngModel.$render = this._render.bind(this);
        }
    };
    
    RatingController.prototype._render = function() {
        this.rating = this._ngModel.$viewValue != null ? this._ngModel.$viewValue : -Number.MAX_VALUE;
    };
    
    RatingController.prototype._calculateOptions = function() {
        if( this.min == null || this.max == null ) {
            this.options = [];
        }
        else {
            this.options = new Array(this.max - this.min + 1);
            for( var i=0; i < this.options.length; i++ ) {
                this.options[i] = this.min + i;
            }
        }
    };
    
    RatingController.prototype.setValue = function(val) {
        this.rating = val;
        // KEY POINT 2
        this._ngModel.$setViewValue(val);
    };
    
    // KEY POINT 3
    Object.defineProperty(RatingController.prototype, 'min', {
        get: function() {
            return this._min;
        },
        set: function(val) {
            this._min = val;
            this._calculateOptions();
        }
    });
    
    Object.defineProperty(RatingController.prototype, 'max', {
        get: function() {
            return this._max;
        },
        set: function(val) {
            this._max = val;
            this._calculateOptions();
        }
    });
    
    return {
        restrict: 'E',
        scope: {
            // KEY POINT 3
            min: '<?',
            max: '<?',
            nullifier: '<?'
        },
        bindToController: true,
        controllerAs: 'ctrl',
        controller: RatingController,
        require: ['rating', 'ngModel'],
        link: function(scope, elem, attrs, ctrls) {
            ctrls[0].setNgModel(ctrls[1]);
        },
        template:
            '<span ng-repeat="o in ctrl.options" href="#" class="rating-option" ng-class="{\'rating-option-active\': o <= ctrl.rating}" ng-click="ctrl.setValue(o)">{{ o }}</span>' +
            '<span ng-if="ctrl.nullifier" ng-click="ctrl.setValue(null)" class="rating-nullifier">&#10006;</span>'
    };
});

Anahtar noktaları:

  1. Modelin görünüm değerini kendi görünümünüze aktarmak için ngModel.$render öğesini uygulayın.
  2. Görünüm değerinin güncellenmesi gerektiğini düşündüğünüzde ngModel.$setViewValue() öğesini arayın.
  3. Kontrol elbette parametrelendirilebilir; Girişi açıkça belirtmek için Açısal >= 1.5 ise parametreler için '<' kapsam bağlamalarını kullanın - tek yönlü bağlama. Bir parametre değiştiğinde harekete geçmeniz gerekiyorsa, birkaç saat kaydetmek için bir JavaScript özelliği kullanabilirsiniz (bkz. Object.defineProperty()).

Not 1: Uygulamayı aşırı karmaşıklaştırmamak için, derecelendirme değerleri bir diziye eklenir - ‘ctrl.options’. Bu gerekli değildir; daha verimli ama aynı zamanda daha karmaşık bir uygulama, “min”/“maks” değiştiğinde derecelendirme eklemek/kaldırmak için DOM manipülasyonunu kullanabilir.

Not 2: '<' kapsam bağlamaları dışında, bu örnek Angular < 1.5 içinde kullanılabilir. Angular >= 1.5 üzerindeyseniz, bunu bir bileşene dönüştürmek ve bunu denetleyicinin yapıcısında yapmak yerine “min” ve “max"ı başlatmak için “$onInit()” yaşam döngüsü kancasını kullanmak iyi bir fikir olacaktır. .

Ve gerekli bir keman: https://jsfiddle.net/h81mgxma/

Birkaç karmaşık kontrol: tam bir nesneyi düzenleyin #

Özel bir denetim, kendisini ilkeller gibi önemsiz şeylerle sınırlamak zorunda değildir; daha ilginç şeyleri düzenleyebilir. Burada, biri kişileri düzenlemek ve biri adresleri düzenlemek için olmak üzere iki tür özel denetim sunuyoruz. Adres kontrolü, kişinin adresini düzenlemek için kullanılır. Bir kullanım örneği şöyle olacaktır:

<input-person ng-model="data.thePerson"></input-person>
<input-address ng-model="data.thePerson.address"></input-address>

Bu örnek için model kasıtlı olarak basitleştirilmiştir:

function Person(data) {
  data = data || {};
  this.name = data.name;
  this.address = data.address ? new Address(data.address) : null;
}

function Address(data) {
  data = data || {};
  this.street = data.street;
  this.number = data.number;
}

Adres düzenleyici:

app.directive('inputAddress', function() {

    InputAddressController.$inject = ['$scope'];
    function InputAddressController($scope) {
        this.$scope = $scope;
        this._ngModel = null;
        this.value = null;
        this._unwatch = angular.noop;
    }

    InputAddressController.prototype.setNgModel = function(ngModel) {
        this._ngModel = ngModel;
        
        if( ngModel ) {
            // KEY POINT 3
            ngModel.$render = this._render.bind(this);
        }
    };
    
    InputAddressController.prototype._makeWatch = function() {
        // KEY POINT 1
        this._unwatch = this.$scope.$watchCollection(
            (function() {
                return this.value;
            }).bind(this),
            (function(newval, oldval) {
                if( newval !== oldval ) { // skip the initial trigger
                    this._ngModel.$setViewValue(newval !== null ? new Address(newval) : null);
                }
            }).bind(this)
        );
    };
    
    InputAddressController.prototype._render = function() {
        // KEY POINT 2
        this._unwatch();
        this.value = this._ngModel.$viewValue ? new Address(this._ngModel.$viewValue) : null;
        this._makeWatch();
    };

    return {
        restrict: 'E',
        scope: {},
        bindToController: true,
        controllerAs: 'ctrl',
        controller: InputAddressController,
        require: ['inputAddress', 'ngModel'],
        link: function(scope, elem, attrs, ctrls) {
            ctrls[0].setNgModel(ctrls[1]);
        },
        template:
            '<div>' +
                '<label><span>Street:</span><input type="text" ng-model="ctrl.value.street" /></label>' +
                '<label><span>Number:</span><input type="text" ng-model="ctrl.value.number" /></label>' +
            '</div>'
    };
});

Anahtar noktaları:

  1. Bir nesneyi düzenliyoruz; ebeveynimizden bize verilen nesneyi doğrudan değiştirmek istemiyoruz (modelimizin değişmezlik ilkesiyle uyumlu olmasını istiyoruz). Böylece, düzenlenmekte olan nesne üzerinde sığ bir saat oluşturuyoruz ve bir özellik değiştiğinde modeli $setViewValue() ile güncelliyoruz. Bir kopyasını ebeveynimize iletiyoruz.
  2. Model dışarıdan ne zaman değişse onu kopyalıyoruz ve kopyayı kapsamımıza kaydediyoruz. Yine değişmezlik ilkeleri, dahili kopya değişmez olmasa da, harici çok iyi olabilir. Ek olarak, model tarafından bize gönderilen değişiklikler için izleyiciyi tetiklemekten kaçınmak için saati (this_unwatch();this._makeWatch();) yeniden oluşturduk. (Saatin yalnızca kullanıcı arayüzünde yapılan değişiklikler için tetiklenmesini istiyoruz.)
  3. Yukarıdaki noktaların dışında, basit bir kontrol için yaptığımız gibi ngModel.$render() uygularız ve ngModel.$setViewValue() çağırırız (derecelendirme örneğine bakın).

Kişi özel denetiminin kodu neredeyse aynıdır. Şablon <input-address> kullanıyor. Daha gelişmiş bir uygulamada, denetleyicileri yeniden kullanılabilir bir modülde ayıklayabiliriz.

app.directive('inputPerson', function() {

    InputPersonController.$inject = ['$scope'];
    function InputPersonController($scope) {
        this.$scope = $scope;
        this._ngModel = null;
        this.value = null;
        this._unwatch = angular.noop;
    }

    InputPersonController.prototype.setNgModel = function(ngModel) {
        this._ngModel = ngModel;
        
        if( ngModel ) {
            ngModel.$render = this._render.bind(this);
        }
    };
    
    InputPersonController.prototype._makeWatch = function() {
        this._unwatch = this.$scope.$watchCollection(
            (function() {
                return this.value;
            }).bind(this),
            (function(newval, oldval) {
                if( newval !== oldval ) { // skip the initial trigger
                    this._ngModel.$setViewValue(newval !== null ? new Person(newval) : null);
                }
            }).bind(this)
        );
    };
    
    InputPersonController.prototype._render = function() {
        this._unwatch();
        this.value = this._ngModel.$viewValue ? new Person(this._ngModel.$viewValue) : null;
        this._makeWatch();
    };

    return {
        restrict: 'E',
        scope: {},
        bindToController: true,
        controllerAs: 'ctrl',
        controller: InputPersonController,
        require: ['inputPerson', 'ngModel'],
        link: function(scope, elem, attrs, ctrls) {
            ctrls[0].setNgModel(ctrls[1]);
        },
        template:
            '<div>' +
                '<label><span>Name:</span><input type="text" ng-model="ctrl.value.name" /></label>' +
                '<input-address ng-model="ctrl.value.address"></input-address>' +
            '</div>'
    };
});

Not: Burada nesneler yazılır, yani uygun kurucuları vardır. Bu zorunlu değildir; model, düz JSON nesneleri olabilir. Bu durumda, kurucular yerine sadece angular.copy() kullanın. Ek bir avantaj, kontrolörün iki kontrol için aynı hale gelmesi ve bazı ortak modüllere kolaylıkla çıkarılabilmesidir.
