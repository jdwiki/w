Javascript + CSS + Yazı Tipleri + Görseller Örneği #

Gerekli modüller

npm install --save-dev webpack extract-text-webpack-plugin file-loader css-loader style-loader

klasör yapısı

.
└── assets
    ├── css
    ├── images
    └── js

webpack.config.js

const webpack = require('webpack');
const ExtractTextPlugin = require('extract-text-webpack-plugin');
const path = require('path');
const glob = require('glob');

module.exports = {
  entry: {
    script: path.resolve(__dirname, './assets/js/app.js'),
    style: path.resolve(__dirname, './assets/css/app.css'),
    images: glob.sync(path.resolve(__dirname, './assets/images/**/*.*')),
  },
  context: __dirname,
  output: {
    path: path.resolve('./dist/assets'),
    publicPath: '/dist/assets',
    filename: '[name].js',
  },
  module: {
    loaders: [
      {
        test: /\.css$/,
        loader: ExtractTextPlugin.extract({
            fallback: 'style-loader',
            use: 'css-loader'
        }),
      },
      {
        test: /(\.woff2?|\.woff|\.ttf|\.eot|\.svg)(\?v=\d+\.\d+\.\d+)?$/,
        loader: 'file-loader?name=[name]-[hash:6].[ext]',
      },
      {
        test: /\.(png|jpe?g|gif|ico)$/,
        loader: 'file-loader?name=[name].[ext]',
      },
    ],
  },
  plugins: [
    new ExtractTextPlugin('app.css' /* optional: , { allChunks: true } */),
  ],
};

glob.sync('./assets/images/**/*.*'), resimler klasöründeki tüm dosyaları giriş olarak gerektirir.

ExtractTextPlugin oluşturulan çıktıyı alır ve paketlenmiş bir css dosyası oluşturur.

Kurulum #

Ön koşullar:

NodeJS ve npm

Webpack’i kurmanın iki yolu vardır: global veya proje başına. Bağımlılığın proje başına yüklenmesi en iyisidir, çünkü bu, her proje için farklı web paketi sürümlerini kullanmanıza izin verir ve kullanıcının web paketini global olarak yüklemesini gerektirmez.

Proje başına yükleme

Projenizin kök klasöründen aşağıdaki komutu çalıştırın:

npm install webpack --save-dev

Ardından, “node_modules"a yüklenen web paketi yürütülebilir dosyasını çalıştırabilirsiniz:

./node_modules/.bin/webpack

Veya ‘package.json’ dosyanızda, ’node_modules’ bölümünü atlayabileceğiniz bir NPM komut dosyası oluşturun - npm, bu klasörü PATH’ine dahil edecek kadar akıllıdır.

// in package.json:
{
  ...
  "scripts": {
    "start": "webpack"
  },
  ...
}

// from terminal:
npm start

Küresel olarak yükleme

Bir istemde aşağıdaki komutu çalıştırın:

npm install webpack -g

babel ile webpack.config.js örneği #

bağımlılıklar

npm i -D webpack babel-loader

webpack.config.js

const path = require('path');

module.exports = {
  entry: {
    app: ['babel-polyfill', './src/'],
  },
  output: {
    path: __dirname,
    filename: './dist/[name].js',
  },
  resolve: {
    extensions: ['', '.js'],
  },
  module: {
    loaders: [{
      test: /\.js$/, 
      loaders: ['babel-loader'],
      include: path.resolve(__dirname, 'src')
    }],
  }
};

Web Paketi Basit Örnek #

Webpack’i kullanmak için gereken minimum aşağıdaki komuttur:

webpack ./src/index.js ./dist/bundle.js

// this is equivalent to:

webpack source-file destination-file

Web paketi kaynak dosyayı alır, çıktı hedefine derler ve kaynak dosyalardaki tüm bağımlılıkları çözer.

Webpack, React JSX, Babel, ES6, basit yapılandırma #

Doğru npm bağımlılıklarını kurduğunuzdan emin olun (babel kendisini bir grup pakete ayırmaya karar verdi, “eş bağımlılıkları” ile ilgili bir şey):

npm web paketini kur webpack-node-externals babel-core babel-loader babel-preset-react babel-preset-latest --save

“webpack.config.js”:

module.exports = {
    context: __dirname, // sets the relative dot (optional)
    entry: "./index.jsx",
    output: {
        filename: "./index-transpiled.js"
    },
    module: {
        loaders: [{
            test: /\.jsx$/,
            loader: "babel?presets[]=react,presets[]=latest" // avoid .babelrc
        }]
    }, // may need libraryTarget: umd if exporting as a module
    externals: [require("webpack-node-externals")()], // probably not required
    devtool: "inline-source-map"
};

“webpack-node-externals”, “node_modules"inizi tarayan ve bunların ön uç kodunuzla birlikte aktarılmamasını ve paketlenmemesini sağlayan bir işlevdir, ancak paketin bunlara referansta kalmasını sağlar. Bu, kitaplıkları yeniden kodlamadığınız için daha hızlı aktarıma yardımcı olur.

webpack.config.js

var path = require('path'); // used to get context

module.exports = {
    context: path.join(__dirname, 'app'), // resolves entry below, must be absolute path
    entry: './app.js', // entry point or loader for the application
    output: {
        path: path.join(__dirname, 'app/lib'), // express static folder is at /app/lib
        filename: '[name].bundle.js', // the file name of the bundle to create.  [name] is replaced by the name of the chunk (code-splitting)
        publicPath: 'static' // example uses express as the webserver
    }
};

web sunucusu.js

var express = require('express'),
    path = require('path'),
    bodyParser = require('body-parser'),
    multer = require('multer')()
    logger = require('morgan'),
    fs = require('fs'),
    http = require('http');

var app = express();
var port = 31416;

app.use(bodyParser.urlencoded({ extended: false }));
app.use(bodyParser.json());
app.use(logger('short'));
app.use('/jsBundles',express.static('lib'));
app.get('/', function(request, response){
    response.sendFile(__dirname + '/index.html');
});

var server = http.createServer(app).listen(port, function(){
    console.log("I feel a disturbance in the port:" + port);
});

app.js

require(['./modules/someFunctions'],function(){
        window.onload = function(){
                var someFunctions  = require('./modules/someFunctions');             
                document.getElementById('doubleIt').onclick = function(){
                        var num = document.getElementById('num').value;
                        document.getElementById('theCount').innerHTML = someFunctions.Double(num);
                };

                document.getElementById('tripleIt').onclick = function(){
                        var num = document.getElementById('num').value;
                        document.getElementById('theCount').innerHTML = someFunctions.Triple(num);
                };
        };
});

çeşitli.js

var self = {};
self.isNumber = function(value){
    // http://stackoverflow.com/questions/9716468/is-there-any-function-like-isnumeric-in-javascript-to-validate-numbers
    return !isNaN(parseFloat(value)) && isFinite(value);
};
module.exports = self;

someFunctions.js

require(['./misc'], function(){
    var misc= require('./misc');

    var self = {};
    self.Double = function(value){
        if(!misc.isNumber(value)){
            return 0;
        };
        return value*2;
    }

    self.Triple = function(value){
        if(!misc.isNumber(value)){
            return 0;
        };
        return value*3;
    }

    module.exports = self;
});
