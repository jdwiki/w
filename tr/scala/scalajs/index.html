Scala.js, sonunda “JVM"nin dışında çalışacak olan “JavaScript"i derleyen “Scala"dan bir bağlantı noktasıdır. Güçlü yazma, derleme zamanında kod optimizasyonu, JavaScript kitaplıkları ile tam birlikte çalışabilirlik gibi faydaları vardır.

Scala.js’de konsol.log #

println("Hello Scala.js") // In ES6: console.log("Hello Scala.js");

Yağ ok işlevleri #

val lastNames = people.map(p => p.lastName)
// Or shorter:
val lastNames = people.map(_.lastName)

Basit Sınıf #

class Person(val firstName: String, val lastName: String) {
  def fullName(): String =
    s"$firstName $lastName"
}

Koleksiyonlar #

val personMap = Map(
  10 -> new Person("Roger", "Moore"),
  20 -> new Person("James", "Bond")
)
val names = for {
  (key, person) <- personMap
  if key > 15
} yield s"$key = ${person.firstName}"

DOM’yi Yönetme #

import org.scalajs.dom
import dom.document

def appendP(target: dom.Node, text: String) = {
  val pNode = document.createElement("p")
  val textNode = document.createTextNode(text)
  pNode.appendChild(textNode)
  target.appendChild(pNode)
}

SBT ile kullanma #

Sbt bağımlılığı #

libraryDependencies += "org.scala-js" %%% "scalajs-dom" % "0.9.1" // (Triple %%%)

Koşma #

sbt run

Sürekli derleme ile çalıştırma: #

sbt ~run

Tek bir JavaScript dosyasına derleyin: #

sbt fastOptJS
