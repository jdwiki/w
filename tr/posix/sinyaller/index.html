Varsayılan eylemle SIGALARM’ı yükseltmek #

“Alarm"ı kullanarak, kullanıcı “SIGALARM” sinyalinin belirtilen aralıktan sonra yükseltilmesini programlayabilir. Kullanıcının bu sinyal için engellememesi, yok saymaması veya açık sinyal işleyicisi belirtmemesi durumunda, bu sinyal için varsayılan eylem varışta gerçekleştirilecektir. Spesifikasyon uyarınca SIGALARM için varsayılan eylem, işlemi sonlandırmaktır:

#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>

int main (int argc, char** argv)
{
    printf("Hello!\n");

    // Set the alarm for five second
    alarm(5); // Per POSIX, this cannot fail

    // Now sleep for 15 seconds
    for (int i = 1; i <= 15; i++)
    {
        printf("%d\n", i);
        sleep(1);
    }

    // And print the message before successful exit
    printf("Goodbye!\n");

    return EXIT_SUCCESS;
}

Sinyal işleyiciyi sigaction kullanarak ayarlama ve yükseltme kullanarak sinyalleri yükseltme #

Bir programın belirli bir sinyale tepki vermesi için, varsayılan eylemi kullanmak dışında, ‘sigaction’ kullanılarak özel sinyal işleyici kurulabilir. “sigaction” üç argüman alır - harekete geçilecek sinyal, “NULL” değilse yeni davranışı tanımlayan “sigaction_t” yapısının işaretçisi ve “NULL” değilse eski davranışla doldurulacak olan “sigaction_t” işaretçisi (böylece biri onu geri yükleyebilir). Yükseltme sinyalleri aynı işlemde ‘yükselt’ yöntemi ile yapılabilmektedir. Daha fazla kontrol gerekiyorsa (sinyali başka bir işleme göndermek için, hedef işlem kimliğini veya iş parçacığı kimliğini kabul eden ‘kill’ veya ‘pthread_kill’ kullanılabilir).

#include <unistd.h>
#include <signal.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// Signals are numbered from 1, signal 0 doesn't exist
volatile sig_atomic_t last_received_signal = 0;

// Signal handler, will set the global variable
// to indicate what is the last signal received.
// There should be as less work as possible inside
// signal handler routine, and one must take care only
// to call reentrant functions (in case of signal arriving
// while program is already executing same function)
void signal_catcher(int signo, siginfo_t *info, void *context)
{
    last_received_signal = info->si_signo;
}

int main (int argc, char** argv)
{
    // Setup a signal handler for SIGUSR1 and SIGUSR2
    struct sigaction act;
    memset(&act, 0, sizeof act);

    // sigact structure holding old configuration
    // (will be filled by sigaction):
    struct sigaction old_1;
    memset(&old_1, 0, sizeof old_1);
    struct sigaction old_2;
    memset(&old_2, 0, sizeof old_2);

    act.sa_sigaction = signal_catcher;
    // When passing sa_sigaction, SA_SIGINFO flag
    // must be specified. Otherwise, function pointed
    // by act.sa_handler will be invoked
    act.sa_flags = SA_SIGINFO;

    if (0 != sigaction(SIGUSR1, &act, &old_1))
    {
        perror("sigaction () failed installing SIGUSR1 handler");
        return EXIT_FAILURE;
    }

    if (0 != sigaction(SIGUSR2, &act, &old_2))
    {
        perror("sigaction() failed installing SIGUSR2 handler");
        return EXIT_FAILURE;
    }

    // Main body of "work" during which two signals
    // will be raised, after 5 and 10 seconds, and which
    // will print last received signal
    for (int i = 1; i <= 15; i++)
    {
        if (i == 5)
        {
            if (0 != raise(SIGUSR1))
            {
                perror("Can't raise SIGUSR1");
                return EXIT_FAILURE;
            }
        }

        if (i == 10)
        {
            if (0 != raise(SIGUSR2))
            {
                perror("Can't raise SIGUSR2");
                return EXIT_FAILURE;
            }
        }

        printf("Tick #%d, last caught signal: %d\n",
            i, last_received_signal);

        sleep(1);
    }

    // Restore old signal handlers
    if (0 != sigaction(SIGUSR1, &old_1, NULL))
    {
        perror("sigaction() failed restoring SIGUSR1 handler");
        return EXIT_FAILURE;
    }

    if (0 != sigaction(SIGUSR2, &old_2, NULL))
    {
        perror("sigaction() failed restoring SIGUSR2 handler");
        return EXIT_FAILURE;
    }

    return EXIT_SUCCESS;
}

kill() kullanarak intihar eden bir süreç #

Bir işlem, kill() işlevini kullanarak herhangi bir başka işleme bir sinyal göndermeyi deneyebilir.

Bunu yapmak için, gönderme işleminin alıcı işlemin PID’sini bilmesi gerekir. Bir ırk tanıtmadan, bir süreç yalnızca kendi PID’sinden (ve çocuklarının PID’lerinden) emin olabileceğinden, ‘kill()’ kullanımını göstermenin en basit örneği, bir sürecin kendisine bir sinyal göndermesini sağlamaktır.

Aşağıda, kendisine bir öldürme sinyali (SIGKILL) göndererek kendi sonlandırmasını başlatan bir süreç örneği verilmiştir:

#define _POSIX_C_SOURCE 1

#include <sys/types.h>
#include <unistd.h>
#include <signal.h>
#include <stdio.h>


int main (void)
{
  pid_t pid = getpid(); /* Get my iown process ID. */

  kill(pid, SIGKILL); /* Send myself a KILL signal. */

  puts("Signal delivery initiated.");  /* Although not guaranteed, 
                                  practically the program never gets here. */

  pause(); /* Wait to die. */

  puts("This never gets printed.");
}

Çıktı:

Killed

(… veya benzeri, uygulamaya bağlı olarak)

write() tarafından oluşturulan SIGPIPE’yi iş parçacığı açısından güvenli bir şekilde kullanın #

write(), okuma ucu kapalı olan adlandırılmış veya adsız bir kanal veya akış soketi için çağrıldığında, iki şey olur:

  1. SIGPIPE sinyali,write() adı verilen sürece gönderilir.
son sürüm
  1. “SIGPIPE” sinyali, “write()” olarak adlandırılan iş parçacığına gönderilir.
son sürüm
  1. ‘write()’ tarafından ‘EPIPE’ hatası döndürülür

“SIGPIPE” ile başa çıkmanın birkaç yolu vardır:

  • Soketler için, Linux’ta ‘MSG_NOSIGNAL’ ve BSD’de ‘SO_NOSIGPIPE’ gibi platforma özel seçenekler ayarlanarak ‘SIGPIPE’ devre dışı bırakılabilir (yalnızca ‘gönder’ için çalışır, ‘yazma’ için değil). Bu taşınabilir değil.

  • FIFO’lar (adlandırılmış borular) için, yazar ‘O_WRONLY’ yerine ‘O_RDWR’ kullanırsa, ‘SIGPIPE’ oluşturulmaz, böylece okuma ucu her zaman açılır. Ancak bu, ‘EPIPE’yi de devre dışı bırakır.

  • SIGPIPEı görmezden gelebilir veya global işleyiciyi ayarlayabiliriz. Bu iyi bir çözümdür, ancak tüm uygulamayı kontrol etmiyorsanız (örneğin bir kitaplık yazıyorsanız) kabul edilemez.

  • En son POSIX sürümleriyle, SIGPIPEnin write() adlı iş parçacığına gönderildiği gerçeğini kullanabilir ve senkron sinyal işleme tekniğini kullanarak işleyebiliriz.

Aşağıdaki kod, POSIX.1-2004 ve sonrası için iş parçacığı güvenli ‘SIGPIPE’ işlemeyi gösterir.

[Bu gönderiden2 esinlenilmiştir:

  • İlk olarak, pthread_sigmask() kullanarak mevcut iş parçacığının sinyal maskesine SIGPIPE ekleyin.
  • sigpending() kullanarak zaten bekleyen SIGPIPE olup olmadığını kontrol edin.
  • write() çağırın. Okuma sonu kapalıysa, bekleyen sinyaller maskesine ‘SIGPIPE’ eklenir ve ‘EPIPE’ döndürülür.
  • “write()”, “EPIPE” döndürdüyse ve “SIGPIPE”, “write()“dan önce zaten beklemede değilse, sigtimedwait() kullanarak onu bekleyen sinyaller maskesinden kaldırın.
  • pthread_sigmask() kullanarak orijinal sinyal maskesini geri yükleyin.

Kaynak kodu:

#include <unistd.h>
#include <time.h>
#include <errno.h>
#include <sys/signal.h>

ssize_t safe_write(int fd, const void* buf, size_t bufsz)
{
    sigset_t sig_block, sig_restore, sig_pending;

    sigemptyset(&sig_block);
    sigaddset(&sig_block, SIGPIPE);

    /* Block SIGPIPE for this thread.
     *
     * This works since kernel sends SIGPIPE to the thread that called write(),
     * not to the whole process.
     */
    if (pthread_sigmask(SIG_BLOCK, &sig_block, &sig_restore) != 0) {
        return -1;
    }

    /* Check if SIGPIPE is already pending.
     */
    int sigpipe_pending = -1;
    if (sigpending(&sig_pending) != -1) {
        sigpipe_pending = sigismember(&sig_pending, SIGPIPE);
    }

    if (sigpipe_pending == -1) {
        pthread_sigmask(SIG_SETMASK, &sig_restore, NULL);
        return -1;
    }

    ssize_t ret;
    while ((ret = write(fd, buf, bufsz)) == -1) {
        if (errno != EINTR)
            break;
    }

    /* Fetch generated SIGPIPE if write() failed with EPIPE.
     *
     * However, if SIGPIPE was already pending before calling write(), it was
     * also generated and blocked by caller, and caller may expect that it can
     * fetch it later. Since signals are not queued, we don't fetch it in this
     * case.
     */
    if (ret == -1 && errno == EPIPE && sigpipe_pending == 0) {
        struct timespec ts;
        ts.tv_sec = 0;
        ts.tv_nsec = 0;

        int sig;
        while ((sig = sigtimedwait(&sig_block, 0, &ts)) == -1) {
            if (errno != EINTR)
                break;
        }
    }

    pthread_sigmask(SIG_SETMASK, &sig_restore, NULL);
    return ret;
}
