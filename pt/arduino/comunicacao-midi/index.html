A intenção deste tópico é demonstrar alguns programas básicos de MIDI que mostram como operar com o protocolo e adicionar progressivamente recursos úteis que aplicativos mais complexos exigem.

Exemplo MIDI THRU #

O MIDI Thru é simples e fácil de testar. Ao trabalhar corretamente, você poderá instalar seu projeto Arduino entre dois dispositivos MIDI, MIDI IN para MIDI OUT e poderá verificar se os dois dispositivos operam juntos. Se você tiver a capacidade de medir a latência, verá um aumento devido às instruções de captura de buffer serial e retransmissão.

// This is a simple MIDI THRU.  Everything in, goes right out.
// This has been validate on an Arduino UNO and a Olimex MIDI Shield  

boolean byteReady; 
unsigned char midiByte;

void setup() {
    // put your setup code here, to run once:
    //  Set MIDI baud rate:
    Serial.begin(31250);
    byteReady = false;
    midiByte = 0;  
}

// The Loop that always gets called...
void loop() {
   if (byteReady) {
        byteReady = false;
        Serial.write(midiByte);
    }
}

// The little function that gets called each time loop is called.  
// This is automated somwhere in the Arduino code.
void serialEvent() {
  if (Serial.available()) {
    // get the new byte:
    midiByte = (unsigned char)Serial.read();
    byteReady = true;
  }
}

MIDI Thru com fila #

// This is a more complex MIDI THRU.  This version uses a queue.  Queues are important because some
// MIDI messages can be interrupted for real time events.  If you are generating your own messages,
// you may need to stop your message to let a "real time" message through and then resume your message.


#define QUEUE_DEPTH 128

// Queue Logic for storing messages
int headQ = 0;
int tailQ = 0;
unsigned char tx_queue[QUEUE_DEPTH];

void setup() {
    // put your setup code here, to run once:
    //  Set MIDI baud rate:
    Serial.begin(31250);
}

// getQDepth checks for roll over.  Folks have told me this 
// is not required.  Feel free to experiment.
int getQDepth() {
int depth = 0;
    if (headQ < tailQ) {
        depth = QUEUE_DEPTH - (tailQ - headQ);
    } else {
        depth = headQ - tailQ;
    }
    return depth;
}

void addQueue (unsigned char myByte) {
    int depth = 0;
    depth = getQDepth();

    if (depth < (QUEUE_DEPTH-2)) {
        tx_queue[headQ] = myByte;
        headQ++;
        headQ = headQ % QUEUE_DEPTH; // Always keep the headQ limited between 0 and 127
    }
}

unsigned char deQueue() {
    unsigned char myByte;
    myByte = tx_queue[tailQ];
    tailQ++;
    tailQ = tailQ % QUEUE_DEPTH;  // Keep this tailQ contained within a limit
    // Now that we dequeed the byte, it must be sent. 
    return myByte;
}

void loop() {
   if (getQDepth>0) {
        Serial.write(deQueue());
    }
}

// The little function that gets called each time loop is called.  
// This is automated somwhere in the Arduino code.
void serialEvent() {
  if (Serial.available()) {
    // get the new byte:
    addQueue((unsigned char)Serial.read());;
  }
}

Geração de relógio MIDI #

// This is a MiDI clk generator.  This takes a #defined BPM and 
// makes the appropriate clk rate.  The queue is used to let other messages 
// through, but allows a clock to go immediately to reduce clock jitter

#define QUEUE_DEPTH 128
#define BPM 121
#define MIDI_SYSRT_CLK 0xF8

// clock tracking and calculation
unsigned long lastClock;
unsigned long captClock;
unsigned long clk_period_us;

// Queue Logic for storing messages
int headQ = 0;
int tailQ = 0;
unsigned char tx_queue[QUEUE_DEPTH];

void setup() {
    //  Set MIDI baud rate:
    Serial.begin(31250);
    clk_period_us = 60000000 / (24 * BPM);
    lastClock = micros();
}

// getQDepth checks for roll over.  Folks have told me this 
// is not required.  Feel free to experiment.
int getQDepth() {
int depth = 0;
    if (headQ < tailQ) {
        depth = QUEUE_DEPTH - (tailQ - headQ);
    } else {
        depth = headQ - tailQ;
    }
    return depth;
}

void addQueue (unsigned char myByte) {
    int depth = 0;
    depth = getQDepth();

    if (depth < (QUEUE_DEPTH-2)) {
        tx_queue[headQ] = myByte;
        headQ++;
        headQ = headQ % QUEUE_DEPTH; // Always keep the headQ limited between 0 and 127
    }
}

unsigned char deQueue() {
    unsigned char myByte;
    myByte = tx_queue[tailQ];
    tailQ++;
    tailQ = tailQ % QUEUE_DEPTH;  // Keep this tailQ contained within a limit
    // Now that we dequeed the byte, it must be sent. 
    return myByte;
}

void loop() {
    captClock = micros();
    
    if (lastClock > captClock) {
        // we have a roll over condition - Again, maybe we don't need to do this.
        if (clk_period_us <= (4294967295 - (lastClock - captClock))) {
            // Add a the ideal clock period for this BPM to the last measurement value
            lastClock = lastClock + clk_period_us;
            // Send a clock, bypasing the transmit queue
            Serial.write(MIDI_SYSRT_CLK);
        }
    } else if (clk_period_us <= captClock-lastClock) {
        // Basically the same two commands above, but not within a roll over check
        lastClock = lastClock + clk_period_us;
        // Send a clock, bypasing the transmit queue
        Serial.write(MIDI_SYSRT_CLK);
    }

    if (getQDepth>0) {
        Serial.write(deQueue());
    }
}

// The little function that gets called each time loop is called.  
// This is automated somwhere in the Arduino code.
void serialEvent() {
  if (Serial.available()) {
    // get the new byte:
    addQueue((unsigned char)Serial.read());;
  }
}

Mensagens MIDI definidas #

Em geral, o protocolo MIDI é dividido em “mensagens”. Existem 4 classes gerais de mensagens:

  • Voz do canal
  • Modo de canal
  • Sistema Comum
  • Mensagens em tempo real do sistema

As mensagens começam com um valor de byte acima de 0x80. Qualquer valor abaixo de 0x7F é considerado dado. Efetivamente significando que 127 é o valor máximo que pode ser codificado em um único byte de dados MIDI. Para codificar valores maiores, são necessários dois ou mais bytes de dados MIDI.

Cabe ressaltar que as mensagens devem ser enviadas do início ao fim sem interrupção… EXCETO… Mensagens do sistema em tempo real, que são de um único byte, que podem ser injetadas no meio de qualquer mensagem.

Mensagens de voz do canal

Estado D7..D0 Bytes de dados Descrição
1000nnnn 0kkkkkkk 0vvvvvvv Evento Note Off. Esta mensagem é enviada quando uma nota é liberada (terminada). (kkkkkkk) é o número chave (nota). (vvvvvvv) é a velocidade.
1001nnnn 0kkkkkkk 0vvvvvvvv Nota No evento. Esta mensagem é enviada quando uma nota é pressionada (iniciar). (kkkkkkk) é o número chave (nota). (vvvvvvv) é a velocidade.
1010nnnn 0kkkkkkk 0vvvvvvvv Pressão de Tecla Polifônica (Aftertouch). Esta mensagem é mais frequentemente enviada pressionando a tecla depois que ela “acaba”. (kkkkkkk) é o número chave (nota). (vvvvvvv) é o valor da pressão.
1011nnnn 0ccccccc 0vvvvvvvv Mudança de controle. Esta mensagem é enviada quando o valor de um controlador é alterado. Os controladores incluem dispositivos como pedais e alavancas. Os números de controlador 120-127 são reservados como “Mensagens do modo de canal” (abaixo). (cccccccc) é o número do controlador (0-119). (vvvvvvv) é o valor do controlador (0-127).
1100nnnn 0pppppppp Mudança de programa. Esta mensagem é enviada quando o número do patch muda. (ppppppp) é o novo número do programa.
1101nnnn 0vvvvvvv Pressão do canal (pós-toque). Esta mensagem é mais frequentemente enviada pressionando a tecla depois que ela “acaba”. Esta mensagem é diferente do pós-toque polifônico. Use esta mensagem para enviar o maior valor de pressão único (de todas as teclas pressionadas atualmente). (vvvvvvv) é o valor da pressão.
1110nnnn 0llllllll 0mmmmmmmm Mudança de Pitch Bend. Esta mensagem é enviada para indicar uma mudança no pitch bender (roda ou alavanca, normalmente). O pitch bender é medido por um valor de quatorze bits. Centro (sem mudança de tom) é 2000H. A sensibilidade é uma função do receptor, mas pode ser definida usando RPN 0. (llllllll) são os 7 bits menos significativos. (mmmmmmmm) são os 7 bits mais significativos.

Mensagens do modo de canal

Estado D7..D0 Bytes de dados Descrição
1011nnnn 0ccccccc 0vvvvvvvv Mensagens do modo de canal. Este é o mesmo código que o Control Change (acima), mas implementa o controle de modo e a mensagem especial usando os números de controlador reservados 120-127. Os comandos são:
Todos os sons desligados. Quando All Sound Off for recebido, todos os osciladores serão desligados e seus envelopes de volume serão zerados o mais rápido possível. c = 120, v = 0: Todos os sons desligados
Redefinir todos os controladores. Quando Reset All Controllers é recebido, todos os valores do controlador são redefinidos para seus valores padrão. (Consulte as práticas recomendadas específicas para padrões).
c = 121, v = x: O valor deve ser apenas zero, a menos que permitido de outra forma em uma prática recomendada específica.
Controle local. Quando o controle local está desativado, todos os dispositivos em um determinado canal responderão apenas aos dados recebidos por MIDI. Os dados reproduzidos, etc. serão ignorados. Local Control On restaura as funções dos controladores normais.
c = 122, v = 0: Controle Local Desligado
c = 122, v = 127: controle local ativado
Todas as notas desligadas. Quando um All Notes Off é recebido, todos os osciladores serão desligados.
c = 123, v = 0: Todas as notas desligadas (Consulte o texto para descrição dos comandos do modo real.)
c = 124, v = 0: Modo Omni desativado
c = 125, v = 0: Modo Omni ativado
c = 126, v = M: Modo Mono On (Poly Off) onde M é o número de canais (Omni Off) ou 0 (Omni On) ​​
c = 127, v = 0: Poly Mode On (Mono Off) (Observação: Essas quatro mensagens também fazem com que todas as notas sejam desativadas)

Mensagens comuns do sistema

Estado D7..D0 Bytes de dados Descrição
11110000 0iiiiiii [0iiiiiii 0iiiiiii] 0ddddddd — — 0dddddddd 11110111 Exclusivo do sistema. Esse tipo de mensagem permite que os fabricantes criem suas próprias mensagens (como dumps em massa, parâmetros de patch e outros dados não especificados) e fornece um mecanismo para criar mensagens de especificação MIDI adicionais. O código de identificação do fabricante (atribuído por MMA ou AMEI) é de 1 byte (0iiiiiii) ou 3 bytes (0iiiiiii 0iiiiiii 0iiiiiii). Dois dos IDs de 1 byte são reservados para extensões chamadas Mensagens Exclusivas Universais, que não são específicas do fabricante. Se um dispositivo reconhecer o código de identificação como seu próprio (ou como uma mensagem universal compatível), ele ouvirá o restante da mensagem (0ddddddd). Caso contrário, a mensagem será ignorada. (Observação: somente mensagens em tempo real podem ser intercaladas com um System Exclusive.)
11110001 0nnndddd Quadro de trimestre de código de tempo MIDI. nnn = Tipo de Mensagem dddd = Valores
11110010 0llllllll 0mmmmmmmm Ponteiro de posição da música. Este é um registro interno de 14 bits que contém o número de batidas MIDI (1 batida = seis relógios MIDI) desde o início da música. l é o LSB, m o MSB.
11110011 0ssssss Seleção de música. O Song Select especifica qual sequência ou música deve ser tocada.
11110100 Indefinido. (Reservado)
11110101 Indefinido. (Reservado)
11110110 Pedido de Sintonia. Ao receber um Tune Request, todos os sintetizadores analógicos devem sintonizar seus osciladores.
11110111 Fim do Exclusivo. Usado para encerrar um dump exclusivo do sistema (veja acima).

Mensagens em tempo real do sistema

Estado D7..D0 Bytes de dados Descrição
11111000 Relógio de tempo. Enviado 24 vezes por semínima quando a sincronização é necessária (ver texto).
11111001 Indefinido. (Reservado)
11111010 Começar. Inicie a reprodução da sequência atual. (Esta mensagem será seguida de Relógios de tempo).
11111011 Continuar. Continue no ponto em que a sequência foi interrompida.
11111100 Pare. Pare a sequência atual.
11111101 Indefinido. (Reservado)
11111110 Sensoriamento Ativo. Essa mensagem deve ser enviada repetidamente para informar ao receptor que uma conexão está ativa. O uso desta mensagem é opcional. Quando recebido inicialmente, o receptor espera receber outra mensagem Active Sensing a cada 300ms (máximo), e se não receber, assumirá que a conexão foi encerrada. Ao terminar, o receptor desligará todas as vozes e retornará à operação normal (detecção não ativa).
11111111 Redefinir. Redefina todos os receptores no sistema para o status de inicialização. Isso deve ser usado com moderação, de preferência sob controle manual. Em particular, não deve ser enviado na inicialização.
