Lendo elementos de array em JSON #

– Crie um JSON de amostra com ARRAY

create table car_sample(dim_id integer, info varchar(2000)); inserir nos valores car_sample (200, ‘{“cars”: [ { “Fabricante”: “Nissan”, “Models”: [{“Nome”:“Sentra”, “portas”:4}, {“Nome”:" Maxima", “portas”:4} ]}, {“Fabricante”: “Ford”, “Modelos”: [{“Nome”:“Touro”, “portas”:4}, {“Nome”:“Escolta” , “portas”:4} ]} ] }’)

– UDF para extrair o array JSON em uma string delimitada por ^
FUNÇÃO CRIAR OU SUBSTITUIR f_extractJson (jsonVar varchar) RETORNA varchar IMUTÁVEL como $$
 def myfunc(myParm):
  importar json
  carros=json.loads(jsonVar)
  parsedString=’’
  para carro em carros[“cars”]:
   para modelo no carro[“Models”]:
    parsedString=parsedString+car[“Manufacturer”]+’~’+model[“Name”]+’^’
  retorna String analisada

 retornar minhafunc(jsonVar)
$$ IDIOMA plpythonu;

– Verifique os dados

selecione dim_id, f_extractJson(info) de car_sample;

– Linhas de pivô

WITH w1 AS (selecione dim_id, f_extractJson(info) info de car_sample)
select dim_id,number, info, split_part(split_part(info,’^’,number),’~’, 1)
Fabricante, split_part(split_part(info,’^’,number),’~’, 2) Modelo
de w1 cross join series1_10
onde número <= regexp_count(info,’[=^=]’) ;
