Os mecanismos de comunicação entre processos (IPC) permitem que diferentes processos independentes se comuniquem entre si. O Padrão C não fornece nenhum mecanismo IPC. Portanto, todos esses mecanismos são definidos pelo sistema operacional do host. O POSIX define um extenso conjunto de mecanismos IPC; O Windows define outro conjunto; e outros sistemas definem suas próprias variantes.

Semáforos #

Semáforos são usados ​​para sincronizar operações entre dois ou mais processos. POSIX define dois conjuntos diferentes de funções de semáforo:

  1. ‘System V IPC’ — semctl(), semop(), semget().
  2. ‘Semáforos POSIX’ — sem_close(), sem_destroy(), sem_getvalue(), sem_init(), sem_open(), sem_post(), sem_trywait(), sem_unlink().

Esta seção descreve os semáforos IPC do System V, assim chamados porque se originaram com o Unix System V.

Primeiro, você precisará incluir os cabeçalhos necessários. Versões antigas do POSIX requeriam #include <sys/types.h>; POSIX moderno e a maioria dos sistemas não o exigem.

#include <sys/sem.h>

Em seguida, você precisará definir uma chave tanto no pai quanto no filho.

#define KEY 0x1111

Essa chave precisa ser a mesma em ambos os programas ou eles não farão referência à mesma estrutura IPC. Existem maneiras de gerar uma chave acordada sem codificar seu valor.

Em seguida, dependendo do seu compilador, você pode ou não precisar fazer esta etapa: declarar uma união para fins de operações de semáforo.

union semun {
    int val;
    struct semid_ds *buf;
    unsigned short  *array;
};

Em seguida, defina suas estruturas try (semwait) e raise (semsignal). Os nomes P e V são originários de holandês

struct sembuf p = { 0, -1, SEM_UNDO}; # semwait
struct sembuf v = { 0, +1, SEM_UNDO}; # semsignal

Agora, comece obtendo o id do seu semáforo IPC.

int id;
// 2nd argument is number of semaphores
// 3rd argument is the mode (IPC_CREAT creates the semaphore set if needed)
if ((id = semget(KEY, 1, 0666 | IPC_CREAT) < 0) {
    /* error handling code */
}

No pai, inicialize o semáforo para ter um contador de 1.

union semun u;
u.val = 1;
if (semctl(id, 0, SETVAL, u) < 0) { // SETVAL is a macro to specify that you're setting the value of the semaphore to that specified by the union u
    /* error handling code */
}

Agora, você pode decrementar ou incrementar o semáforo conforme necessário. No início de sua seção crítica, você diminui o contador usando a função semop():

if (semop(id, &p, 1) < 0) {
    /* error handling code */
}

Para incrementar o semáforo, você usa &v em vez de &p:

if (semop(id, &v, 1) < 0) {
    /* error handling code */
}

Observe que cada função retorna 0 em caso de sucesso e -1 em caso de falha. Não verificar esses status de retorno pode causar problemas devastadores.

Exemplo 1.1: Corrida com Threads #

O programa abaixo terá um processo fork um filho e tanto o pai quanto o filho tentarão imprimir caracteres no terminal sem qualquer sincronização.

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>

int main()
{
    int pid;
    pid =  fork();
    srand(pid);
    if(pid < 0)
    {
        perror("fork"); exit(1);
    }
    else if(pid)
    {
        char *s = "abcdefgh";
        int l = strlen(s);
        for(int i = 0; i < l; ++i)
        {
            putchar(s[i]);
            fflush(stdout);
            sleep(rand() % 2);
            putchar(s[i]);
            fflush(stdout);
            sleep(rand() % 2);
        }
    }
    else
    {
        char *s = "ABCDEFGH";
        int l = strlen(s);
        for(int i = 0; i < l; ++i)
        {
            putchar(s[i]);
            fflush(stdout);
            sleep(rand() % 2);
            putchar(s[i]);
            fflush(stdout);
            sleep(rand() % 2);
        }
    }
}

Saída (1ª execução):

aAABaBCbCbDDcEEcddeFFGGHHeffgghh

(2ª corrida):

aabbccAABddBCeeCffgDDghEEhFFGGHH

Compilar e executar este programa deve fornecer uma saída diferente a cada vez.

Exemplo 1.2: Evite competir com semáforos #

Modificando o Exemplo 1.1 para usar semáforos, temos:

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/sem.h>

#define KEY 0x1111

union semun {
    int val;
    struct semid_ds *buf;
    unsigned short  *array;
};

struct sembuf p = { 0, -1, SEM_UNDO};
struct sembuf v = { 0, +1, SEM_UNDO};

int main()
{
    int id = semget(KEY, 1, 0666 | IPC_CREAT);
    if(id < 0)
    {
        perror("semget"); exit(11);
    }
    union semun u;
    u.val = 1;
    if(semctl(id, 0, SETVAL, u) < 0)
    {
        perror("semctl"); exit(12);
    }
    int pid;
    pid =  fork();
    srand(pid);
    if(pid < 0)
    {
        perror("fork"); exit(1);
    }
    else if(pid)
    {
        char *s = "abcdefgh";
        int l = strlen(s);
        for(int i = 0; i < l; ++i)
        {
            if(semop(id, &p, 1) < 0)
            {
                perror("semop p"); exit(13);
            }
            putchar(s[i]);
            fflush(stdout);
            sleep(rand() % 2);
            putchar(s[i]);
            fflush(stdout);
            if(semop(id, &v, 1) < 0)
            {
                perror("semop p"); exit(14);
            }

            sleep(rand() % 2);
        }
    }
    else
    {
        char *s = "ABCDEFGH";
        int l = strlen(s);
        for(int i = 0; i < l; ++i)
        {
            if(semop(id, &p, 1) < 0)
            {
                perror("semop p"); exit(15);
            }
            putchar(s[i]);
            fflush(stdout);
            sleep(rand() % 2);
            putchar(s[i]);
            fflush(stdout);
            if(semop(id, &v, 1) < 0)
            {
                perror("semop p"); exit(16);
            }

            sleep(rand() % 2);
        }
    }
}

Resultado:

aabbAABBCCccddeeDDffEEFFGGHHgghh

Compilar e executar este programa fornecerá a mesma saída todas as vezes.
