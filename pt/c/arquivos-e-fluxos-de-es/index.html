Sintaxe #

  • #include <stdio.h> /* Inclua isso para usar qualquer uma das seguintes seções */
  • FILE *fopen(const char *path, const char *mode); /* Abre um stream no arquivo em path com o mode */
  • FILE *freopen(const char *path, const char *mode, FILE *stream); /* Reabra um fluxo existente no arquivo em path com o mode */
  • int fclose(FILE *stream); /* Fecha um stream aberto */
  • size_t fread(void *ptr, size_t size, size_t nmemb, FILE *stream); /* Lê no máximo nmemb elementos de size bytes cada do stream e escreve-os em ptr. Retorna o número de elementos lidos. */
  • size_t fwrite(const void *ptr, size_t size, size_t nmemb, FILE *stream); /* Grava nmemb elementos de size bytes cada um de ptr para o stream. Retorna o número de elementos escritos. */
  • int fseek(FILE *stream, deslocamento longo, int de onde); /* Configura o cursor do fluxo para offset, relativo ao deslocamento informado por whence, e retorna 0 se for bem-sucedido. */
  • longo ftell(FILE *stream); /* Retorna o deslocamento da posição atual do cursor desde o início do fluxo. */
  • void retroceder(FILE *stream); /* Configura a posição do cursor para o início do arquivo. */
  • int fprintf(FILE *fout, const char *fmt, …); /* Escreve a string de formato printf em fout */
  • ARQUIVO *stdin; /* Fluxo de entrada padrão */
  • ARQUIVO *stdout; /* Fluxo de saída padrão */
  • ARQUIVO *stderr; /* Fluxo de erro padrão */

Parâmetros #

Parâmetro Detalhes
const char *modo Uma string que descreve o modo de abertura do fluxo com suporte de arquivo. Consulte as observações para valores possíveis.
int de onde Pode ser SEEK_SET para definir desde o início do arquivo, SEEK_END para definir a partir de seu final, ou SEEK_CUR para definir em relação ao valor atual do cursor. Nota: SEEK_END não é portátil.

Cadeias de modos: #

Strings de modo em fopen() e [freopen()](http://pubs.opengroup.org/ onlinepubs/9699919799/functions/freopen.html) pode ser um desses valores:

  • "r": Abra o arquivo em modo somente leitura, com o cursor definido para o início do arquivo.
  • "r+": Abre o arquivo no modo leitura-escrita, com o cursor posicionado no início do arquivo.
  • "w": Abra ou crie o arquivo no modo somente gravação, com seu conteúdo truncado para 0 bytes. O cursor é definido para o início do arquivo.
  • "w+": Abra ou crie o arquivo no modo leitura-escrita, com seu conteúdo truncado para 0 bytes. O cursor é definido para o início do arquivo.
  • "a": Abra ou crie o arquivo no modo somente gravação, com o cursor definido para o final do arquivo.
  • "a+": Abra ou crie o arquivo no modo leitura-escrita, com o cursor de leitura definido para o início do arquivo. A saída, no entanto, será sempre anexada ao final do arquivo.

Cada um desses modos de arquivo pode ter um b adicionado após a letra inicial (por exemplo, "rb" ou "a+b" ou "ab+"). O b significa que o arquivo deve ser tratado como um arquivo binário em vez de um arquivo de texto nos sistemas onde há uma diferença. Não faz diferença em sistemas do tipo Unix; é importante em sistemas Windows. (Além disso, o Windows fopen permite um t explícito em vez de b para indicar ‘arquivo de texto’ — e várias outras opções específicas da plataforma.)

  • "wx": Cria um arquivo de texto no modo somente gravação. O arquivo pode não existir.
  • "wbx": Cria um arquivo binário no modo somente gravação. O arquivo pode não existir.

O x, se presente, deve ser o último caractere na string de modo.

Abra e grave no arquivo #

#include <stdio.h>   /* for perror(), fopen(), fputs() and fclose() */
#include <stdlib.h>  /* for the EXIT_* macros */
 
int main(int argc, char **argv)
{
    int e = EXIT_SUCCESS;

    /* Get path from argument to main else default to output.txt */
    char *path = (argc > 1) ? argv[1] : "output.txt";

    /* Open file for writing and obtain file pointer */
    FILE *file = fopen(path, "w");
    
    /* Print error message and exit if fopen() failed */
    if (!file) 
    {
        perror(path);
        return EXIT_FAILURE;
    }

    /* Writes text to file. Unlike puts(), fputs() does not add a new-line. */
    if (fputs("Output in file.\n", file) == EOF)
    {
        perror(path);
        e = EXIT_FAILURE;
    }

    /* Close file */
    if (fclose(file)) 
    {
        perror(path);
        return EXIT_FAILURE;
    }
    return e;
}

Este programa abre o arquivo com o nome dado no argumento para main, padronizando para output.txt se nenhum argumento for fornecido. Se já existir um arquivo com o mesmo nome, seu conteúdo é descartado e o arquivo é tratado como um novo arquivo vazio. Se os arquivos ainda não existirem, a chamada fopen() o cria.

Se a chamada fopen() falhar por algum motivo, ela retornará um valor NULL e definirá o valor da variável global errno. Isto significa que o programa pode testar o valor retornado após a chamada fopen() e usar perror() se fopen() falhar.

Se a chamada fopen() for bem sucedida, ela retornará um ponteiro FILE válido. Este ponteiro pode então ser usado para referenciar este arquivo até que fclose() seja chamado nele.

A função fputs() escreve o texto dado no arquivo aberto, substituindo qualquer conteúdo anterior do arquivo. Da mesma forma que fopen(), a função fputs() também define o valor errno se falhar, embora neste caso a função retorne EOF para indicar a falha (caso contrário, retorna um valor não negativo) .

A função fclose() libera qualquer buffer, fecha o arquivo e libera a memória apontada por FILE *. O valor de retorno indica a conclusão assim como fputs() (embora retorne ‘0’ se for bem-sucedido), novamente também definindo o valor errno no caso de uma falha.

Executar processo #

#include <stdio.h>

void print_all(FILE *stream)
{
    int c;
    while ((c = getc(stream)) != EOF)
        putchar(c);
}
int main(void)
{
    FILE *stream;

    /* call netstat command. netstat is available for Windows and Linux */
    if ((stream = popen("netstat", "r")) == NULL)
        return 1;
  
    print_all(stream);
    pclose(stream);
    return 0;
}

Este programa executa um processo (netstat) via popen() e lê toda a saída padrão do processo e a ecoa na saída padrão.

Nota: popen() não existe na biblioteca C padrão, mas é uma parte de POSIX C)

fprintf #

Você pode usar fprintf em um arquivo como faria em um console com printf. Por exemplo, para acompanhar as vitórias, derrotas e empates do jogo, você pode escrever

/* saves wins, losses and, ties */
void savewlt(FILE *fout, int wins, int losses, int ties)
{
    fprintf(fout, "Wins: %d\nTies: %d\nLosses: %d\n", wins, ties, losses);
}

Uma nota lateral: Alguns sistemas (infamemente, Windows) não usam o que a maioria dos programadores chamaria de terminações de linha “normais”. Enquanto os sistemas do tipo UNIX usam \n para terminar linhas, o Windows usa um par de caracteres: \r (retorno de carro) e \n (alimentação de linha). Esta sequência é comumente chamada de CRLF. No entanto, sempre que usar C, você não precisa se preocupar com esses detalhes altamente dependentes da plataforma. Um compilador C é necessário para converter cada instância de \n para o final de linha de plataforma correto. Portanto, um compilador do Windows converteria \n em \r\n, mas um compilador do UNIX o manteria como está.

Obter linhas de um arquivo usando getline() #

A biblioteca POSIX C define a função getline(). Esta função aloca um buffer para conter o conteúdo da linha e retorna a nova linha, o número de caracteres na linha e o tamanho do buffer.

Programa de exemplo que obtém cada linha de example.txt:

#include <stdlib.h>
#include <stdio.h>  


#define FILENAME "example.txt"

int main(void)
{
  /* Open the file for reading */
  char *line_buf = NULL;
  size_t line_buf_size = 0;
  int line_count = 0;
  ssize_t line_size;
  FILE *fp = fopen(FILENAME, "r");
  if (!fp)
  {
    fprintf(stderr, "Error opening file '%s'\n", FILENAME);
    return EXIT_FAILURE;
  }

  /* Get the first line of the file. */
  line_size = getline(&line_buf, &line_buf_size, fp);

  /* Loop through until we are done with the file. */
  while (line_size >= 0)
  {
    /* Increment our line count */
    line_count++;

    /* Show the line details */
    printf("line[%06d]: chars=%06zd, buf size=%06zu, contents: %s", line_count,
        line_size, line_buf_size, line_buf);

    /* Get the next line */
    line_size = getline(&line_buf, &line_buf_size, fp);
  }

  /* Free the allocated line buffer */
  free(line_buf);
  line_buf = NULL;

  /* Close the file now that we are done with it */
  fclose(fp);

  return EXIT_SUCCESS;
}

No exemplo, getline() é inicialmente chamado sem buffer alocado. Durante esta primeira chamada, getline() aloca um buffer, lê a primeira linha e coloca o conteúdo da linha no novo buffer. Nas chamadas subsequentes, getline() atualiza o mesmo buffer e apenas realoca o buffer quando não é mais grande o suficiente para caber em toda a linha. O buffer temporário é então liberado quando terminamos com o arquivo.

Outra opção é getdelim(). Isto é o mesmo que getline() exceto que você especifica o caractere final de linha. Isso só é necessário se o último caractere da linha do seu tipo de arquivo não for ‘\n’. getline() funciona mesmo com arquivos de texto do Windows porque com a linha multibyte terminando ("\r\n") ’\n’` ainda é o último caractere na linha.

Exemplo de implementação de getline() #

#include <stdlib.h>
#include <stdio.h>
#include <errno.h>
#include <stdint.h>


#if !(defined _POSIX_C_SOURCE)
typedef long int ssize_t;
#endif

/* Only include our version of getline() if the POSIX version isn't available. */

#if !(defined _POSIX_C_SOURCE) || _POSIX_C_SOURCE < 200809L

#if !(defined SSIZE_MAX)
#define SSIZE_MAX (SIZE_MAX >> 1)
#endif

ssize_t getline(char **pline_buf, size_t *pn, FILE *fin)
{
  const size_t INITALLOC = 16;
  const size_t ALLOCSTEP = 16;
  size_t num_read = 0;

  /* First check that none of our input pointers are NULL. */
  if ((NULL == pline_buf) || (NULL == pn) || (NULL == fin))
  {
    errno = EINVAL;
    return -1;
  }

  /* If output buffer is NULL, then allocate a buffer. */
  if (NULL == *pline_buf)
  {
    *pline_buf = malloc(INITALLOC);
    if (NULL == *pline_buf)
    {
      /* Can't allocate memory. */
      return -1;
    }
    else
    {
      /* Note how big the buffer is at this time. */
      *pn = INITALLOC;
    }
  }

  /* Step through the file, pulling characters until either a newline or EOF. */

  {
    int c;
    while (EOF != (c = getc(fin)))
    {
      /* Note we read a character. */
      num_read++;

      /* Reallocate the buffer if we need more room */
      if (num_read >= *pn)
      {
        size_t n_realloc = *pn + ALLOCSTEP;
        char * tmp = realloc(*pline_buf, n_realloc + 1); /* +1 for the trailing NUL. */
        if (NULL != tmp)
        {
          /* Use the new buffer and note the new buffer size. */
          *pline_buf = tmp;
          *pn = n_realloc;
        }
        else
        {
          /* Exit with error and let the caller free the buffer. */
          return -1;
        }

        /* Test for overflow. */
        if (SSIZE_MAX < *pn)
        {
          errno = ERANGE;
          return -1;
        }
      }

      /* Add the character to the buffer. */
      (*pline_buf)[num_read - 1] = (char) c;

      /* Break from the loop if we hit the ending character. */
      if (c == '\n')
      {
        break;
      }
    }

    /* Note if we hit EOF. */
    if (EOF == c)
    {
      errno = 0;
      return -1;
    }
  }

  /* Terminate the string by suffixing NUL. */
  (*pline_buf)[num_read] = '\0';

  return (ssize_t) num_read;
}

#endif

Abra e escreva em um arquivo binário #

#include <stdlib.h>
#include <stdio.h>


int main(void)
{
   result = EXIT_SUCCESS;

   char file_name[] = "outbut.bin";
   char str[] = "This is a binary file example";
   FILE * fp = fopen(file_name, "wb");
   
   if (fp == NULL)  /* If an error occurs during the file creation */
   {
     result = EXIT_FAILURE;
     fprintf(stderr, "fopen() failed for '%s'\n", file_name);
   }
   else
   {
     size_t element_size = sizeof *str;
     size_t elements_to_write = sizeof str;

     /* Writes str (_including_ the NUL-terminator) to the binary file. */
     size_t elements_written = fwrite(str, element_size, elements_to_write, fp); 
     if (elements_written != elements_to_write)
     {
       result = EXIT_FAILURE;
       /* This works for >=c99 only, else the z length modifier is unknown. */
       fprintf(stderr, "fwrite() failed: wrote only %zu out of %zu elements.\n", 
         elements_written, elements_to_write);
       /* Use this for <c99: *
       fprintf(stderr, "fwrite() failed: wrote only %lu out of %lu elements.\n", 
         (unsigned long) elements_written, (unsigned long) elements_to_write);
        */
     }

     fclose(fp);
   }

   return result;
}

Este programa cria e escreve texto na forma binária através da função fwrite no arquivo output.bin.

Se já existir um arquivo com o mesmo nome, seu conteúdo é descartado e o arquivo é tratado como um novo arquivo vazio.

Um fluxo binário é uma sequência ordenada de caracteres que pode registrar dados internos de forma transparente. Neste modo, os bytes são escritos entre o programa e o arquivo sem nenhuma interpretação.

Para escrever inteiros de forma portátil, deve-se saber se o formato do arquivo os espera no formato big ou little-endian e o tamanho (geralmente 16, 32 ou 64 bits). O deslocamento e o mascaramento de bits podem ser usados ​​para escrever os bytes na ordem correta. Não é garantido que os inteiros em C tenham representação de complemento de dois (embora quase todas as implementações tenham). Felizmente, uma conversão para unsigned é garantida para usar complemento de dois. O código para escrever um inteiro assinado em um arquivo binário é, portanto, um pouco surpreendente.

/* write a 16-bit little endian integer */
int fput16le(int x, FILE *fp)
{
    unsigned int rep = x;
    int e1, e2;

    e1 = fputc(rep & 0xFF, fp);
    e2 = fputc((rep >> 8) & 0xFF, fp);

    if(e1 == EOF || e2 == EOF)
        return EOF;
    return 0;  
}

As demais funções seguem o mesmo padrão com pequenas modificações de tamanho e ordem de bytes.

fscanf() #

Digamos que temos um arquivo de texto e queremos ler todas as palavras desse arquivo, para fazer alguns requisitos.

arquivo.txt:

This is just
a test file
to be used by fscanf()

Esta é a função principal:

#include <stdlib.h>
#include <stdio.h>

void printAllWords(FILE *);

int main(void)
{
  FILE *fp;

  if ((fp = fopen("file.txt", "r")) == NULL) {
      perror("Error opening file");
      exit(EXIT_FAILURE);
  }

  printAllWords(fp);

  fclose(fp);
  
  return EXIT_SUCCESS;
}

void printAllWords(FILE * fp)
{
    char tmp[20];
    int i = 1;

    while (fscanf(fp, "%19s", tmp) != EOF) {
        printf("Word %d: %s\n", i, tmp);
        i++;
    }
}

Word 4: a 
Word 5: test
Word 6: file
Word 7: to
Word 8: be 
Word 9: used
Word 10: by
Word 11: fscanf()

Ler linhas de um arquivo #

O cabeçalho stdio.h define a função fgets(). Esta função lê uma linha de um stream e a armazena em uma string especificada. A função para de ler o texto do fluxo quando os caracteres n - 1 são lidos, o caractere de nova linha ('\n') é lido ou o final do arquivo (EOF) é atingido.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define MAX_LINE_LENGTH 80

int main(int argc, char **argv)
{
    char *path;
    char line[MAX_LINE_LENGTH] = {0};
    unsigned int line_count = 0;
    
    if (argc < 1)
        return EXIT_FAILURE;
    path = argv[1];
    
    /* Open file */
    FILE *file = fopen(path, "r");
    
    if (!file)
    {
        perror(path);
        return EXIT_FAILURE;
    }
    
    /* Get each line until there are none left */
    while (fgets(line, MAX_LINE_LENGTH, file))
    {
        /* Print each line */
        printf("line[%06d]: %s", ++line_count, line);
        
        /* Add a trailing newline to lines that don't already have one */
        if (line[strlen(line) - 1] != '\n')
            printf("\n");
    }
    
    /* Close file */
    if (fclose(file))
    {
        return EXIT_FAILURE;
        perror(path);
    }
}

Este exemplo muito simples permite um comprimento de linha máximo fixo, de modo que linhas mais longas sejam efetivamente contadas como duas linhas. A função fgets() requer que o código de chamada forneça a memória a ser usada como destino para a linha lida.

POSIX faz o [getline()](https://www.wikiod.com/pt/c/arquivos-e-fluxos-de-es#Obter linhas de um arquivo usando getline() =201703141659155440184) função disponível que, em vez disso, aloca memória internamente para aumentar o buffer conforme necessário para uma linha de qualquer comprimento (desde que haja memória suficiente).
