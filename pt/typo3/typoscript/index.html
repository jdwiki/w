Sintaxe #

  • #comentário de uma linha
  • /* comentários de várias linhas */
  • parâmetro = valor
  • parâmetro.propriedade = valor2
  • parameter.property > # delete property
  • parameter.property2 < parameter.property # copia propriedades (deep)
  • parameter.property := addToList(35) # Adiciona valor à lista separada por vírgulas
  • parameter.property\.with\.dots = value # Pontos de escape para permitir propriedades com pontos

TypeScript básico: Olá, mundo #

page = PAGE
page.10 = TEXT
page.10.value = HELLO WORLD

Normalmente, esses snippets de typoScript são adicionados à Web » Template » Info/Modify » setup

Este snippet abre um novo objeto PAGE. Dentro do objeto PAGE, a 10ª entrada é definida como um objeto TEXT. O valor do objeto TEXT é definido como HELLO WORLD. É convenção usar apenas os incrementos de 10 para definir objetos de conteúdo para permitir que mais entradas sejam adicionadas posteriormente.

Imagem e Recurso de Imagem #

Obter imagem do arquivo

lib.myImage = IMAGE
lib.myImage.file = fileadmin/My-Image.png
lib.myImage.file.width = 100
lib.myImage.file.height = 100

Obter imagem das propriedades da página

lib.pageImage = IMAGE
lib.pageImage {
    file.import.data = levelmedia: -1, "slide"
    file.import = uploads/media/
    file.import.listNum = 0
    file.import.override.field = media

    border = 0
    altText = xy
    titleText = xy
}

obter imagens dos recursos da página (FAL)

lib.pageResources = FILES 
lib.pageResources { 
  references { 
    table = pages 
    uid.data = uid
    fieldName = media
  } 
  renderObj = IMAGE 
  renderObj {
    file { 
      import.data = file:current:uid 
      treatIdAsReference = 1 
      width = 150c 
      height = 150c 
    } 
    altText.data = file:current:alternative
    titleText.data = file:current:title
  } 
  maxItems = 3
}

Obter recurso de imagem

lib.myImage = IMAGE
lib.myImage.file = fileadmin/My-Image.png
lib.myImage.file.width = 100
lib.myImage.file.height = 100
lib.myImage.stdWrap.wrap (
 <div style="background-image:url(|); width:100px; height:100px;">Headline</div>
)

Obtenha imagens de elementos de conteúdo

lib.ceImages = FILES
lib.ceImages {
    stdWrap.wrap = <div class="pic">|</div>
    references {
        table = tt_content
        // current CE (needs context)
        #uid.data = uid
        // for CSC
        fieldName = image
        // for FSC
        fieldname = assets
    }
    renderObj = IMAGE
    renderObj {
        file {
            import.data = file:current:uid
            treatIdAsReference = 1
            width = 150c
            height = 150c
        }
        altText.data = file:current:alternative
        titleText.data = file:current:title
        stdWrap.typolink.parameter.data = file:current:link
    }
    maxItems = 5
}

Obter data no objeto #

Obter data e hora atuais

lib.date = TEXT
lib.date {
  data = date:U
  strftime = %d.%m.%Y %H:%M:%S
  wrap = Today is |
}

Obtenha a data e hora do último login de fe_users

lib.date = TEXT
lib.date {
  data = TSFE:fe_user|user|lastlogin
  strftime = %d.%m.%Y %H:%M:%S
  wrap = Last login is at |
}

Considere o URL como http://test.com/page1/?tstamp=1469683852

lib.date = TEXT
lib.date {
  data = GP : tstamp
  strftime = %d.%m.%Y %H:%M:%S
  wrap = Current tstamp is |
}

Criar link #

Um link para qualquer texto usando o objeto typolink

lib.link = TEXT
lib.link {
    value = Here is link text
    
    typolink {
        
        #You can give page uid or any external url here
        parameter = http://www.example.com/

        #Target of link
        extTarget = _blank

        #Additional parameters bound to link
        ATagParams = class="linkclass" title="Here is a link"
    }
}

A seguir criará um link com pop-up javascript

lib.link = TEXT
lib.link {
     value = Open a popup window.

     stdWrap.typolink {
          # The first parameter is the page ID of the target page,
          # second parameter is the size of the popup window.
          parameter = 10 500x400

          # The title attribute of the link.
          title = Click here to open a popup window.

          # The parameters of the popup window.
          JSwindow_params = menubar=0, scrollbars=0, toolbar=0, resizable=1

     }
}

Obtém o título da página atual #

lib.pagetitle = TEXT
lib.pagetitle.data = page : title

Obter URL atual #

lib.currentURL= TEXT
lib.currentURL.data = getIndpEnv:TYPO3_REQUEST_URL
