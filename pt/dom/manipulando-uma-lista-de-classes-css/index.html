Testando para uma classe #

Os navegadores modernos fornecem um objeto classList para facilitar a manipulação do atributo class do elemento. Navegadores mais antigos requerem manipulação direta da propriedade className do elemento.

* Observe que os nomes das classes não são armazenados na propriedade do elemento em nenhuma ordem específica

Testar se um elemento contém uma classe requer um pouco de manipulação da propriedade className. Este exemplo está usando um método de matriz para testar a classe.

function hasClass(el, name) {
  var classes = (el && el.className || "").split(/\s+/);
  return classes.indexOf(name) > -1;
}
var el = document.getElementById("link1");
console.log(hasClass(el, "foo"));

Testar vários nomes de classe exigiria um loop.

function hasClass(el, name) {
  name = name.split(/[\s.]+/);
  var hasClass = true,
    classes = (el && el.className || "").split(/\s+/),
    index = name.length;
  while (index--) {
    hasClass = hasClass && classes.indexOf(name[index]) > -1;
  }
  return hasClass;
}
var el = document.getElementById("link1");
console.log(hasClass(el, "foo"));

Em vez de usar .indexOf(), você também pode considerar usar uma expressão regular.

function hasClass(el, name) {
  return new RegExp("\\b" + name+ "\\b").test(el.className);
}
var el = document.getElementById("link1");
console.log(hasClass(el, "foo"));

O teste para um único nome de classe é feito da seguinte forma:

var hasClass = document.querySelector("#link1").classList.contains("foo");

Para vários nomes de classe, é mais fácil usar matches. Observe o uso do seletor de classe; O seletor pode ser qualquer seletor de string válido (id, atributo, pseudo-classes, etc).

var hasClass = document.querySelector("#link1").matches('.foo.bar');
var hasClass = document.querySelector("#link2").matches('a.bar[href]');

Adicionando uma classe #

Os navegadores modernos fornecem um objeto classList para facilitar a manipulação do atributo class do elemento. Navegadores mais antigos requerem manipulação direta da propriedade className do elemento.

Um método simples para adicionar uma classe a um elemento é anexá-lo ao final da propriedade className. Isso não impedirá nomes de classe duplicados e espaços devem ser incluídos entre os nomes de classe.

document.getElementById("link1").className += " foo";
document.getElementById("link2").className += " foo bar";

Para vários elementos, você precisará adicionar os nomes das classes dentro de um loop

var els = document.getElementsByClassName("foo"),
  indx = els.length;
while (indx--) {
  els[indx].className += " bar baz";
}

Um único nome de classe pode ser adicionado como uma string. Para adicionar vários nomes de classe, use o operador spread do ES6:

document.querySelector("#link1").classList.add("foo");
document.querySelector("#link2").classList.add(...['foo', 'bar']);

Para vários elementos, você precisará adicionar os nomes das classes dentro de um loop

document.querySelectorAll(".foo").forEach(el => {
  el.classList.add(...['bar', 'baz']);
});

Removendo uma classe #

Os navegadores modernos fornecem um objeto classList para facilitar a manipulação do atributo class do elemento. Navegadores mais antigos requerem manipulação direta da propriedade className do elemento.

* Observe que os nomes das classes não são armazenados na propriedade do elemento em nenhuma ordem específica

Remover uma classe de um elemento requer um pouco de manipulação da propriedade className.

var toRemove = "bar",
  el = document.getElementById("link1");
el.className = el.className.replace(new RegExp("\\b" + toRemove + "\\b", "g"), "").trim();

A remoção de vários nomes de classe exigiria um loop. Os exemplos restantes usarão uma função para isolar o trabalho

function removeClass(el, name) {
  name = name.split(/\s+/);
  var index = name.length,
    classes = el.className;
  while (index--) {
    classes = classes.replace(new RegExp("\\b" + name[index] + "\\b", "g"), "").trim();
  }
  el.className = classes;
}
var el = document.getElementById("link1");
removeClass(el, "bar baz");

Vários elementos com vários nomes de classe para remover exigiriam dois loops

function removeClass(els, name) {
  name = name.split(/\s+/);
  var regex, len,
    index = name.length;
  while (index--) {
    regex = new RegExp("\\b" + name[index] + "\\b", "g");
    len = els.length;
    while (len--) {
      els[len].className = els[len].className.replace(regex, "").trim();
    }
  }
}
var els = document.getElementsByTagName("a");
removeClass(els, "bar baz");

Um único nome de classe pode ser removido como uma string. Para remover vários nomes de classe, use o operador spread do ES6:

document.querySelector("#link1").classList.remove("foo");
document.querySelector("#link2").classList.remove(...['foo', 'bar']);

Para vários elementos, você precisará remover os nomes das classes dentro de um loop

document.querySelectorAll(".foo").forEach(el => {
  el.classList.remove(...['bar', 'baz']);
});
