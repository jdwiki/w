Parâmetros #

Parâmetro Detalhes
SuiteScript 2.0 -
scriptContext {Objeto}
scriptContext.newRecord {N/record.Record} Uma referência ao registro que está sendo carregado do banco de dados
scriptContext.type {UserEventType} O tipo de ação que acionou este evento de usuário
scriptContext.form {N/ui/serverWidget.Form} Uma referência ao formulário de interface do usuário que será renderizado
SuiteScript 1.0 -
tipo {Object} O tipo de ação que acionou este User Event
formulário {nlobjForm} Uma referência ao formulário de interface do usuário que será renderizado
pedido {nlobjRequest} a solicitação HTTP GET; disponível apenas quando acionado por solicitações do navegador

antes de carregar #

O evento Before Load é acionado por qualquer operação de leitura em um registro. Sempre que um usuário, um script, uma importação de CSV ou uma solicitação de serviço da Web tenta ler um registro do banco de dados, o evento Before Load é acionado.

Grave ações que acionam um evento beforeLoad:

  • Crio
  • Editar
  • Visualizar / Carregar
  • Cópia de
  • Imprimir
  • E-mail
  • Olhada rápida

Casos de uso típicos para beforeLoad #

  • Modifique o formulário da interface do usuário antes que o usuário o veja
  • Definir valores de campo padrão
  • Pré-processamento de dados

Eventos do usuário não encadeados #

O código escrito em eventos de usuário não acionará nenhum evento de usuário em outros registros. Por exemplo, carregar o registro do cliente associado do beforeLoad de um registro do pedido de venda não acionará o beforeLoad do registro do cliente. Mesmo se você estiver carregando outro registro de transação, seus eventos de usuário não serão acionados.

O NetSuite faz isso para evitar que os eventos do usuário sejam disparados entre si em um loop infinito. Se você precisa que os eventos do usuário sejam acionados em uma sequência encadeada, outros tipos de script (por exemplo, RESTlets, Suitelets, scripts agendados) precisarão ser injetados entre os eventos.

Event Handler retorna void #

O tipo de retorno do manipulador de eventos beforeLoad é void. Quaisquer dados retornados de nosso manipulador de eventos não têm efeito no sistema. Não precisamos retornar nada de nossa função de manipulador, pois não podemos fazer nada com seu valor retornado.

Mínimo: Registre uma mensagem no Before Load #

// 1.0
function beforeLoad(type, form, request) {
    nlapiLogExecution("DEBUG", "Before Load", "type=" + type);
}

// 2.0
/**
 * @NApiVersion 2.x
 * @NScriptType UserEventScript
 * @NModuleScope SameAccount
 */
define(["N/log"], function (log) {
    function beforeLoad(context) {
        log.debug({
            "title": "Before Load",
            "details": "type=" + context.type
        });
    }

    return {
        "beforeLoad": beforeLoad
    };
});

Modificando o formulário da interface do usuário #

// 1.0
// Revealing Module pattern, structures 1.0 similar to 2.0
var myNamespace = myNamespace || {};
myNamespace.example = (function () {

    /** @appliedtorecord employee */
    var exports = {};
    
    function beforeLoad(type, form, request) {
        showBonusEligibility(form);
    }

    function showBonusEligibility(form) {
        var field = form.addField("custpage_is_bonus_eligible",
            "checkbox", "Eligible for Bonus?");
        field.setDefaultValue(isEligibleForBonus(nlapiGetNewRecord()) ? "T" : "F");
    }

    function isEligibleForBonus(rec) {
        // Implement actual business rules for bonus eligibility here
        return true;
    }
    
    exports.beforeLoad = beforeLoad;
    return exports;
})();

// 2.0
/**
 * @appliedtorecord employee
 * @NScriptType UserEventScript
 * @NApiVersion 2.x
 */
define(["N/log", "N/ui/serverWidget"], function (log, ui) {
    var exports = {};
    
    function beforeLoad(context) {
        showBonusEligibility(context.form);
    }
    
    function showBonusEligibility(form) {
        var field = form.addField({
            "id": "custpage_is_bonus_eligible",
            "label": "Eligible for Bonus?",
            "type": ui.FieldType.CHECKBOX
        });
        field.defaultValue = (isEligibleForBonus() ? "T" : "F");
    }

    function isEligibleForBonus(rec) {
        // Implement actual business rules for bonus eligibility here
        return true;
    }
    
    exports.beforeLoad = beforeLoad;
    return exports;
});

Restringir a execução com base na ação que acionou o evento do usuário #

// 1.0
// Utilize the type argument and raw Strings to filter your
// execution by the action
function beforeLoad(type, form, request) {
    // Don't do anything on APPROVE
    // Note that `type` is an Object, so we must use ==, not ===
    if (type == "approve") {
        return;
    }
    
    // Continue with normal business logic...
}

// 2.0
/**
 * @appliedtorecord employee
 * @NScriptType UserEventScript
 * @NApiVersion 2.x
 */
define([], function () {
    var exports = {};
    
    // Utilize context.type value and context.UserEventType enumeration
    // to filter your execution by the action
    function beforeLoad(context) {
        // Don't do anything on APPROVE
        if (context.type === context.UserEventType.APPROVE) {
            return;
        }
        
        // Continue with normal business logic...
    }
    
    exports.beforeLoad = beforeLoad;
    return exports;
});

Restringir a execução com base no contexto que acionou o evento do usuário #

No SuiteScript 1.0, recuperamos o contexto de execução atual usando nlapiGetContext().getExecutionContext() e comparamos o resultado com as Strings brutas apropriadas.

// 1.0 in Revealing Module pattern
var myNamespace = myNamespace || {};
myNamespace.example = (function () {
    var exports = {};
    
    function beforeLoad(type, form, request) {
        showBonusEligibility(form);
    }
    
    function showBonusEligibility(form) {
        // Doesn't make sense to modify UI form when the request
        // did not come from the UI
        var currentContext = nlapiGetContext().getExecutionContext();
        if (!wasTriggeredFromUi(currentContext)) {
            return;
        }
        
        // Continue with form modification...
    }
    
    function wasTriggeredFromUi(context) {
        // Current context must be compared to raw Strings
        return (context === "userinterface");
    }
    
    function isEligibleForBonus() {
        return true;
    }
    
    exports.beforeLoad = beforeLoad;
    return exports;
})();

No SuiteScript 2.0, obtemos o contexto de execução atual importando o módulo N/runtime e inspecionando sua propriedade executionContext. Podemos então comparar seu valor com os valores da enumeração runtime.ContextType em vez de Strings brutas.

// 2.0
/**
 * @NScriptType UserEventScript
 * @NApiVersion 2.x
 */
define(["N/ui/serverWidget", "N/runtime"], function (ui, runtime) {
    var exports = {};
    
    function beforeLoad(scriptContext) {
        showBonusEligibility(scriptContext.form);
    }
    
    function showBonusEligibility(form) {
        // Doesn't make sense to modify the form if the 
        if (!wasTriggeredFromUi(runtime.executionContext)) {
            return;
        }
        
        // Continue with form modification...
    }

    function wasTriggeredFromUi(context) {
        // Context can be compared to enumeration from runtime module
        return (context === runtime.ContextType.USER_INTERFACE);
    }
    
    exports.beforeLoad = beforeLoad;
    return exports;
});
