Meça e verifique suas suposições de desempenho #

Este exemplo foi escrito com F# em mente, mas as idéias são aplicáveis ​​em todos ambientes

A primeira regra ao otimizar o desempenho é não confiar em suposições; Sempre Meça e Verifique suas suposições.

Como não estamos escrevendo código de máquina diretamente, é difícil prever como o compilador e JIT:er transformar seu programa em código de máquina. É por isso que precisamos medir o tempo de execução para ver que obtemos a melhoria de desempenho que esperamos e Verifique se o programa real não contém nenhuma sobrecarga oculta.

A verificação é o processo bastante simples que envolve a engenharia reversa do executável usando, por exemplo, ferramentas como ILSpy. O JIT:er complica a verificação, pois ver o código de máquina real é complicado, mas factível. No entanto, geralmente examinar o ‘código IL’ traz grandes ganhos.

O problema mais difícil é Medir; mais difícil porque é complicado configurar realista situações que permitem medir melhorias no código. Ainda Medir é inestimável.

Analisando funções F# simples

Vamos examinar algumas funções F# simples que acumulam todos os inteiros em 1..n escrito de várias maneiras diferentes. Como o intervalo é uma série aritmética simples o resultado pode ser calculado diretamente, mas para o propósito deste exemplo nós iterar no intervalo.

Primeiro, definimos algumas funções úteis para medir o tempo que uma função leva:

// now () returns current time in milliseconds since start
let now : unit -> int64 =
  let sw = System.Diagnostics.Stopwatch ()
  sw.Start ()
  fun () -> sw.ElapsedMilliseconds

// time estimates the time 'action' repeated a number of times
let time repeat action : int64*'T =
  let v = action ()  // Warm-up and compute value

  let b = now ()
  for i = 1 to repeat do
    action () |> ignore
  let e = now ()

  e - b, v

time executa uma ação repetidamente, precisamos executar os testes por algumas centenas milissegundos para reduzir a variação.

Então definimos algumas funções que acumulam todos os inteiros em 1..n em jeitos diferentes.

// Accumulates all integers 1..n using 'List'
let accumulateUsingList n =
  List.init (n + 1) id
  |> List.sum

// Accumulates all integers 1..n using 'Seq'
let accumulateUsingSeq n =
  Seq.init (n + 1) id
  |> Seq.sum

// Accumulates all integers 1..n using 'for-expression'
let accumulateUsingFor n =
  let mutable sum = 0
  for i = 1 to n do
    sum <- sum + i
  sum

// Accumulates all integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach n =
  let mutable sum = 0
  for i in 1..n do
    sum <- sum + i
  sum

// Accumulates all integers 1..n using 'foreach-expression' over list range
let accumulateUsingForEachOverList n =
  let mutable sum = 0
  for i in [1..n] do
    sum <- sum + i
  sum

// Accumulates every second integer 1..n using 'foreach-expression' over range
let accumulateUsingForEachStep2 n =
  let mutable sum = 0
  for i in 1..2..n do
    sum <- sum + i
  sum

// Accumulates all 64 bit integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach64 n =
  let mutable sum = 0L
  for i in 1L..int64 n do
    sum <- sum + i
  sum |> int

// Accumulates all integers n..1 using 'for-expression' in reverse order
let accumulateUsingReverseFor n =
  let mutable sum = 0
  for i = n downto 1 do
    sum <- sum + i
  sum

// Accumulates all 64 integers n..1 using 'tail-recursion' in reverse order
let accumulateUsingReverseTailRecursion n =
  let rec loop sum i =
    if i > 0 then
      loop (sum + i) (i - 1)
    else
      sum
  loop 0 n

Assumimos que o resultado é o mesmo (exceto por uma das funções que usa incremento de 2), mas há diferença no desempenho. Para medir isso o seguinte função é definida:

let testRun (path : string) =
  use testResult = new System.IO.StreamWriter (path)
  let write   (l : string)  = testResult.WriteLine l
  let writef  fmt           = FSharp.Core.Printf.kprintf write fmt

  write "Name\tTotal\tOuter\tInner\tCC0\tCC1\tCC2\tTime\tResult"

  // total is the total number of iterations being executed
  let total   = 10000000
  // outers let us variate the relation between the inner and outer loop
  //  this is often useful when the algorithm allocates different amount of memory
  //  depending on the input size. This can affect cache locality
  let outers  = [| 1000; 10000; 100000 |]
  for outer in outers do
    let inner = total / outer

    // multiplier is used to increase resolution of certain tests that are significantly
    //  faster than the slower ones

    let testCases =
      [|
    //   Name of test                         multiplier    action
        "List"                              , 1           , accumulateUsingList
        "Seq"                               , 1           , accumulateUsingSeq
        "for-expression"                    , 100         , accumulateUsingFor
        "foreach-expression"                , 100         , accumulateUsingForEach
        "foreach-expression over List"      , 1           , accumulateUsingForEachOverList
        "foreach-expression increment of 2" , 1           , accumulateUsingForEachStep2
        "foreach-expression over 64 bit"    , 1           , accumulateUsingForEach64
        "reverse for-expression"            , 100         , accumulateUsingReverseFor
        "reverse tail-recursion"            , 100         , accumulateUsingReverseTailRecursion
      |]
    for name, multiplier, a in testCases do
      System.GC.Collect (2, System.GCCollectionMode.Forced, true)
      let cc g = System.GC.CollectionCount g

      printfn "Accumulate using %s with outer=%d and inner=%d ..." name outer inner

      // Collect collection counters before test run
      let pcc0, pcc1, pcc2 = cc 0, cc 1, cc 2

      let ms, result       = time (outer*multiplier) (fun () -> a inner)
      let ms               = (float ms / float multiplier)

      // Collect collection counters after test run
      let acc0, acc1, acc2 = cc 0, cc 1, cc 2
      let cc0, cc1, cc2    = acc0 - pcc0, acc1 - pcc1, acc1 - pcc1
      printfn "  ... took: %f ms, GC collection count %d,%d,%d and produced %A" ms cc0 cc1 cc2 result

      writef "%s\t%d\t%d\t%d\t%d\t%d\t%d\t%f\t%d" name total outer inner cc0 cc1 cc2 ms result

Vemos uma diferença dramática e alguns dos resultados são inesperadamente ruins.

Vejamos os casos ruins:

Lista

// Accumulates all integers 1..n using 'List'
let accumulateUsingList n =
  List.init (n + 1) id
  |> List.sum

O que acontece aqui é uma lista completa contendo todos os inteiros 1..n é criada e reduzido usando uma soma. Isso deve ser mais caro do que apenas iterar e acumulando ao longo do intervalo, parece cerca de 42x mais lento que o loop for.

Além disso, podemos ver que o GC executou cerca de 100x durante a execução do teste porque o código alocava muitos objetos. Isso também custa CPU.

Seq

// Accumulates all integers 1..n using 'Seq'
let accumulateUsingSeq n =
  Seq.init (n + 1) id
  |> Seq.sum

A versão Seq não aloca uma Lista completa, então é um pouco surpreendente que isso ~270x mais lento que o loop for. Além disso, vemos que o GC executou 661x.

Seq é ineficiente quando a quantidade de trabalho por item é muito pequena (neste caso agregando dois inteiros).

O ponto não é nunca usar Seq. A questão é medir.

(manofstick edit: Seq.init é o culpado deste grave problema de desempenho. É muito mais eficiente usar a expressão { 0 .. n } em vez de Seq.init (n+1) id. Mesmo após o lançamento, porém, o arquivo Seq.init . .. |> Seq.sum ainda será lento, mas um pouco contra-intuitivo, Seq.init ... |> Seq.map id |> Seq.sum será bastante rápido. Isso foi feito para manter a compatibilidade com versões anteriores com A implementação do Seq.init, que não calcula o Current inicialmente, mas os envolve em um objeto Lazy - embora isso também deva ter um desempenho um pouco melhor devido a [este PR](https://github. com/dotnet/coreclr/pull/8963). expressão foreach sobre lista

// Accumulates all integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach n =
  let mutable sum = 0
  for i in 1..n do
    sum <- sum + i
  sum

// Accumulates all integers 1..n using 'foreach-expression' over list range
let accumulateUsingForEachOverList n =
  let mutable sum = 0
  for i in [1..n] do
    sum <- sum + i
  sum

A diferença entre essas duas funções é muito sutil, mas o desempenho diferença não é, aproximadamente ~ 76x. Por quê? Vamos fazer engenharia reversa do código ruim:

public static int accumulateUsingForEach(int n)
{
  int sum = 0;
  int i = 1;
  if (n >= i)
  {
    do
    {
      sum += i;
      i++;
    }
    while (i != n + 1);
  }
  return sum;
}

public static int accumulateUsingForEachOverList(int n)
{
  int sum = 0;
  FSharpList<int> fSharpList = SeqModule.ToList<int>(Operators.CreateSequence<int>(Operators.OperatorIntrinsics.RangeInt32(1, 1, n)));
  for (FSharpList<int> tailOrNull = fSharpList.TailOrNull; tailOrNull != null; tailOrNull = fSharpList.TailOrNull)
  {
    int i = fSharpList.HeadOrDefault;
    sum += i;
    fSharpList = tailOrNull;
  }
  return sum;
}

accumulateUsingForEach é implementado como um loop while eficiente, mas for i in [1..n] é convertido em:

FSharpList<int> fSharpList =
  SeqModule.ToList<int>(
    Operators.CreateSequence<int>(
      Operators.OperatorIntrinsics.RangeInt32(1, 1, n)));

Isso significa que primeiro criamos um Seq sobre 1..n e finalmente chamamos ToList.

Caro.

incremento de expressão foreach de 2

// Accumulates all integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach n =
  let mutable sum = 0
  for i in 1..n do
    sum <- sum + i
  sum

// Accumulates every second integer 1..n using 'foreach-expression' over range
let accumulateUsingForEachStep2 n =
  let mutable sum = 0
  for i in 1..2..n do
    sum <- sum + i
  sum

Mais uma vez a diferença entre essas duas funções é sutil, mas o desempenho a diferença é brutal: ~25x

Mais uma vez vamos executar o ILSpy:

public static int accumulateUsingForEachStep2(int n)
{
  int sum = 0;
  IEnumerable<int> enumerable = Operators.OperatorIntrinsics.RangeInt32(1, 2, n);
  foreach (int i in enumerable)
  {
    sum += i;
  }
  return sum;
}

Um Seq é criado sobre 1..2..n e então iteramos sobre Seq usando o enumerador.

Esperávamos que F# criasse algo assim:

public static int accumulateUsingForEachStep2(int n)
{
  int sum = 0;
  for (int i = 1; i < n; i += 2)
  {
    sum += i;
  }
  return sum;
}

No entanto, o compilador F# suporta apenas loops eficientes para intervalos int32 que incrementar em um. Para todos os outros casos, recai sobre Operators.OperatorIntrinsics.RangeInt32. O que explicará a próxima surpresa resultado

expressão foreach acima de 64 bits

// Accumulates all 64 bit integers 1..n using 'foreach-expression' over range
let accumulateUsingForEach64 n =
  let mutable sum = 0L
  for i in 1L..int64 n do
    sum <- sum + i
  sum |> int

Isso executa ~ 47x mais lento que o loop for, a única diferença é que iteramos mais de 64 bits inteiros. O ILSpy nos mostra o porquê:

public static int accumulateUsingForEach64(int n)
{
  long sum = 0L;
  IEnumerable<long> enumerable = Operators.OperatorIntrinsics.RangeInt64(1L, 1L, (long)n);
  foreach (long i in enumerable)
  {
    sum += i;
  }
  return (int)sum;
}

F# só suporta loops eficientes para números int32 ele tem que usar o fallback Operators.OperatorIntrinsics.RangeInt64.

Os outros casos executam aproximadamente semelhante:

A razão pela qual o desempenho diminui para execuções de teste maiores é que a sobrecarga de invocar o action está crescendo à medida que trabalhamos cada vez menos em action.

Fazer um loop para ‘0’ às vezes pode oferecer benefícios de desempenho, pois pode economizar um registrador da CPU, mas neste caso a CPU tem registradores de sobra, então não parece para fazer a diferença.

Conclusão

Medir é importante porque, caso contrário, poderíamos pensar que todas essas alternativas são equivalentes, mas algumas alternativas são ~270x mais lentas que outras.

A etapa de verificação envolve engenharia reversa, o executável nos ajuda explique por que obtivemos ou não o desempenho que esperávamos. Além disso, A verificação pode nos ajudar a prever o desempenho nos casos em que é muito difícil faça uma medição adequada.

É difícil prever o desempenho sempre Meça, sempre Verifique seu suposições de desempenho.

Usando a recursão de cauda para uma iteração eficiente #

Vindo de linguagens imperativas, muitos desenvolvedores se perguntam como escrever um for-loop que sai mais cedo, pois F# não suporta break, continue ou return. A resposta em F# é usar tail-recursion que é um e idiomática para iterar enquanto ainda oferece excelente desempenho.

Digamos que queremos implementar tryFind para List. Se F# suportasse return, escreveríamos tryFind um pouco assim:

let tryFind predicate vs =
  for v in vs do
    if predicate v then
      return Some v
  None

Isso não funciona em F#. Em vez disso, escrevemos a função usando a recursão da cauda:

let tryFind predicate vs =
  let rec loop = function
    | v::vs -> if predicate v then 
                   Some v 
               else 
                   loop vs
    | _ -> None
  loop vs

A recursão de cauda tem desempenho em F# porque quando o compilador F# detecta que uma função é recursiva de cauda, ​​ela reescreve a recursão em um enquanto-loop eficiente. Usando ILSpy podemos ver que isso é verdade para nossa função loop:

internal static FSharpOption<a> loop@3-10<a>(FSharpFunc<a, bool> predicate, FSharpList<a> _arg1)
{
  while (_arg1.TailOrNull != null)
  {
    FSharpList<a> fSharpList = _arg1;
    FSharpList<a> vs = fSharpList.TailOrNull;
    a v = fSharpList.HeadOrDefault;
    if (predicate.Invoke(v))
    {
      return FSharpOption<a>.Some(v);
    }
    FSharpFunc<a, bool> arg_2D_0 = predicate;
    _arg1 = vs;
    predicate = arg_2D_0;
  }
  return null;
}

Além de algumas atribuições desnecessárias (que esperamos que o JIT-er elimine), este é essencialmente um loop eficiente.

Além disso, a recursão de cauda é idiomática para F#, pois nos permite evitar o estado mutável. Considere uma função sum que soma todos os elementos em uma List. Uma primeira tentativa óbvia seria esta:

let sum vs =
  let mutable s = LanguagePrimitives.GenericZero
  for v in vs do
    s <- s + v
  s

Se reescrevermos o loop em recursão de cauda, ​​podemos evitar o estado mutável:

let sum vs =
  let rec loop s = function
    | v::vs -> loop (s + v) vs
    | _ -> s
  loop LanguagePrimitives.GenericZero vs

Para eficiência, o compilador F# transforma isso em um while-loop que usa o estado mutável.

Comparação de diferentes pipelines de dados F# #

Em F# existem muitas opções para criar pipelines de dados, por exemplo: List, Seq e Array.

Qual pipeline de dados é preferível do ponto de vista do uso da memória e do desempenho?

Para responder a isso, compararemos o desempenho e o uso de memória usando diferentes tubulações.

** Pipeline de dados**

Para medir a sobrecarga, usaremos um pipeline de dados com baixo custo de CPU por itens processados:

let seqTest n =
  Seq.init (n + 1) id
  |> Seq.map    int64
  |> Seq.filter (fun v -> v % 2L = 0L)
  |> Seq.map    ((+) 1L)
  |> Seq.sum

Vamos criar pipelines equivalentes para todas as alternativas e compará-las.

Vamos variar o tamanho de n, mas vamos deixar o número total de trabalho ser o mesmo.

Alternativas de pipeline de dados

Vamos comparar as seguintes alternativas:

  1. Código imperativo
  2. Matriz (não preguiçoso)
  3. Lista (não preguiçoso)
  4. LINQ (fluxo de pull lento)
  5. Seq (fluxo de pull lento)
  6. Nessos (fluxo lento de puxar/empurrar)
  7. PullStream (fluxo de pull simplista)
  8. PushStream (stream push simplista)

Embora não seja um pipeline de dados, compararemos com o código Imperativo, pois mais próximo de como a CPU executa o código. Deve ser o mais rápido possível maneira de calcular o resultado, permitindo-nos medir a sobrecarga de desempenho dos pipelines de dados.

Array e List calculam um Array/List completo em cada etapa, então esperamos sobrecarga de memória.

LINQ e Seq são ambos baseados em IEnumerable<'T> que é um fluxo de pull lento (puxar significa que o fluxo do consumidor está extraindo dados do fluxo do produtor). Nós portanto, espere que o desempenho e o uso de memória sejam idênticos.

Nessos é uma biblioteca de fluxo de alto desempenho que suporta push & pull (como Java Stream).

PullStream e PushStream são implementações simplistas de fluxos Pull e Push.

Resultados de desempenho da execução em: F# 4.0 - .NET 4.6.1 - x64

As barras mostram o tempo decorrido, quanto mais baixo melhor. A quantidade total de trabalho útil é o mesmo para todos os testes, então os resultados são comparáveis. Isso também significa que poucas execuções implicam em conjuntos de dados maiores.

Como de costume, ao medir um, veja resultados interessantes.

  1. O baixo desempenho da ‘Lista’ é comparado a outras alternativas para grandes conjuntos de dados. Isso pode ser devido a GC ou localidade de cache ruim.
  2. Desempenho do ‘Array’ melhor do que o esperado.
  3. LINQ tem um desempenho melhor que Seq, isso é inesperado porque ambos são baseados em IEnumerable<'T>. No entanto, o Seq internamente é baseado em uma implementação genérica para todos os algoritmos, enquanto o LINQ usa algoritmos especializados.
  4. O Push tem um desempenho melhor do que o Pull. Isso é esperado, pois o pipeline de dados push tem menos verificações
  5. Os pipelines de dados Push simplistas têm desempenho comparável ao Nessos. No entanto, Nessos suporta pull e paralelismo.
  6. Para pipelines de dados pequenos, o desempenho do Nessos pode ser degradado devido à sobrecarga de configuração dos pipelines.
  7. Como esperado, o código “Imperativo” teve o melhor desempenho

Contagem de coleção GC de execução em: F# 4.0 - .NET 4.6.1 - x64

As barras mostram o número total de contagens de coleta GC durante o teste, quanto menor, melhor. Esta é uma medida de quantos objetos são criados pelo pipeline de dados.

Como de costume, ao medir um, veja resultados interessantes.

  1. Espera-se que List crie mais objetos do que Array porque uma List é essencialmente uma única lista vinculada de nós. Um array é uma área de memória contínua.
  2. Observando os números subjacentes, tanto List quanto Array forçam coleções de 2 gerações. Esse tipo de coleção é caro.
  3. Seq está provocando uma quantidade surpreendente de cobranças. É surpreendentemente ainda pior do que ‘List’ a este respeito.
  4. LINQ, Nessos, Push e Pull não acionam coleções para poucas execuções. No entanto, os objetos são alocados para que o GC eventualmente tenha que ser executado.
  5. Como esperado, uma vez que o código Imperativo não aloca objetos, nenhuma coleção GC foi acionada.

Conclusão

Todos os pipelines de dados fazem a mesma quantidade de trabalho útil em todos os casos de teste, mas vemos diferenças significativas no desempenho e no uso de memória entre os diferentes pipelines.

Além disso, notamos que a sobrecarga dos pipelines de dados difere dependendo o tamanho dos dados processados. Por exemplo, para tamanhos pequenos, Array está funcionando muito bem.

Deve-se ter em mente a quantidade de trabalho realizado em cada etapa do pipeline é muito pequeno para medir a sobrecarga. Em situações “reais” a sobrecarga de Seq pode não importar porque o trabalho real é mais demorado.

De maior preocupação são as diferenças de uso de memória. GC não é gratuito e é benéfico para aplicativos de longa duração para manter a pressão do GC baixa.

Para desenvolvedores de F# preocupados com desempenho e uso de memória, é recomendado verificar fora Nessos Streams.

Se você precisa de um desempenho de alto nível, vale a pena considerar o código Imperativo estrategicamente colocado.

Finalmente, quando se trata de desempenho, não faça suposições. Medir e verificar.

module PushStream =
  type Receiver<'T> = 'T -> bool
  type Stream<'T>   = Receiver<'T> -> unit

  let inline filter (f : 'T -> bool) (s : Stream<'T>) : Stream<'T> =
    fun r -> s (fun v -> if f v then r v else true)

  let inline map (m : 'T -> 'U) (s : Stream<'T>) : Stream<'U> =
    fun r -> s (fun v -> r (m v))

  let inline range b e : Stream<int> =
    fun r ->
      let rec loop i = if i <= e && r i then loop (i + 1)
      loop b

  let inline sum (s : Stream<'T>) : 'T =
    let mutable state = LanguagePrimitives.GenericZero<'T>
    s (fun v -> state <- state + v; true)
    state

module PullStream =

  [<Struct>]
  [<NoComparison>]
  [<NoEqualityAttribute>]
  type Maybe<'T>(v : 'T, hasValue : bool) =
    member    x.Value        = v
    member    x.HasValue     = hasValue
    override  x.ToString ()  =
      if hasValue then
        sprintf "Just %A" v
      else
        "Nothing"

  let Nothing<'T>     = Maybe<'T> (Unchecked.defaultof<'T>, false)
  let inline Just v   = Maybe<'T> (v, true)

  type Iterator<'T> = unit -> Maybe<'T>
  type Stream<'T>   = unit -> Iterator<'T>

  let filter (f : 'T -> bool) (s : Stream<'T>) : Stream<'T> =
    fun () ->
      let i = s ()
      let rec pop () =
        let mv = i ()
        if mv.HasValue then
          let v = mv.Value
          if f v then Just v else pop ()
        else
          Nothing
      pop

  let map (m : 'T -> 'U) (s : Stream<'T>) : Stream<'U> =
    fun () ->
      let i = s ()
      let pop () =
        let mv = i ()
        if mv.HasValue then
          Just (m mv.Value)
        else
          Nothing
      pop

  let range b e : Stream<int> =
    fun () ->
      let mutable i = b
      fun () ->
        if i <= e then
          let p = i
          i <- i + 1
          Just p
        else
          Nothing

  let inline sum (s : Stream<'T>) : 'T =
    let i = s ()
    let rec loop state =
      let mv = i ()
      if mv.HasValue then
        loop (state + mv.Value)
      else
        state
    loop LanguagePrimitives.GenericZero<'T>

module PerfTest =

  open System.Linq
#if USE_NESSOS
  open Nessos.Streams
#endif

  let now =
    let sw = System.Diagnostics.Stopwatch ()
    sw.Start ()
    fun () -> sw.ElapsedMilliseconds

  let time n a =
    let inline cc i       = System.GC.CollectionCount i

    let v                 = a ()

    System.GC.Collect (2, System.GCCollectionMode.Forced, true)

    let bcc0, bcc1, bcc2  = cc 0, cc 1, cc 2
    let b                 = now ()

    for i in 1..n do
      a () |> ignore

    let e = now ()
    let ecc0, ecc1, ecc2  = cc 0, cc 1, cc 2

    v, (e - b), ecc0 - bcc0, ecc1 - bcc1, ecc2 - bcc2

  let arrayTest n =
    Array.init (n + 1) id
    |> Array.map    int64
    |> Array.filter (fun v -> v % 2L = 0L)
    |> Array.map    ((+) 1L)
    |> Array.sum

  let imperativeTest n =
    let rec loop s i =
      if i >= 0L then
        if i % 2L = 0L then
          loop (s + i + 1L) (i - 1L)
        else
          loop s (i - 1L)
      else
        s
    loop 0L (int64 n)

  let linqTest n =
    (((Enumerable.Range(0, n + 1)).Select int64).Where(fun v -> v % 2L = 0L)).Select((+) 1L).Sum()

  let listTest n =
    List.init (n + 1) id
    |> List.map     int64
    |> List.filter  (fun v -> v % 2L = 0L)
    |> List.map     ((+) 1L)
    |> List.sum

#if USE_NESSOS
  let nessosTest n =
    Stream.initInfinite id
    |> Stream.take    (n + 1)
    |> Stream.map     int64
    |> Stream.filter  (fun v -> v % 2L = 0L)
    |> Stream.map     ((+) 1L)
    |> Stream.sum
#endif

  let pullTest n =
    PullStream.range 0 n
    |> PullStream.map     int64
    |> PullStream.filter  (fun v -> v % 2L = 0L)
    |> PullStream.map     ((+) 1L)
    |> PullStream.sum

  let pushTest n =
    PushStream.range 0 n
    |> PushStream.map     int64
    |> PushStream.filter  (fun v -> v % 2L = 0L)
    |> PushStream.map     ((+) 1L)
    |> PushStream.sum

  let seqTest n =
    Seq.init (n + 1) id
    |> Seq.map    int64
    |> Seq.filter (fun v -> v % 2L = 0L)
    |> Seq.map    ((+) 1L)
    |> Seq.sum

  let perfTest (path : string) =
    let testCases =
      [|
        "array"       , arrayTest       
        "imperative"  , imperativeTest  
        "linq"        , linqTest        
        "list"        , listTest        
        "seq"         , seqTest         
#if USE_NESSOS
        "nessos"      , nessosTest      
#endif
        "pull"        , pullTest        
        "push"        , pushTest        
      |]
    use out                   = new System.IO.StreamWriter (path)
    let write (msg : string)  = out.WriteLine msg
    let writef fmt            = FSharp.Core.Printf.kprintf write fmt

    write "Name\tTotal\tOuter\tInner\tElapsed\tCC0\tCC1\tCC2\tResult"

    let total   = 10000000
    let outers = [| 10; 1000; 1000000 |]
    for outer in outers do
      let inner = total / outer
      for name, a in testCases do
        printfn "Running %s with total=%d, outer=%d, inner=%d ..." name total outer inner
        let v, ms, cc0, cc1, cc2 = time outer (fun () -> a inner)
        printfn "  ... %d ms, cc0=%d, cc1=%d, cc2=%d, result=%A" ms cc0 cc1 cc2 v
        writef "%s\t%d\t%d\t%d\t%d\t%d\t%d\t%d\t%d" name total outer inner ms cc0 cc1 cc2 v

[<EntryPoint>]
let main argv =
  System.Environment.CurrentDirectory <- System.AppDomain.CurrentDomain.BaseDirectory
  PerfTest.perfTest "perf.tsv"
