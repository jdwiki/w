Informações básicas e concatenação de strings #

Strings em JavaScript podem ser colocadas entre aspas simples 'hello', aspas duplas "Hello" e (de ES2015, ES6) em literais de modelo (backticks) `hello`.

var hello  = "Hello";
var world  = 'world';
var helloW = `Hello World`;              // ES2015 / ES6

Strings podem ser criadas a partir de outros tipos usando a função String().

var intString = String(32); // "32"
var booleanString = String(true); // "true"
var nullString = String(null); // "null"

Ou toString() pode ser usado para converter Números, Booleanos ou Objetos em Strings.

var intString = (5232).toString(); // "5232"
var booleanString = (false).toString(); // "false"
var objString = ({}).toString(); // "[object Object]"

Strings também podem ser criadas usando o método String.fromCharCode.

String.fromCharCode(104,101,108,108,111) //"hello"

Criar um objeto String usando a palavra-chave new é permitido, mas não é recomendado, pois ele se comporta como Objetos, ao contrário de strings primitivas.

var objectString = new String("Yes, I am a String object");
typeof objectString;//"object"
typeof objectString.valueOf();//"string"

Concatenando Strings #

A concatenação de strings pode ser feita com o operador de concatenação +, ou com o método embutido concat() no protótipo do objeto String.

var foo = "Foo";
var bar = "Bar";
console.log(foo + bar);        // => "FooBar"
console.log(foo + " " + bar);  // => "Foo Bar"

foo.concat(bar)            // => "FooBar"
"a".concat("b", " ", "d")  // => "ab d"

Strings podem ser concatenadas com variáveis ​​não string, mas converterão as variáveis ​​não string em strings.

var string = "string";
var number = 32;
var boolean = true;

console.log(string + number + boolean); // "string32true"

Modelos de Strings #

Strings podem ser criadas usando literais de modelo (backticks) `hello`.

var greeting = `Hello`;

Com literais de modelo, você pode fazer interpolação de string usando ${variable} dentro de literais de modelo:

var place = `World`;
var greet = `Hello ${place}!`

console.log(greet); // "Hello World!"

Você pode usar String.raw para obter barras invertidas na string sem modificação.

`a\\b` // =  a\b
String.raw`a\\b` // = a\\b

Cadeia Reversa #

A maneira mais “popular” de reverter uma string em JavaScript é o seguinte fragmento de código, que é bastante comum:

function reverseString(str) {
    return str.split('').reverse().join('');
}

reverseString('string');    // "gnirts"

No entanto, isso funcionará apenas enquanto a string que está sendo invertida não contém pares substitutos. Símbolos astrais, ou seja, caracteres fora do plano multilíngue básico, podem ser representados por duas unidades de código e levarão essa técnica ingênua a produzir resultados errados. Além disso, caracteres com marcas de combinação (por exemplo, trema) aparecerão no caractere lógico “próximo” em vez do original com o qual foi combinado.

'𝌆■.'.split('').reverse().join(''); //fails

Embora o método funcione bem para a maioria dos idiomas, um algoritmo que respeite a codificação verdadeiramente preciso para a reversão de strings é um pouco mais complicado. Uma dessas implementações é uma pequena biblioteca chamada Esrever, que usa expressões regulares para combinar marcas de combinação e pares substitutos para realizar a reversão perfeitamente.

Explicação #

Seção Explicação Resultado
str A string de entrada "cadeia de caracteres"
String.prototype.split(deliminator) Divide a string str em um array. O parâmetro "" significa dividir entre cada caractere. ["s","t","r","i","n","g"]
Array.prototype.reverse() Retorna a matriz da string dividida com seus elementos na ordem inversa. ["g","n","i","r","t","s"]
Array.prototype.join(deliminator) Une os elementos da matriz em uma string. O parâmetro "" significa um delimitador vazio (ou seja, os elementos do array são colocados um ao lado do outro). "Grandes"

Usando o operador de spread

function reverseString(str) {
    return [...String(str)].reverse().join('');    
}

console.log(reverseString('stackoverflow'));  // "wolfrevokcats"
console.log(reverseString(1337));             // "7331"
console.log(reverseString([1, 2, 3]));        // "3,2,1"

Função reverse() personalizada

function reverse(string) {
    var strRev = "";
    for (var i = string.length - 1; i >= 0; i--) {
        strRev += string[i];
    }
    return strRev; 
}

reverse("zebra");  // "arbez"

Comparando Strings Lexicograficamente #

Para comparar strings em ordem alfabética, use localeCompare(). Isso retorna um valor negativo se a string de referência estiver lexicograficamente (em ordem alfabética) antes da string comparada (o parâmetro), um valor positivo se vier depois e um valor de ‘0’ se forem iguais.

var a = "hello";
var b = "world";

console.log(a.localeCompare(b)); // -1

Os operadores > e < também podem ser usados ​​para comparar strings lexicograficamente, mas não podem retornar um valor zero (isso pode ser testado com o operador de igualdade ==). Como resultado, uma forma da função localeCompare() pode ser escrita assim:

function strcmp(a, b) {
    if(a === b) {
        return 0;
    }

    if (a > b) {
        return 1;
    }

    return -1;
}

console.log(strcmp("hello", "world")); // -1
console.log(strcmp("hello", "hello")); //  0
console.log(strcmp("world", "hello")); //  1

Isso é especialmente útil ao usar uma função de classificação que compara com base no sinal do valor de retorno (como sort).

var arr = ["bananas", "cranberries", "apples"];
arr.sort(function(a, b) {
    return a.localeCompare(b);
});
console.log(arr); // [ "apples", "bananas", "cranberries" ]

Acessar o caractere no índice na string #

Use charAt() para obter um caractere no índice especificado na string.

var string = "Hello, World!";
console.log( string.charAt(4) ); // "o"

Como alternativa, como as strings podem ser tratadas como arrays, use o índice por meio de notação de colchetes.

var string = "Hello, World!";
console.log( string[4] ); // "o"

Para obter o código do caractere em um índice especificado, use charCodeAt() .

var string = "Hello, World!";
console.log( string.charCodeAt(4) ); // 111

Observe que esses métodos são todos métodos getter (retornam um valor). Strings em JavaScript são imutáveis. Em outras palavras, nenhum deles pode ser usado para definir um caractere em uma posição na string.

Escapando citações #

Se sua string estiver entre aspas simples, você precisa escapar da aspa literal interna com barra invertida \

var text = 'L\'albero means tree in Italian';
console.log( text ); \\ "L'albero means tree in Italian"

O mesmo vale para aspas duplas:

var text = "I feel \"high\"";

Atenção especial deve ser dada às citações de escape se você estiver armazenando representações HTML em uma String, pois as strings HTML fazem grande uso de citações, ou seja, em atributos:

var content = "<p class=\"special\">Hello World!</p>";        // valid String
var hello   = '<p class="special">I\'d like to say "Hi"</p>'; // valid String

Aspas em strings HTML também podem ser representadas usando &apos; (ou &#39;) como aspas simples e &quot; (ou &#34;) como aspas duplas.

var hi    = "<p class='special'>I'd like to say &quot;Hi&quot;</p>"; // valid String
var hello = '<p class="special">I&apos;d like to say "Hi"</p>';      // valid String

Observação: O uso de &apos; e &quot; não substituirá aspas duplas que os navegadores podem colocar automaticamente em aspas de atributos. Por exemplo <p class=special> sendo feito para <p class="special">, usando &quot; pode levar a <p class=""special""> onde \" seja <p class="special">.

Se uma string tem ' e " você pode querer considerar o uso de literais de modelo (também conhecidos como strings de modelo nas edições anteriores do ES6), que não requerem que você escape ' e ". Eles usam acentos graves (`) em vez de aspas simples ou duplas.

var x = `"Escaping " and ' can become very annoying`;

Aparar espaço em branco #

Para cortar espaços em branco das bordas de uma string, use String.prototype.trim:

"    some whitespaced string  ".trim();  // "some whitespaced string"

Muitos mecanismos JavaScript, mas não o Internet Explorer, implementaram trimLeft não padrão e métodos trimRight. Existe uma proposta, atualmente na Fase 1 do processo, para métodos padronizados trimStart e trimEnd, alias para trimLeft e trimRight para compatibilidade.

// Stage 1 proposal
"    this is me    ".trimStart();  // "this is me    "
"    this is me    ".trimEnd();  // "    this is me"

// Non-standard methods, but currently implemented by most engines
"    this is me    ".trimLeft();  // "this is me    "
"    this is me    ".trimRight();  // "    this is me"

Dividindo uma string em um array #

Use .split para ir de strings para um array de substrings divididas:

var s = "one, two, three, four, five"
s.split(", ");  // ["one", "two", "three", "four", "five"]

Use o método array .join para voltar para uma string:

s.split(", ").join("--");  // "one--two--three--four--five"

Todas as strings JavaScript são unicode!

var s = "some ∆≈ƒ unicode ¡™£¢¢¢";
s.charCodeAt(5);  // 8710

Não há bytes brutos ou strings binárias em JavaScript. Para manipular dados binários com eficiência, use [Typed Arrays](https://www.wikiod.com/pt/javascript/dados-binarios#Usando TypedArrays).

Detectando uma string #

Para detectar se um parâmetro é uma string primitiva, use typeof:

var aString = "my string";
var anInt = 5;
var anObj = {};
typeof aString === "string";   // true
typeof anInt === "string";     // false
typeof anObj === "string";     // false

Se você já teve um objeto String, via new String("somestr"), então o acima não funcionará. Neste caso, podemos usar instanceof:

var aStringObj = new String("my string");
aStringObj instanceof String;    // true

Para cobrir ambas as instâncias, podemos escrever uma função auxiliar simples:

var isString = function(value) {
    return typeof value === "string" || value instanceof String;
};

var aString = "Primitive String";
var aStringObj = new String("String Object");
isString(aString); // true
isString(aStringObj); // true
isString({}); // false
isString(5); // false

Ou podemos usar a função toString de Object. Isso pode ser útil se tivermos que verificar outros tipos também, digamos, em uma instrução switch, pois esse método também suporta outros tipos de dados, como typeof.

var pString = "Primitive String";
var oString = new String("Object Form of String");
Object.prototype.toString.call(pString);//"[object String]"
Object.prototype.toString.call(oString);//"[object String]"

Funções de busca e substituição de strings #

Para procurar uma string dentro de uma string, existem várias funções:

indexOf( searchString ) e [lastIndexOf( searchString )](https ://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/String/lastIndexOf) #

indexOf() retornará o índice da primeira ocorrência de searchString na string. Se searchString não for encontrado, então -1 será retornado.

var string = "Hello, World!";
console.log( string.indexOf("o") ); // 4
console.log( string.indexOf("foo") ); // -1

Da mesma forma, lastIndexOf() retornará o índice da última ocorrência de searchstring ou -1 se não for encontrado.

var string = "Hello, World!";
console.log( string.lastIndexOf("o") );   // 8
console.log( string.lastIndexOf("foo") ); // -1

includes( searchString, start ) #

includes() retornará um booleano que informa se searchString existe na string, começando do índice start (o padrão é 0). Isso é melhor que indexOf() se você simplesmente precisar testar a existência de uma substring.

var string = "Hello, World!";
console.log( string.includes("Hello") ); // true
console.log( string.includes("foo") );   // false

replace( regexp|substring, replace|replaceFunction ) #

replace() retornará uma string que tem todas as ocorrências de substrings correspondentes ao RegExp regexp ou string substring com uma string replacement ou o valor retornado de replaceFunction.

Observe que isso não modifica a string no lugar, mas retorna a string com substituições.

var string = "Hello, World!";
string = string.replace( "Hello", "Bye" );
console.log( string ); // "Bye, World!"

string = string.replace( /W.{3}d/g, "Universe" );
console.log( string ); // "Bye, Universe!"

replaceFunction pode ser usado para substituições condicionais para objetos de expressão regular (ou seja, com uso com regexp). Os parâmetros estão na seguinte ordem:

Parâmetro Significado
combinar a substring que corresponde a toda a expressão regularg
g1, g2, g3, … os grupos correspondentes na expressão regular
deslocamento o deslocamento da correspondência em toda a string
string toda a cadeia

Observe que todos os parâmetros são opcionais.

var string = "heLlo, woRlD!";
string = string.replace( /([a-zA-Z])([a-zA-Z]+)/g, function(match, g1, g2) {
    return g1.toUpperCase() + g2.toLowerCase();
}); 
console.log( string ); // "Hello, World!"

Representações de String de Números #

JavaScript tem conversão nativa de Number para sua String representação para qualquer base de 2 a 36.

A representação mais comum após decimal (base 10) é hexadecimal (base 16), mas o conteúdo desta seção funciona para todas as bases do intervalo.

Para converter um Number de decimal (base 10) para hexadecimal (base 16) representação de string o método toString pode ser usado com radix 16.

// base 10 Number
var b10 = 12;

// base 16 String representation
var b16 = b10.toString(16); // "c"

Se o número representado for um inteiro, a operação inversa para isso pode ser feita com parseInt e a radix 16 novamente

// base 16 String representation
var b16 = 'c';

// base 10 Number
var b10 = parseInt(b16, 16); // 12

Para converter um número arbitrário (ou seja, não inteiro) de sua representação de String em um Number, a operação deve ser dividida em duas partes; a parte inteira e a parte fracionária.

let b16 = '3.243f3e0370cdc';
// Split into integer and fraction parts
let [i16, f16] = b16.split('.');

// Calculate base 10 integer part
let i10 = parseInt(i16, 16); // 3

// Calculate the base 10 fraction part
let f10 = parseInt(f16, 16) / Math.pow(16, f16.length); // 0.14158999999999988

// Put the base 10 parts together to find the Number
let b10 = i10 + f10; // 3.14159

Nota 1: Cuidado, pois pequenos erros podem ocorrer no resultado devido a diferenças no que é possível representar em diferentes bases. Pode ser desejável realizar algum tipo de arredondamento posteriormente. Observação 2: Representações muito longas de números também podem resultar em erros devido à precisão e aos valores máximos de Numbers do ambiente em que as conversões estão ocorrendo.

Código do caractere #

O método charCodeAt recupera o código de caractere Unicode de um único caractere:

var charCode = "µ".charCodeAt(); // The character code of the letter µ is 181

Para obter o código de caractere de um caractere em uma string, a posição do caractere baseada em 0 é passada como parâmetro para charCodeAt:

var charCode = "ABCDE".charCodeAt(3); // The character code of "D" is 68

Alguns símbolos Unicode não cabem em um único caractere e, em vez disso, exigem dois pares substitutos UTF-16 para serem codificados. Este é o caso de códigos de caracteres além de 216 - 1 ou 63553. Esses códigos de caracteres estendidos ou valores de code point podem ser recuperados com codePointAt:

// The Grinning Face Emoji has code point 128512 or 0x1F600
var codePoint = "😀".codePointAt();

console.log('qwerty'.toUpperCase()); // 'QWERTY'

Encontre o índice de uma substring dentro de uma string #

O método .indexOf retorna o índice de uma substring dentro de outra string (se existir, ou -1 caso contrário)

'Hellow World'.indexOf('Wor');    // 7

.indexOf também aceita um argumento numérico adicional que indica em qual índice a função deve começar a procurar

"harr dee harr dee harr".indexOf("dee", 10); // 14

Você deve notar que .indexOf diferencia maiúsculas de minúsculas

 'Hellow World'.indexOf('WOR');    // -1

Repetir uma string #

Isso pode ser feito usando o método .repeat():

"abc".repeat(3);  // Returns "abcabcabc"
"abc".repeat(0);  // Returns ""
"abc".repeat(-1); // Throws a RangeError

No caso geral, isso deve ser feito usando um polyfill correto para o método ES6 String.prototype.repeat(). Caso contrário, o idioma new Array(n + 1).join(myString) pode repetir n vezes a string myString:

var myString = "abc";
var n = 3;

new Array(n + 1).join(myString);  // Returns "abcabcabc"
