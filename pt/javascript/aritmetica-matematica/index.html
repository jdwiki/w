Restante / Módulo (%) #

O operador resto/módulo (%) retorna o resto após a divisão (inteiro).

console.log( 42 %  10); //  2
console.log( 42 % -10); //  2
console.log(-42 %  10); // -2
console.log(-42 % -10); // -2
console.log(-40 %  10); // -0
console.log( 40 %  10); //  0

Este operador retorna o restante quando um operando é dividido por um segundo operando. Quando o primeiro operando for um valor negativo, o valor de retorno será sempre negativo e vice-versa para valores positivos.

No exemplo acima, 10 pode ser subtraído quatro vezes de 42 antes que não haja o suficiente para subtrair novamente sem mudar de sinal. O resto é assim: 42 - 4 * 10 = 2.

O operador restante pode ser útil para os seguintes problemas:

  1. Teste se um número inteiro é (não) divisível por outro número:

     x % 4 == 0 // true if x is divisible by 4
 x % 2 == 0 // true if x is even number
 x % 2 != 0 // true if x is odd number

Como 0 === -0, isso também funciona para x <= -0.

  1. Implemente o incremento/decremento cíclico do valor dentro do intervalo [0, n).

Suponha que precisamos incrementar o valor inteiro de 0 para (mas não incluindo) n, de modo que o próximo valor após n-1 se torne 0. Isso pode ser feito por tal pseudocódigo:

var n = ...; // given n
var i = 0;
function inc() {
    i = (i + 1) % n;
}
while (true) {
    inc();
    // update something with i
}

Agora generalize o problema acima e suponha que precisamos permitir tanto incrementar quanto decrementar esse valor de 0 para (não incluindo) n, de modo que o próximo valor após n-1 se torne 0 e o valor anterior antes de 0 se tornar n-1.

var n = ...; // given n
var i = 0;
function delta(d) { // d - any signed integer
    i = (i + d + n) % n; // we add n to (i+d) to ensure the sum is positive
}

Agora podemos chamar a função delta() passando qualquer inteiro, tanto positivo quanto negativo, como parâmetro delta.

Usando o módulo para obter a parte fracionária de um número #

 var myNum = 10 / 4;       // 2.5
 var fraction = myNum % 1; // 0.5
 myNum = -20 / 7;          // -2.857142857142857
 fraction = myNum % 1;     // -0.857142857142857

Arredondamento #

Arredondamento #

Math.round() arredondará o valor para o inteiro mais próximo usando half round up para desempate.

var a = Math.round(2.3);       // a is now 2  
var b = Math.round(2.7);       // b is now 3
var c = Math.round(2.5);       // c is now 3

Mas

var c = Math.round(-2.7);       // c is now -3
var c = Math.round(-2.5);       // c is now -2

Observe como -2.5 é arredondado para -2. Isso ocorre porque os valores intermediários são sempre arredondados para cima, ou seja, eles são arredondados para o inteiro com o próximo valor mais alto.

Arredondamento para cima #

Math.ceil() arredondará o valor para cima.

var a = Math.ceil(2.3);        // a is now 3
var b = Math.ceil(2.7);        // b is now 3

teto de um número negativo será arredondado para zero

var c = Math.ceil(-1.1);       // c is now 1

Arredondando para baixo #

Math.floor() arredondará o valor para baixo.

var a = Math.floor(2.3);        // a is now 2
var b = Math.floor(2.7);        // b is now 2

pisar um número negativo o arredondará para longe de zero.

var c = Math.floor(-1.1);       // c is now -1

Truncando #

Aviso: o uso de operadores bit a bit (exceto >>>) se aplica apenas a números entre -2147483649 e 2147483648.

2.3  | 0;                       // 2 (floor)
-2.3 | 0;                       // -2 (ceil)
NaN  | 0;                       // 0

Math.trunc()

Math.trunc(2.3);                // 2 (floor)
Math.trunc(-2.3);               // -2 (ceil)
Math.trunc(2147483648.1);       // 2147483648 (floor)
Math.trunc(-2147483649.1);      // -2147483649 (ceil)
Math.trunc(NaN);                // NaN

Arredondamento para casas decimais #

Math.floor, Math.ceil() e Math.round() podem ser usados ​​para arredondar para um número de casas decimais

Para arredondar para 2 casas decimais:

 var myNum = 2/3;               // 0.6666666666666666
 var multiplier = 100;
 var a = Math.round(myNum * multiplier) / multiplier;  // 0.67
 var b = Math.ceil (myNum * multiplier) / multiplier;  // 0.67
 var c = Math.floor(myNum * multiplier) / multiplier;  // 0.66

Você também pode arredondar para um número de dígitos:

 var myNum = 10000/3;           // 3333.3333333333335
 var multiplier = 1/100;
 var a = Math.round(myNum * multiplier) / multiplier;  // 3300
 var b = Math.ceil (myNum * multiplier) / multiplier;  // 3400
 var c = Math.floor(myNum * multiplier) / multiplier;  // 3300

Como uma função mais útil:

 // value is the value to round
 // places if positive the number of decimal places to round to
 // places if negative the number of digits to round to
 function roundTo(value, places){
     var power = Math.pow(10, places);
     return Math.round(value * power) / power;
 }
 var myNum = 10000/3;    // 3333.3333333333335
 roundTo(myNum, 2);  // 3333.33
 roundTo(myNum, 0);  // 3333
 roundTo(myNum, -2); // 3300

E as variantes para ceil e floor:

 function ceilTo(value, places){
     var power = Math.pow(10, places);
     return Math.ceil(value * power) / power;
 }
 function floorTo(value, places){
     var power = Math.pow(10, places);
     return Math.floor(value * power) / power;
 }

Trigonometria #

Todos os ângulos abaixo estão em radianos. Um ângulo r em radianos tem medida 180 * r / Math.PI em graus.

Seno #

Math.sin(r);

Isso retornará o seno de r, um valor entre -1 e 1.

Math.asin(r);

Isso retornará o arco seno (o reverso do seno) de r.

Math.asinh(r)

Isso retornará o arco seno hiperbólico de r.

Cosseno #

Math.cos(r);

Incrementando (++) #

O operador Increment (++) incrementa seu operando em um.

  • Se usado como um postfix, retorna o valor antes de incrementar.
  • Se usado como prefixo, retorna o valor após o incremento.
//postfix
var a = 5,    // 5
    b = a++,  // 5
    c = a     // 6

Neste caso, a é incrementado após definir b. Então, b será 5 e c será 6.

//prefix
var a = 5,    // 5
    b = ++a,  // 6
    c = a     // 6

Neste caso, a é incrementado antes de definir b. Então, b será 6 e c será 6.

Os operadores de incremento e decremento são comumente usados ​​em loops for, por exemplo:

for(var i = 0; i < 42; ++i)
{
  // do something awesome!
}

Observe como a variante prefix é usada. Isso garante que uma variável temporária não seja criada desnecessariamente (para retornar o valor anterior à operação).

Operadores bit a bit #

Observe que todas as operações bit a bit operam em inteiros de 32 bits passando quaisquer operandos para a função interna ToInt32.

Bit a bit ou #

var a;
a = 0b0011 | 0b1010; // a === 0b1011
// truth table
// 1010 | (or)
// 0011  
// 1011  (result)

Bitwise and #

a = 0b0011 & 0b1010; // a === 0b0010
// truth table
// 1010 & (and)
// 0011  
// 0010  (result)

Bitwise not #

a = ~0b0011; // a === 0b1100
// truth table
// 10 ~(not)
// 01  (result)

Bitwise xor (exclusivo ou) #

a = 0b1010 ^ 0b0011; // a === 0b1001
// truth table
// 1010 ^ (xor)
// 0011  
// 1001  (result)

Bitwise shift para a esquerda #

a = 0b0001 << 1; // a === 0b0010
a = 0b0001 << 2; // a === 0b0100
a = 0b0001 << 3; // a === 0b1000

Deslocar para a esquerda é equivalente a multiplicar inteiro por Math.pow(2, n). Ao fazer matemática inteira, o deslocamento pode melhorar significativamente a velocidade de algumas operações matemáticas.

var n = 2;
var a = 5.4;
var result = (a << n) === Math.floor(a) * Math.pow(2,n);
// result is true
a = 5.4 << n; // 20

Bitwise shift para a direita >> (Shift de propagação de sinal) >>> (Deslocamento para a direita com preenchimento zero) #

a = 0b1001 >> 1; // a === 0b0100
a = 0b1001 >> 2; // a === 0b0010
a = 0b1001 >> 3; // a === 0b0001

a = 0b1001 >>> 1; // a === 0b0100
a = 0b1001 >>> 2; // a === 0b0010
a = 0b1001 >>> 3; // a === 0b0001

Um valor negativo de 32 bits sempre tem o bit mais à esquerda ligado:

a = 0b11111111111111111111111111110111 | 0;   
console.log(a); // -9
b = a >> 2;     // leftmost bit is shifted 1 to the right then new left most bit is set to on (1)
console.log(b); // -3
b = a >>> 2;    // leftmost bit is shifted 1 to the right. the new left most bit is set to off (0)
console.log(b); // 2147483643

O resultado de uma operação >>> é sempre positivo. O resultado de um >> é sempre o mesmo sinal do valor deslocado.

Deslocamento para a direita em números positivos é o equivalente a dividir pelo Math.pow(2,n) e reduzir o resultado:

a = 256.67;
n = 4;
result = (a >> n) === Math.floor( Math.floor(a) / Math.pow(2,n) );
// result is true
a = a >> n; //  16

result = (a >>> n) === Math.floor( Math.floor(a) / Math.pow(2,n) );
// result is true
a = a >>> n; //  16

O preenchimento zero com deslocamento para a direita (>>>) em números negativos é diferente. Como o JavaScript não converte em inteiros não assinados ao fazer operações de bits, não há equivalente operacional:

a = -256.67;
result = (a >>> n) === Math.floor( Math.floor(a) / Math.pow(2,n) );
// result is false

Simulando eventos com diferentes probabilidades #

Às vezes, você pode precisar apenas simular um evento com dois resultados, talvez com probabilidades diferentes, mas pode se encontrar em uma situação que exige muitos resultados possíveis com probabilidades diferentes. Vamos imaginar que você queira simular um evento que tenha seis resultados igualmente prováveis. Isso é bem simples.

function simulateEvent(numEvents) {
    var event = Math.floor(numEvents*Math.random());
    return event;
}

// simulate fair die
console.log("Rolled a "+(simulateEvent(6)+1));  // Rolled a 2

No entanto, você pode não querer resultados igualmente prováveis. Digamos que você tenha uma lista de três resultados representados como uma matriz de probabilidades em porcentagens ou múltiplos de probabilidade. Um exemplo pode ser um dado ponderado. Você pode reescrever a função anterior para simular tal evento.

function simulateEvent(chances) {
    var sum = 0;
    chances.forEach(function(chance) {
        sum+=chance;
    });
    var rand = Math.random();
    var chance = 0;
    for(var i=0; i<chances.length; i++) {
        chance+=chances[i]/sum;
        if(rand<chance) {
            return i;
        }
    }
    
    // should never be reached unless sum of probabilities is less than 1
    // due to all being zero or some being negative probabilities
    return -1;
}

// simulate weighted dice where 6 is twice as likely as any other face
// using multiples of likelihood
console.log("Rolled a "+(simulateEvent([1,1,1,1,1,2])+1));  // Rolled a 1

// using probabilities
console.log("Rolled a "+(simulateEvent([1/7,1/7,1/7,1/7,1/7,2/7])+1));  // Rolled a 6

Como você provavelmente notou, essas funções retornam um índice, para que você possa ter resultados mais descritivos armazenados em uma matriz. Aqui está um exemplo.

var rewards = ["gold coin","silver coin","diamond","god sword"];
var likelihoods = [5,9,1,0];
// least likely to get a god sword (0/15 = 0%, never),
// most likely to get a silver coin (9/15 = 60%, more than half the time)

// simulate event, log reward
console.log("You get a "+rewards[simulateEvent(likelihoods)]);  // You get a silver coin

O operador de decremento (--) decrementa os números em um.

Nesse caso, b é definido com o novo valor de a. Então, b será 4 e c será 4.

Aleatório com distribuição gaussiana #

A função Math.random() deve fornecer números aleatórios que tenham um desvio padrão próximo de 0. Ao escolher um baralho de cartas, ou simular um lançamento de dados, é isso que queremos.

Mas na maioria das situações isso é irreal. No mundo real, a aleatoriedade tende a se reunir em torno de um valor normal comum. Se plotado em um gráfico, você obtém a curva de sino clássica ou distribuição gaussiana.

Fazer isso com a função Math.random() é relativamente simples.

var randNum = (Math.random() + Math.random()) / 2; 
var randNum = (Math.random() + Math.random() + Math.random()) / 3; 
var randNum = (Math.random() + Math.random() + Math.random() + Math.random()) / 4;

Adicionar um valor aleatório ao último aumenta a variância dos números aleatórios. Dividir pelo número de vezes que você adiciona normaliza o resultado para um intervalo de 0–1

Como adicionar mais do que alguns aleatórios é confuso, uma função simples permitirá que você selecione uma variação desejada.

// v is the number of times random is summed and should be over >= 1
// return a random number between 0-1 exclusive
function randomG(v){ 
    var r = 0;
    for(var i = v; i > 0; i --){
        r += Math.random();
    }
    return r / v;
}

A imagem mostra a distribuição de valores aleatórios para diferentes valores de v. O canto superior esquerdo é o padrão único Math.random(), o canto inferior direito é o Math.random() somado 8 vezes. Isso é de 5.000.000 de amostras usando o Chrome

Sin & Cos para criar um vetor com direção e distância #

Se você tiver um vetor na forma polar (direção e distância), você desejará convertê-lo em um vetor cartesiano com um componente x e y. Para referência, o sistema de coordenadas da tela tem direções como pontos de 0 graus da esquerda para a direita, 90 (PI/2) apontam para baixo na tela e assim por diante no sentido horário.

var dir = 1.4536; // direction in radians
var dist = 200; // distance
var vec = {};
vec.x = Math.cos(dir) * dist; // get the x component
vec.y = Math.sin(dir) * dist; // get the y component

Você também pode ignorar a distância para criar um vetor normalizado (1 unidade de comprimento) na direção de dir

var dir = 1.4536; // direction in radians
var vec = {};
vec.x = Math.cos(dir); // get the x component
vec.y = Math.sin(dir); // get the y component

Se o seu sistema de coordenadas tiver y para cima, você precisará alternar cos e sin. Neste caso, uma direção positiva está no sentido anti-horário a partir do eixo x.

// get the directional vector where y points up
var dir = 1.4536; // direction in radians
var vec = {};
vec.x = Math.sin(dir); // get the x component
vec.y = Math.cos(dir); // get the y component

Funções periódicas usando Math.sin #

Math.sin e Math.cos são cíclicos com um período de 2*PI radianos (360 graus) eles emitem uma onda com uma amplitude de 2 na faixa de -1 a 1.

Gráfico da função seno e cosseno: (cortesia Wikipedia) [](https://en.wikipedia.org/wiki /Sine_wave)

Ambos são muito úteis para muitos tipos de cálculos periódicos, desde a criação de ondas sonoras até animações e até codificar e decodificar dados de imagem

Este exemplo mostra como criar uma onda senoidal simples com controle sobre período/frequência, fase, amplitude e deslocamento.

A unidade de tempo utilizada é segundos. A forma mais simples com controle apenas sobre a frequência.

// time is the time in seconds when you want to get a sample
// Frequency represents the number of oscillations per second
function oscillator(time, frequency){  
    return Math.sin(time * 2 * Math.PI * frequency);
}

Em quase todos os casos, você desejará fazer algumas alterações no valor retornado. Os termos comuns para modificações

  • Fase: O offset em termos de frequência desde o início das oscilações. É um valor na faixa de 0 a 1 onde o valor 0,5 move a onda para frente no tempo pela metade de sua frequência. Um valor de 0 ou 1 não faz nenhuma alteração.
  • Amplitude: A distância entre o valor mais baixo e o valor mais alto durante um ciclo. Uma amplitude de 1 tem um intervalo de 2. Do ponto mais baixo (vale) -1 ao mais alto (pico) 1. Para uma onda com frequência 1, o pico está em 0,25 segundos e o vale em 0,75.
  • Offset: move toda a onda para cima ou para baixo.

Para incluir tudo isso na função:

function oscillator(time, frequency = 1, amplitude = 1, phase = 0, offset = 0){
    var t = time * frequency * Math.PI * 2; // get phase at time
    t += phase * Math.PI * 2; // add the phase offset
    var v = Math.sin(t); // get the value at the calculated position in the cycle
    v *= amplitude; // set the amplitude
    v += offset; // add the offset
    return v;
}

Ou de uma forma mais compacta (e um pouco mais rápida):

function oscillator(time, frequency = 1, amplitude = 1, phase = 0, offset = 0){
    return Math.sin(time * frequency * Math.PI * 2 + phase * Math.PI * 2) * amplitude + offset; 
}

Todos os argumentos além do tempo são opcionais

Obtendo máximo e mínimo #

A função Math.max() retorna o maior de zero ou mais números.

Math.max(4, 12);   //  12
Math.max(-1, -15); // -1

A função Math.min() retorna o menor de zero ou mais números.

Math.min(4, 12);   //  4
Math.min(-1, -15); // -15

Obtendo máximo e mínimo de um array: #

var arr = [1, 2, 3, 4, 5, 6, 7, 8, 9],
    max = Math.max.apply(Math, arr),
    min = Math.min.apply(Math, arr);

console.log(max); // Logs: 9
console.log(min); // Logs: 1

ECMAScript 6 operador de propagação, obtendo o máximo e o mínimo de um array:

var arr = [1, 2, 3, 4, 5, 6, 7, 8, 9],
    max = Math.max(...arr),
    min = Math.min(...arr);

console.log(max); // Logs: 9
console.log(min); // Logs: 1

Restringir número ao intervalo mínimo/máximo #

Se você precisar prender um número para mantê-lo dentro de um limite de intervalo específico

function clamp(min, max, val) {
    return Math.min(Math.max(min, +val), max);
}

console.log(clamp(-10, 10, "4.30")); // 4.3
console.log(clamp(-10, 10, -8));     // -8
console.log(clamp(-10, 10, 12));     // 10
console.log(clamp(-10, 10, -15));    // -10

Obtendo raízes de um número #

Raiz quadrada #

Use Math.sqrt() para encontrar a raiz quadrada de um número

Math.sqrt(16)   #=> 4

Raiz cúbica #

Para encontrar a raiz cúbica de um número, use a função Math.cbrt()

Math.cbrt(27)  #=> 3

Encontrando raízes n #

Para encontrar a raiz n, use a função Math.pow() e passe um expoente fracionário.

Math.pow(64, 1/6) #=> 2
