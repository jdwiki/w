As matrizes tipadas foram originalmente especificadas por um rascunho do editor Khronos e posteriormente padronizadas em ECMAScript 6 [§24](http://www. ecma-international.org/ecma-262/6.0/#sec-structured-data) e §22.2 .

Os blobs são especificados pelo rascunho de trabalho da W3C File API.

Obtendo representação binária de um arquivo de imagem #

Este exemplo é inspirado por esta pergunta.

Vamos supor que você saiba como carregar um arquivo usando a API de arquivos.

// preliminary code to handle getting local file and finally printing to console
// the results of our function ArrayBufferToBinary().
var file = // get handle to local file.
var reader = new FileReader();
reader.onload = function(event) {
    var data = event.target.result;
    console.log(ArrayBufferToBinary(data));
};
reader.readAsArrayBuffer(file); //gets an ArrayBuffer of the file

Agora realizamos a conversão real dos dados do arquivo em 1’s e 0’s usando um DataView:

function ArrayBufferToBinary(buffer) {
   // Convert an array buffer to a string bit-representation: 0 1 1 0 0 0...
   var dataView = new DataView(buffer);
   var response = "", offset = (8/8); 
   for(var i = 0; i < dataView.byteLength; i += offset) {
       response += dataView.getInt8(i).toString(2); 
   }
   return response;
}

DataViews permitem ler/escrever dados numéricos; getInt8 converte os dados da posição do byte - aqui 0, o valor passado - no ArrayBuffer para representação inteira de 8 bits com sinal, e toString(2) converte o inteiro de 8 bits para representação binária formato (ou seja, uma cadeia de 1’s e 0’s).

Os arquivos são salvos como bytes. O valor de deslocamento ‘mágico’ é obtido observando que estamos pegando arquivos armazenados como bytes, ou seja, como inteiros de 8 bits e lendo-os em representação de inteiros de 8 bits. Se estivéssemos tentando ler nossos arquivos salvos em bytes (ou seja, 8 bits) para inteiros de 32 bits, observaríamos que 32/8 = 4 é o número de espaços de byte, que é nosso valor de deslocamento de byte.

Para esta tarefa, DataViews são um exagero. Eles são normalmente usados ​​nos casos em que são encontradas endianness ou heterogeneidade de dados (por exemplo, na leitura de arquivos PDF, que possuem cabeçalhos codificados em diferentes bases e gostaríamos de extrair esse valor de forma significativa). Porque queremos apenas uma representação textual, não nos importamos com a heterogeneidade, pois nunca há necessidade de

Uma solução muito melhor - e mais curta - pode ser encontrada usando um array do tipo UInt8Array, que trata todo o ArrayBuffer como composto de inteiros de 8 bits sem sinal:

function ArrayBufferToBinary(buffer) {
    var uint8 = new Uint8Array(buffer);
    return uint8.reduce((binary, uint8) => binary + uint8.toString(2), "");
}

Conversão entre Blobs e ArrayBuffers #

JavaScript tem duas maneiras principais de representar dados binários no navegador. ArrayBuffers/TypedArrays contêm dados binários mutáveis ​​(embora ainda de tamanho fixo) que você pode manipular diretamente. Blobs contêm dados binários imutáveis ​​que só podem ser acessados ​​por meio da interface de arquivo assíncrona.

Converta um Blob em um ArrayBuffer (assíncrono) #

var blob = new Blob(["\x01\x02\x03\x04"]),
    fileReader = new FileReader(),
    array;

fileReader.onload = function() {
    array = this.result;
    console.log("Array contains", array.byteLength, "bytes.");
};

fileReader.readAsArrayBuffer(blob);

Converta um Blob para um ArrayBuffer usando um Promise (assíncrono) #

var blob = new Blob(["\x01\x02\x03\x04"]);

var arrayPromise = new Promise(function(resolve) {
    var reader = new FileReader();

    reader.onloadend = function() {
        resolve(reader.result);
    };

    reader.readAsArrayBuffer(blob);
});

arrayPromise.then(function(array) {
    console.log("Array contains", array.byteLength, "bytes.");
});

Converta um ArrayBuffer ou array digitado em um Blob #

var array = new Uint8Array([0x04, 0x06, 0x07, 0x08]);

var blob = new Blob([array]);

Manipulando ArrayBuffers com DataViews #

DataViews fornecem métodos para ler e gravar valores individuais de um ArrayBuffer, em vez de visualizar a coisa inteira como uma matriz de um único tipo. Aqui definimos dois bytes individualmente e os interpretamos juntos como um inteiro sem sinal de 16 bits, primeiro big-endian e depois little-endian.

var buffer = new ArrayBuffer(2);
var view = new DataView(buffer);

view.setUint8(0, 0xFF);
view.setUint8(1, 0x01);

console.log(view.getUint16(0, false)); // 65281
console.log(view.getUint16(0, true));  // 511

Criando um TypedArray a partir de uma string Base64 #

var data = 
   'iVBORw0KGgoAAAANSUhEUgAAAAUAAAAFCAYAAACN' +
   'byblAAAAHElEQVQI12P4//8/w38GIAXDIBKE0DHx' +
   'gljNBAAO9TXL0Y4OHwAAAABJRU5ErkJggg==';

var characters = atob(data);

var array = new Uint8Array(characters.length);

for (var i = 0; i < characters.length; i++) {
  array[i] = characters.charCodeAt(i);
}

Usando TypedArrays #

TypedArrays são um conjunto de tipos que fornecem diferentes visualizações em ArrayBuffers binários mutáveis ​​de comprimento fixo. Na maioria das vezes, eles agem como Arrays que forçam todos os valores atribuídos a um determinado tipo numérico. Você pode passar uma instância ArrayBuffer para um construtor TypedArray para criar uma nova exibição de seus dados.

var buffer = new ArrayBuffer(8);
var byteView = new Uint8Array(buffer);
var floatView = new Float64Array(buffer);

console.log(byteView);  // [0, 0, 0, 0, 0, 0, 0, 0]
console.log(floatView); // [0]
byteView[0] = 0x01;
byteView[1] = 0x02;
byteView[2] = 0x04;
byteView[3] = 0x08;
console.log(floatView); // [6.64421383e-316]

ArrayBuffers podem ser copiados usando o método .slice(...), diretamente ou através de uma visão TypedArray.

var byteView2 = byteView.slice();
var floatView2 = new Float64Array(byteView2.buffer);
byteView2[6] = 0xFF;
console.log(floatView);  // [6.64421383e-316]
console.log(floatView2); // [7.06327456e-304]

Iterando através de um arrayBuffer #

Para uma maneira conveniente de iterar por meio de um arrayBuffer, você pode criar um iterador simples que implementa os métodos DataView sob o capô:

var ArrayBufferCursor = function() {
  var ArrayBufferCursor = function(arrayBuffer) {
    this.dataview = new DataView(arrayBuffer, 0);
    this.size = arrayBuffer.byteLength;
    this.index = 0;
  }

  ArrayBufferCursor.prototype.next = function(type) {
    switch(type) {
      case 'Uint8':
        var result = this.dataview.getUint8(this.index);
        this.index += 1;
        return result;
      case 'Int16':
        var result = this.dataview.getInt16(this.index, true);
        this.index += 2;
        return result;
      case 'Uint16':
        var result = this.dataview.getUint16(this.index, true);
        this.index += 2;
        return result;
      case 'Int32':
        var result = this.dataview.getInt32(this.index, true);
        this.index += 4;
        return result;
      case 'Uint32':
        var result = this.dataview.getUint32(this.index, true);
        this.index += 4;
        return result;
      case 'Float':
      case 'Float32':
        var result = this.dataview.getFloat32(this.index, true);
        this.index += 4;
        return result;
      case 'Double':
      case 'Float64':
        var result = this.dataview.getFloat64(this.index, true);
        this.index += 8;
        return result;
      default:
        throw new Error("Unknown datatype");
    }
  };

  ArrayBufferCursor.prototype.hasNext = function() {
    return this.index < this.size;
  }

  return ArrayBufferCursor;
});

Você pode então criar um iterador como este:

var cursor = new ArrayBufferCursor(arrayBuffer);

Você pode usar o hasNext para verificar se ainda há itens

for(;cursor.hasNext();) {
    // There's still items to process
}

Você pode usar o método next para obter o próximo valor:

var nextValue = cursor.next('Float');

Com esse iterador, escrever seu próprio analisador para processar dados binários se torna muito fácil.
