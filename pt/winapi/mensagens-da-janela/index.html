Sintaxe #

  • #include <windows.h>
  • BOOL WINAPI DestroyWindow(HWND hwnd);
  • VOID WINAPI PostQuitMessage(int exitcode);
  • BOOL WINAPI MoveWindow(HWND hwnd, int x, int y, int cx, int cy, BOOL bRepaint);

WM_DESTROY #

Esta mensagem é enviada ao seu procedimento de janela quando uma janela está sendo destruída. Ele é enviado depois que a janela é removida da tela. A maioria dos aplicativos libera quaisquer recursos, como memória ou handles, obtidos em WM_CREATE. Se você lidar com esta mensagem, retorne 0.

LRESULT CALLBACK winproc(HWND hwnd, UINT wm, WPARAM wp, LPARAM lp)
{
    static char *text;
    switch (wm) {
        case WM_CREATE:
            text = malloc(256);
            /* use the allocated memory */
            return 0;
        case WM_CLOSE:
             switch (MessageBox(hwnd, "Save changes?", "", MB_YESNOCANCEL)) {
                 case IDYES:
                      savedoc();
                                         /* fall through */
                 case IDNO:
                      DestroyWindow(hwnd);
                      break;
             }
             return 0;
        case WM_DESTROY:
            /* free the memory */
            free(text);
            PostQuitMessage(0);
            return 0;
    }
    return DefWindowProc(hwnd, wm, wp, lp);
}

WM_CLOSE #

Enviado quando o botão Fechar de um aplicativo é clicado. Não confunda isso com WM_DESTROY que é enviado quando uma janela será destruída. A principal diferença está no fato de que o fechamento pode ser cancelado em WM_CLOSE (pense no Microsoft Word pedindo para salvar suas alterações), versus a destruição é quando a janela já foi fechada (pense no Microsoft Word liberando memória).

LRESULT CALLBACK winproc(HWND hwnd, UINT wm, WPARAM wp, LPARAM lp)
{
    static char *text;
    switch (wm) {
        case WM_CREATE:
            text = malloc(256);
            /* use the allocated memory */
            return 0;
        case WM_CLOSE:
             switch (MessageBox(hwnd, "Save changes?", "", MB_YESNOCANCEL)) {
                 case IDYES:
                      savedoc();
                                         /* fall through */
                 case IDNO:
                      DestroyWindow(hwnd);
                      break;
             }
             return 0;
        case WM_DESTROY:
            /* free the memory */
            free(text);
            PostQuitMessage(0);
            return 0;
    }
    return DefWindowProc(hwnd, wm, wp, lp);
}

WM_COMMAND #

Enviado para um procedimento de janela quando:

  • o usuário seleciona um item de um menu
  • um controle envia uma notificação para sua janela pai
  • um pressionamento de tecla do acelerador é traduzido
Origem da mensagem HIWORD(wp) LOWORD(wp) lp
Cardápio 0 ID do menu (IDM_*) 0
Acelerador 1 ID de aceleração (IDM_*) 0
Controle código de notificação ID de controle HWND da janela de controle

Por exemplo, no Bloco de Notas, quando um usuário clica em “Arquivo->Abrir”, uma caixa de diálogo é exibida para permitir que o usuário abra um arquivo. Os itens de menu são processados ​​na mensagem WM_CREATE do procedimento de janela assim:

LRESULT CALLBACK winproc(HWND hwnd, UINT wm, WPARAM wp, LPARAM lp)
{
    switch (wm) {
        case WM_COMMAND:
            switch (LOWORD(wp) {
                case ID_FILE_OPEN:
                    /* show file open dialog */
                    break;
                case ID_FILE_NEW:
                    /* create new instance */
                    break;
            }
            return 0;
   }
   return DefWindowProc(hwnd, wm, wp, lp);
}

WM_CREATE #

Uma mensagem WM_CREATE é enviada para seu procedimento de janela durante a chamada CreateWindowEx da janela. O argumento lp contém um ponteiro para um CREATESTRUCT que contém os argumentos passados ​​para CreateWindowEx. Se um aplicativo retornar 0 de WM_CREATE, a janela será criada. Se um aplicativo retornar -1, a criação será cancelada.

LRESULT CALLBACK winproc(HWND hwnd, UINT wm, WPARAM wp, LPARAM lp)
{
    switch (wm) {
        case WM_CREATE:
            CREATESTRUCT *cs = (CREATESTRUCT *) lp;
            if (MessageBox(hwnd, 
                "Do you want to continue creating the window?", "", MB_YESNO)  
                                         == IDYES) {
                /* create window controls */
                return 0;
            }
            /* cancel creation */
            return -1;
    }
    return DefWindowProc(hwnd, wm, wp, lp);
}

WM_SIZE #

Esta mensagem é enviada para o procedimento de janela da janela depois que seu tamanho foi alterado. O motivo mais comum para lidar com essa mensagem é ajustar a posição de qualquer janela filho. Por exemplo, no Bloco de Notas, quando a janela é redimensionada, a janela filho (controle de edição) também é redimensionada. Retorne 0 se você manipular esta mensagem.

Argumento Valor
wp Uma das constantes de dimensionamento de janela.
lp LOWORD(lp) é a nova largura da área do cliente
HIWORD(lp) é a nova altura da área do cliente.		 
LRESULT CALLBACK winproc(HWND hwnd, UINT wm, WPARAM wp, LPARAM lp)
{
    switch (wm) {
        case WM_SIZE:
            /* hwndEdit is the handle of the edit control window */
            MoveWindow(hwndEdit, 0, 0, LOWORD(lp), HIWORD(lp), TRUE);
            return 0;
   }
   return DefWindowProc(hwnd, wm, wp, lp);
}
