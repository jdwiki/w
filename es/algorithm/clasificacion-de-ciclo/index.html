    item = array[cycleStart]
    pos = cycleStart
    for i from cycleStart + 1 to length(array) - 1
        if array[i] < item:
            pos += 1
    if pos == cycleStart:
        continue
    while item == array[pos]:
        pos += 1
    array[pos], item = item, array[pos]
    writes += 1
    while pos != cycleStart:

            if array[i] < item:
                pos += 1
        while item == array[pos]:
            pos += 1
        array[pos], item = item, array[pos]
        writes += 1

Ciclo de clasificación Información básica #

Ordenación cíclica es un algoritmo de clasificación que utiliza clasificación de comparación que es teóricamente óptimo en términos del número total de escrituras en la matriz original, a diferencia de cualquier otro algoritmo de clasificación en el lugar. Cycle sort es un algoritmo de clasificación inestable. Se basa en la idea de permutación en la que las permutaciones se factorizan en ciclos, que giran individualmente y devuelven una salida ordenada.

Espacio auxiliar: O(1)
Complejidad de tiempo: O(n^2)

Implementación de C# #

public class CycleSort
{
    public static void SortCycle(int[] input)
    {
        for (var i = 0; i < input.Length; i++)
        {
            var item = input[i];
            var position = i;
            do
            {
                var k = input.Where((t, j) => position != j && t < item).Count();
                if (position == k) continue;
                while (position != k && item == input[k])
                {
                    k++;
                }
                var temp = input[k];
                input[k] = item;
                item = temp;
                position = k;
            } while (position != i);
        }
    }

    public static int[] Main(int[] input)
    {
        SortCycle(input);
        return input;
    }
}
