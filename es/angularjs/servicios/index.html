Creando un servicio usando angular.factory #

Primero defina el servicio (en este caso usa el patrón de fábrica):

.factory('dataService', function() {
    var dataObject = {};
    var service = {
        // define the getter method
        get data() {
            return dataObject;
        },
        // define the setter method
        set data(value) {
            dataObject = value || {};
        }
    };
    // return the "service" object to expose the getter/setter
    return service;
})

Ahora puede usar el servicio para compartir datos entre controladores:

.controller('controllerOne', function(dataService) {
    // create a local reference to the dataService
    this.dataService = dataService;
    // create an object to store
    var someObject = {
        name: 'SomeObject',
        value: 1
    };
    // store the object
    this.dataService.data = someObject;
})

.controller('controllerTwo', function(dataService) {
    // create a local reference to the dataService
    this.dataService = dataService;
    // this will automatically update with any changes to the shared data object
    this.objectFromControllerOne = this.dataService.data;
})

Cómo crear un Servicio #

angular.module("app")
  .service("counterService", function(){

    var service = {
        number: 0
    };
    
    return service;
   });

Cómo utilizar un servicio #

    angular.module("app")
        
        // Custom services are injected just like Angular's built-in services
        .controller("step1Controller", ['counterService', '$scope', function(counterService, $scope) {
            counterService.number++;
            // bind to object (by reference), not to value, for automatic sync
            $scope.counter = counterService;
        })

En la plantilla que usa este controlador, escribiría:

// editable
<input ng-model="counter.number" />

o

// read-only
<span ng-bind="counter.number"></span>

Por supuesto, en el código real, interactuaría con el servicio utilizando métodos en el controlador, que a su vez delegaría al servicio. El ejemplo anterior simplemente incrementa el valor del contador cada vez que se usa el controlador en una plantilla.

Los servicios en Angularjs son singletons:

Los servicios son objetos únicos que se instancian solo una vez por aplicación (por el $injector) y se cargan de forma diferida (se crean solo cuando es necesario).

Un singleton es una clase que solo permite que una instancia de sí misma sea creado - y brinda un acceso simple y fácil a dicha instancia. Como se indica aquí

$sce: desinfecte y represente contenido y recursos en plantillas #

$sce (“Strict Contextual Escaping”) es un servicio angular incorporado que desinfecta automáticamente el contenido y las fuentes internas en las plantillas.

inyectar fuentes externas y HTML sin formato en la plantilla requiere el ajuste manual de $sce.

En este ejemplo, crearemos un filtro de saneamiento $sce simple:`.

Demostración

.filter('sanitizer', ['$sce', [function($sce) {
     return function(content) {
          return $sce.trustAsResourceUrl(content);
      };
}]);

Uso en la plantilla

<div ng-repeat="item in items">
    
    // Sanitize external sources
    <ifrmae ng-src="{{item.youtube_url | sanitizer}}">
    
    // Sanitaize and render HTML 
    <div ng-bind-html="{{item.raw_html_content| sanitizer}}"></div>

</div>

Diferencia entre Servicio y Fábrica #

1) Servicios

Un servicio es una función de constructor que se invoca una vez en tiempo de ejecución con nuevo, tal como lo haríamos con javascript simple con solo diferencia que AngularJs está llamando al nuevo detrás de escena.

Hay una regla general para recordar en caso de servicios

  1. Los servicios son constructores que se llaman con nuevo

Veamos un ejemplo simple en el que registraríamos un servicio que usa el servicio $http para obtener los detalles del estudiante y usarlo en el controlador.

function StudentDetailsService($http) {
  this.getStudentDetails = function getStudentDetails() {
    return $http.get('/details');
  };
}

angular.module('myapp').service('StudentDetailsService', StudentDetailsService);

Simplemente inyectamos este servicio en el controlador.

function StudentController(StudentDetailsService) {
  StudentDetailsService.getStudentDetails().then(function (response) {
      // handle response
    });
}
angular.module('app').controller('StudentController', StudentController);

¿Cuándo usar?

Use .service() donde quiera usar un constructor. Por lo general, se usa para crear API públicas como getStudentDetails(). pero si tu no quiere usar un constructor y desea usar un patrón de API simple en su lugar, entonces no hay mucha flexibilidad en .service().

2) Fábrica

Aunque podemos lograr todas las cosas usando .factory() que haríamos usando .services(), no hace que .factory() sea “igual que” .service(). Es mucho más poderoso y flexible que .service()

Un .factory() es un patrón de diseño que se usa para devolver un valor.

Hay dos reglas básicas para recordar en el caso de las fábricas

  1. Valores devueltos por las fábricas
  2. Las fábricas (pueden) crear objetos (Cualquier object)

Veamos algunos ejemplos de lo que podemos hacer usando .factory()

Devolver objetos literales

Veamos un ejemplo en el que se usa factory para devolver un objeto usando un patrón de módulo de revelación básico

function StudentDetailsService($http) {
  function getStudentDetails() {
    return $http.get('/details');
  }
  return {
    getStudentDetails: getStudentDetails
  };
}

angular.module('myapp').factory('StudentDetailsService', StudentDetailsService);

Uso dentro de un controlador

function StudentController(StudentDetailsService) {
  StudentDetailsService.getStudentDetails().then(function (response) {
      // handle response
    });
}
angular.module('app').controller('StudentController', StudentController);

Cierres de devolución

¿Qué es un cierre?

Los cierres son funciones que se refieren a variables que se usan localmente, PERO definidas en un ámbito envolvente.

El siguiente es un ejemplo de un cierre

function closureFunction(name) {
  function innerClosureFunction(age) { // innerClosureFunction() is the inner function, a closure
    // Here you can manipulate 'age' AND 'name' variables both
  };
};

La parte “maravillosa” es que puede acceder al nombre que está en el ámbito principal.

Usemos el ejemplo de cierre anterior dentro de .factory()

function StudentDetailsService($http) {
  function closureFunction(name) {
  function innerClosureFunction(age) {
    // Here you can manipulate 'age' AND 'name' variables
    };
  };
};

angular.module('myapp').factory('StudentDetailsService', StudentDetailsService);

Uso dentro de un controlador

function StudentController(StudentDetailsService) {
  var myClosure = StudentDetailsService('Student Name'); // This now HAS the innerClosureFunction()
  var callMyClosure = myClosure(24); // This calls the innerClosureFunction()
};

angular.module('app').controller('StudentController', StudentController);

Creación de constructores/instancias

.service() crea constructores con una llamada a nuevo como se ve arriba. .factory() también puede crear constructores con una llamada a nuevo

Veamos un ejemplo de cómo lograr esto.

function StudentDetailsService($http) {
  function Student() {
    this.age = function () {
        return 'This is my age';
    };
  }
  Student.prototype.address = function () {
        return 'This is my address';
  };
  return Student;
};

angular.module('myapp').factory('StudentDetailsService', StudentDetailsService);

Uso dentro de un controlador

function StudentController(StudentDetailsService) {
  var newStudent = new StudentDetailsService();
  
  //Now the instance has been created. Its properties can be accessed.

 newStudent.age();
 newStudent.address();
  
};

angular.module('app').controller('StudentController', StudentController);

Cómo crear un servicio con dependencias usando ‘sintaxis de matriz’ #

angular.module("app")
  .service("counterService", ["fooService", "barService", function(anotherService, barService){

    var service = {
        number: 0,
        foo: function () {
            return fooService.bazMethod(); // Use of 'fooService'
        },
        bar: function () {
            return barService.bazMethod(); // Use of 'barService'
        }
    };
    
    return service;
   }]);

Registro de un servicio #

La forma más común y flexible de crear un servicio utiliza la fábrica de API angular.module:

angular.module('myApp.services', []).factory('githubService', function() {       
   var serviceInstance = {};
   // Our first service
   return serviceInstance;
});

La función de fábrica de servicios puede ser una función o una matriz, al igual que la forma en que creamos controladores:

// Creating the factory through using the
// bracket notation
angular.module('myApp.services', [])
.factory('githubService', [function($http) {
}]);

Para exponer un método en nuestro servicio, podemos colocarlo como un atributo en el objeto del servicio.

angular.module('myApp.services', [])
    .factory('githubService', function($http) {
        var githubUrl = 'https://api.github.com';
        var runUserRequest = function(username, path) {
        // Return the promise from the $http service
        // that calls the Github API using JSONP
        return $http({
            method: 'JSONP',
            url: githubUrl + '/users/' +
            username + '/' +
            path + '?callback=JSON_CALLBACK'
        });
    }
// Return the service object with a single function
// events
return {
    events: function(username) {
    return runUserRequest(username, 'events');
    }
};
