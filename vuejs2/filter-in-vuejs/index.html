                    }
                    console.log(this.bindList)
                }
            }
        },

            reverseH1_:function(){
                return this.h1.split('').reverse().join('');
            },

                get: function(){
                    return this.price + this.total
                },
                set : function(newvalue){
                    this.price = "#"
                }
            }

    })
